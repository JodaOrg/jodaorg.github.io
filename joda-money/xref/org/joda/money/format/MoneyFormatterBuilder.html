193     /**
194      * Appends the specified printer and parser to this builder.
195      * <p>
196      * Either the printer or parser must be non-null.
197      * 
198      * @param printer  the printer to append, null makes the formatter unable to print
199      * @param parser  the parser to append, null makes the formatter unable to parse
200      * @return this for chaining, never null
201      */
202     private MoneyFormatterBuilder appendInternal(MoneyPrinter printer, MoneyParser parser) {
203         printers.add(printer);
204         parsers.add(parser);
205         return this;
206     }

225     /**
226      * Builds the formatter from the builder setting the locale.
227      * <p>
228      * Once the builder is in the correct state it must be converted to a
229      * {@code MoneyFormatter} to be used. Calling this method does not
230      * change the state of this instance, so it can still be used.
231      * <p>
232      * This method uses the specified locale within the returned formatter.
233      * It can be changed by calling {@link MoneyFormatter#withLocale(Locale)}.
234      * 
235      * @param locale  the initial locale for the formatter, not null
236      * @return the formatter built from this builder, never null
237      */
238     @SuppressWarnings("cast")
239     public MoneyFormatter toFormatter(Locale locale) {
240         MoneyFormatter.checkNotNull(locale, "Locale must not be null");
241         MoneyPrinter[] printersCopy = (MoneyPrinter[]) printers.toArray(new MoneyPrinter[printers.size()]);
242         MoneyParser[] parsersCopy = (MoneyParser[]) parsers.toArray(new MoneyParser[parsers.size()]);
243         return new MoneyFormatter(locale, printersCopy, parsersCopy);
244     }

331     /**
332      * Handles the singleton outputs.
333      */
334     private static enum Singletons implements MoneyPrinter, MoneyParser {
335         CODE("${code}") {
336             @Override
337             public void print(MoneyPrintContext context, Appendable appendable, BigMoney money) throws IOException {
338                 appendable.append(money.getCurrencyUnit().getCode());
339             }
340             @Override
341             public void parse(MoneyParseContext context) {
342                 int endPos = context.getIndex() + 3;
343                 if (endPos > context.getTextLength()) {
344                     context.setError();
345                 } else {
346                     String code = context.getTextSubstring(context.getIndex(), endPos);
347                     try {
348                         context.setCurrency(CurrencyUnit.of(code));
349                         context.setIndex(endPos);
350                     } catch (IllegalCurrencyException ex) {
351                         context.setError();
352                     }
353                 }
354             }
355         },
356         NUMERIC_3_CODE("${numeric3Code}") {
357             @Override
358             public void print(MoneyPrintContext context, Appendable appendable, BigMoney money) throws IOException {
359                 appendable.append(money.getCurrencyUnit().getNumeric3Code());
360             }
361             @Override
362             public void parse(MoneyParseContext context) {
363                 int endPos = context.getIndex() + 3;
364                 if (endPos > context.getTextLength()) {
365                     context.setError();
366                 } else {
367                     String code = context.getTextSubstring(context.getIndex(), endPos);
368                     try {
369                         context.setCurrency(CurrencyUnit.ofNumericCode(code));
370                         context.setIndex(endPos);
371                     } catch (IllegalCurrencyException ex) {
372                         context.setError();
373                     }
374                 }
375             }
376         },
377         NUMERIC_CODE("${numericCode}") {
378             @Override
379             public void print(MoneyPrintContext context, Appendable appendable, BigMoney money) throws IOException {
380                 appendable.append(Integer.toString(money.getCurrencyUnit().getNumericCode()));
381             }
382             @Override
383             public void parse(MoneyParseContext context) {
384                 int count = 0;
385                 for ( ; count < 3 && context.getIndex() + count < context.getTextLength(); count++) {
386                     char ch = context.getText().charAt(context.getIndex() + count);
387                     if (ch < '0' || ch > '9') {
388                         break;
389                     }
390                 }
391                 int endPos = context.getIndex() + count;
392                 String code = context.getTextSubstring(context.getIndex(), endPos);
393                 try {
394                     context.setCurrency(CurrencyUnit.ofNumericCode(code));
395                     context.setIndex(endPos);
396                 } catch (IllegalCurrencyException ex) {
397                     context.setError();
398                 }
399             }
400         };
401         private final String toString;
402         private Singletons(String toString) {
403             this.toString = toString;
404         }
405         @Override
406         public String toString() {
407             return toString;
408         }
409     }
