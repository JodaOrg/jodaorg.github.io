29  /**
30   * An amount of money with the standard decimal places defined by the currency.
31   * <p>
32   * This class represents a quantity of money, stored as a {@code BigDecimal} amount
33   * in a single {@link CurrencyUnit currency}.
34   * <p>
35   * Every currency has a certain standard number of decimal places.
36   * This is typically 2 (Euro, British Pound, US Dollar) but might be
37   * 0 (Japanese Yen), 1 (Vietnamese Dong) or 3 (Bahrain Dinar).
38   * The {@code Money} class is fixed to this number of decimal places.
39   * <p>
40   * For example, US dollars has a standard number of decimal places of 2.
41   * The major units are dollars. The minor units are cents, 100 to the dollar.
42   * This class does not allow calculations on fractions of a cent.
43   * <p>
44   * This class is immutable and thread-safe.
45   */
46  public final class Money implements BigMoneyProvider, Comparable<BigMoneyProvider>, Serializable {

58      //-----------------------------------------------------------------------
59      /**
60       * Obtains an instance of {@code Money} from a {@code BigDecimal}.
61       * <p>
62       * This allows you to create an instance with a specific currency and amount.
63       * No rounding is performed on the amount, so it must have a scale compatible
64       * with the currency.
65       *
66       * @param currency  the currency, not null
67       * @param amount  the amount of money, not null
68       * @return the new instance, never null
69       * @throws ArithmeticException if the scale exceeds the currency scale
70       */
71      public static Money of(CurrencyUnit currency, BigDecimal amount) {
72          MoneyUtils.checkNotNull(currency, "Currency must not be null");
73          MoneyUtils.checkNotNull(amount, "Amount must not be null");
74          if (amount.scale() > currency.getDecimalPlaces()) {
75              throw new ArithmeticException("Scale of amount " + amount + " is greater than the scale of the currency " + currency);
76          }
77          return Money.of(currency, amount, RoundingMode.UNNECESSARY);
78      }
79  
80      /**
81       * Obtains an instance of {@code Money} from a {@code BigDecimal}, rounding as necessary.
82       * <p>
83       * This allows you to create an instance with a specific currency and amount.
84       * If the amount has a scale in excess of the scale of the currency then the excess
85       * fractional digits are rounded using the rounding mode.
86       *
87       * @param currency  the currency, not null
88       * @param amount  the amount of money, not null
89       * @param roundingMode  the rounding mode to use, not null
90       * @return the new instance, never null
91       * @throws ArithmeticException if the rounding fails
92       */
93      public static Money of(CurrencyUnit currency, BigDecimal amount, RoundingMode roundingMode) {
94          MoneyUtils.checkNotNull(currency, "CurrencyUnit must not be null");
95          MoneyUtils.checkNotNull(amount, "Amount must not be null");
96          MoneyUtils.checkNotNull(roundingMode, "RoundingMode must not be null");
97          amount = amount.setScale(currency.getDecimalPlaces(), roundingMode);
98          return new Money(BigMoney.of(currency, amount));
99      }

234     /**
235      * Obtains an instance of {@code Money} as the total value of an array.
236      * <p>
237      * The array must contain at least one monetary value.
238      * Subsequent amounts are added as though using {@link #plus(Money)}.
239      * All amounts must be in the same currency.
240      * 
241      * @param monies  the monetary values to total, not empty, no null elements, not null
242      * @return the total, never null
243      * @throws IllegalArgumentException if the array is empty
244      * @throws CurrencyMismatchException if the currencies differ
245      */
246     public static Money total(Money... monies) {
247         MoneyUtils.checkNotNull(monies, "Money array must not be null");
248         if (monies.length == 0) {
249             throw new IllegalArgumentException("Money array must not be empty");
250         }
251         Money total = monies[0];
252         MoneyUtils.checkNotNull(total, "Money arary must not contain null entries");
253         for (int i = 1; i < monies.length; i++) {
254             total = total.plus(monies[i]);
255         }
256         return total;
257     }
258 
259     /**
260      * Obtains an instance of {@code Money} as the total value of a collection.
261      * <p>
262      * The iterable must provide at least one monetary value.
263      * Subsequent amounts are added as though using {@link #plus(Money)}.
264      * All amounts must be in the same currency.
265      * 
266      * @param monies  the monetary values to total, not empty, no null elements, not null
267      * @return the total, never null
268      * @throws IllegalArgumentException if the iterable is empty
269      * @throws CurrencyMismatchException if the currencies differ
270      */
271     public static Money total(Iterable<Money> monies) {
272         MoneyUtils.checkNotNull(monies, "Money iterator must not be null");
273         Iterator<Money> it = monies.iterator();
274         if (it.hasNext() == false) {
275             throw new IllegalArgumentException("Money iterator must not be empty");
276         }
277         Money total = it.next();
278         MoneyUtils.checkNotNull(total, "Money iterator must not contain null entries");
279         while (it.hasNext()) {
280             total = total.plus(it.next());
281         }
282         return total;
283     }

320     /**
321      * Parses an instance of {@code Money} from a string.
322      * <p>
323      * The string format is '$currencyCode $amount' where there may be
324      * zero to many spaces between the two parts.
325      * The currency code must be a valid three letter currency.
326      * The amount must match the regular expression {@code [+-]?[0-9]*[.]?[0-9]*}.
327      * This matches the output from {@link #toString()}.
328      * <p>
329      * For example, {@code parse("USD 25")} creates the instance {@code USD 25.00}
330      * while {@code parse("USD 25.95")} creates the instance {@code USD 25.95}.
331      *
332      * @param moneyStr  the money string to parse, not null
333      * @return the parsed instance, never null
334      * @throws IllegalArgumentException if the string is malformed
335      * @throws ArithmeticException if the amount is too large
336      */
337     @FromString
338     public static Money parse(String moneyStr) {
339         return Money.of(BigMoney.parse(moneyStr));
340     }

342     //-----------------------------------------------------------------------
343     /**
344      * Ensures that a {@code Money} is not {@code null}.
345      * <p>
346      * If the input money is not {@code null}, then it is returned, providing
347      * that the currency matches the specified currency.
348      * If the input money is {@code null}, then zero money in the currency is returned.
349      * 
350      * @param money  the monetary value to check, may be null
351      * @param currency  the currency to use, not null
352      * @return the input money or zero in the specified currency, never null
353      * @throws CurrencyMismatchException if the input money is non-null and the currencies differ
354      */
355     public static Money nonNull(Money money, CurrencyUnit currency) {
356         if (money == null) {
357             return zero(currency);
358         }
359         if (money.getCurrencyUnit().equals(currency) == false) {
360             MoneyUtils.checkNotNull(currency, "Currency must not be null");
361             throw new CurrencyMismatchException(money.getCurrencyUnit(), currency);
362         }
363         return money;
364     }
