33  /**
34   * BaseSingleFieldPeriod is an abstract implementation of ReadablePeriod that
35   * manages a single duration field, such as days or minutes.
36   * <p>
37   * This class should generally not be used directly by API users.
38   * The {@link ReadablePeriod} interface should be used when different 
39   * kinds of period objects are to be referenced.
40   * <p>
41   * BaseSingleFieldPeriod subclasses may be mutable and not thread-safe.
42   *
43   * @author Stephen Colebourne
44   * @since 1.4
45   */
46  public abstract class BaseSingleFieldPeriod
47          implements ReadablePeriod, Comparable<BaseSingleFieldPeriod>, Serializable {

51      /** The start of 1972. */
52      private static final long START_1972 = 2L * 365L * 86400L * 1000L;

54      /** The period in the units of this period. */
55      private volatile int iPeriod;

57      //-----------------------------------------------------------------------
58      /**
59       * Calculates the number of whole units between the two specified datetimes.
60       *
61       * @param start  the start instant, validated to not be null
62       * @param end  the end instant, validated to not be null
63       * @param field  the field type to use, must not be null
64       * @return the period
65       * @throws IllegalArgumentException if the instants are null or invalid
66       */
67      protected static int between(ReadableInstant start, ReadableInstant end, DurationFieldType field) {
68          if (start == null || end == null) {
69              throw new IllegalArgumentException("ReadableInstant objects must not be null");
70          }
71          Chronology chrono = DateTimeUtils.getInstantChronology(start);
72          int amount = field.getField(chrono).getDifference(end.getMillis(), start.getMillis());
73          return amount;
74      }

76      //-----------------------------------------------------------------------
77      /**
78       * Calculates the number of whole units between the two specified partial datetimes.
79       * <p>
80       * The two partials must contain the same fields, for example you can specify
81       * two <code>LocalDate</code> objects.
82       *
83       * @param start  the start partial date, validated to not be null
84       * @param end  the end partial date, validated to not be null
85       * @param zeroInstance  the zero instance constant, must not be null
86       * @return the period
87       * @throws IllegalArgumentException if the partials are null or invalid
88       */
89      protected static int between(ReadablePartial start, ReadablePartial end, ReadablePeriod zeroInstance) {
90          if (start == null || end == null) {
91              throw new IllegalArgumentException("ReadablePartial objects must not be null");
92          }
93          if (start.size() != end.size()) {
94              throw new IllegalArgumentException("ReadablePartial objects must have the same set of fields");
95          }
96          for (int i = 0, isize = start.size(); i < isize; i++) {
97              if (start.getFieldType(i) != end.getFieldType(i)) {
98                  throw new IllegalArgumentException("ReadablePartial objects must have the same set of fields");
99              }
100         }
101         if (DateTimeUtils.isContiguous(start) == false) {
102             throw new IllegalArgumentException("ReadablePartial objects must be contiguous");
103         }
104         Chronology chrono = DateTimeUtils.getChronology(start.getChronology()).withUTC();
105         int[] values = chrono.get(zeroInstance, chrono.set(start, START_1972), chrono.set(end, START_1972));
106         return values[0];
107     }

109     /**
110      * Creates a new instance representing the number of complete standard length units
111      * in the specified period.
112      * <p>
113      * This factory method converts all fields from the period to hours using standardised
114      * durations for each field. Only those fields which have a precise duration in
115      * the ISO UTC chronology can be converted.
116      * <ul>
117      * <li>One week consists of 7 days.
118      * <li>One day consists of 24 hours.
119      * <li>One hour consists of 60 minutes.
120      * <li>One minute consists of 60 seconds.
121      * <li>One second consists of 1000 milliseconds.
122      * </ul>
123      * Months and Years are imprecise and periods containing these values cannot be converted.
124      *
125      * @param period  the period to get the number of hours from, must not be null
126      * @param millisPerUnit  the number of milliseconds in one standard unit of this period
127      * @throws IllegalArgumentException if the period contains imprecise duration values
128      */
129     protected static int standardPeriodIn(ReadablePeriod period, long millisPerUnit) {
130         if (period == null) {
131             return 0;
132         }
133         Chronology iso = ISOChronology.getInstanceUTC();
134         long duration = 0L;
135         for (int i = 0; i < period.size(); i++) {
136             int value = period.getValue(i);
137             if (value != 0) {
138                 DurationField field = period.getFieldType(i).getField(iso);
139                 if (field.isPrecise() == false) {
140                     throw new IllegalArgumentException(
141                             "Cannot convert period to duration as " + field.getName() +
142                             " is not precise in the period " + period);
143                 }
144                 duration = FieldUtils.safeAdd(duration, FieldUtils.safeMultiply(field.getUnitMillis(), value));
145             }
146         }
147         return FieldUtils.safeToInt(duration / millisPerUnit);
148     }

291     /**
292      * Compares this object with the specified object for equality based on the
293      * value of each field. All ReadablePeriod instances are accepted, but only
294      * those with a matching <code>PeriodType</code> can return true.
295      *
296      * @param period  a readable period to check against
297      * @return true if all the field values are equal, false if
298      *  not or the period is null or of an incorrect type

300     public boolean equals(Object period) {
301         if (this == period) {
302             return true;
303         }
304         if (period instanceof ReadablePeriod == false) {
305             return false;
306         }
307         ReadablePeriod other = (ReadablePeriod) period;
308         return (other.getPeriodType() == getPeriodType() && other.getValue(0) == getValue());
309     }

323     /**
324      * Compares this period to another object of the same class.
325      *
326      * @param other  the other period, must not be null
327      * @return zero if equal, positive if greater, negative if less
328      * @throws NullPointerException if the other period is null
329      * @throws ClassCastException if the other period is of a different type
330      */
331     public int compareTo(BaseSingleFieldPeriod other) {
332         if (other.getClass() != getClass()) {
333             throw new ClassCastException(getClass() + " cannot be compared to " + other.getClass());
334         }
335         int otherValue = other.getValue();
336         int thisValue = getValue();
337         if (thisValue > otherValue) {
338             return 1;
339         }
340         if (thisValue < otherValue) {
341             return -1;
342         }
343         return 0;
344     }
