28  /**
29   * AbstractDateTime provides the common behaviour for datetime classes.
30   * <p>
31   * This class should generally not be used directly by API users.
32   * The {@link ReadableDateTime} interface should be used when different 
33   * kinds of date/time objects are to be referenced.
34   * <p>
35   * Whenever you want to implement <code>ReadableDateTime</code> you should
36   * extend this class.
37   * <p>
38   * AbstractDateTime subclasses may be mutable and not thread-safe.
39   *
40   * @author Brian S O'Neill
41   * @author Stephen Colebourne
42   * @since 1.0
43   */
44  public abstract class AbstractDateTime
45          extends AbstractInstant
46          implements ReadableDateTime {

55      //-----------------------------------------------------------------------
56      /**
57       * Get the value of one of the fields of a datetime.
58       * <p>
59       * This method uses the chronology of the datetime to obtain the value.
60       * It is essentially a generic way of calling one of the get methods.
61       *
62       * @param type  a field type, usually obtained from DateTimeFieldType
63       * @return the value of that field
64       * @throws IllegalArgumentException if the field type is null
65       */
66      public int get(DateTimeFieldType type) {
67          if (type == null) {
68              throw new IllegalArgumentException("The DateTimeFieldType must not be null");
69          }
70          return type.getField(getChronology()).get(getMillis());
71      }

252     //-----------------------------------------------------------------------
253     /**
254      * Get the date time as a <code>java.util.Calendar</code>, assigning
255      * exactly the same millisecond instant.
256      * The locale is passed in, enabling Calendar to select the correct
257      * localized subclass.
258      * <p>
259      * The JDK and Joda-Time both have time zone implementations and these
260      * differ in accuracy. Joda-Time's implementation is generally more up to
261      * date and thus more accurate - for example JDK1.3 has no historical data.
262      * The effect of this is that the field values of the <code>Calendar</code>
263      * may differ from those of this object, even though the milliseond value
264      * is the same. Most of the time this just means that the JDK field values
265      * are wrong, as our time zone information is more up to date.
266      *
267      * @param locale  the locale to get the Calendar for, or default if null
268      * @return a localized Calendar initialised with this datetime
269      */
270     public Calendar toCalendar(Locale locale) {
271         if (locale == null) {
272             locale = Locale.getDefault();
273         }
274         DateTimeZone zone = getZone();
275         Calendar cal = Calendar.getInstance(zone.toTimeZone(), locale);
276         cal.setTime(toDate());
277         return cal;
278     }

281      * Get the date time as a <code>java.util.GregorianCalendar</code>,
282      * assigning exactly the same millisecond instant.
283      * <p>
284      * The JDK and Joda-Time both have time zone implementations and these
285      * differ in accuracy. Joda-Time's implementation is generally more up to
286      * date and thus more accurate - for example JDK1.3 has no historical data.
287      * The effect of this is that the field values of the <code>Calendar</code>
288      * may differ from those of this object, even though the milliseond value
289      * is the same. Most of the time this just means that the JDK field values
290      * are wrong, as our time zone information is more up to date.
291      *
292      * @return a GregorianCalendar initialised with this datetime
293      */
294     public GregorianCalendar toGregorianCalendar() {
295         DateTimeZone zone = getZone();
296         GregorianCalendar cal = new GregorianCalendar(zone.toTimeZone());
297         cal.setTime(toDate());
298         return cal;
299     }

302     /**
303      * Output the date time in ISO8601 format (yyyy-MM-ddTHH:mm:ss.SSSZZ).
304      * <p>
305      * Note that this method does not output the chronology or time-zone.
306      * This can be confusing, as the equals and hashCode methods use both
307      * chronology and time-zone. If two objects are not {@code equal} but have the
308      * same {@code toString} then either the chronology or time-zone differs.
309      * 
310      * @return ISO8601 time formatted string, not null
311      */
312     @ToString
313     public String toString() {
314         return super.toString();
315     }
316 
317     /**
318      * Output the instant using the specified format pattern.
319      *
320      * @param pattern  the pattern specification, null means use <code>toString</code>
321      * @return the formatted string, not null
322      * @see  org.joda.time.format.DateTimeFormat
323      */
324     public String toString(String pattern) {
325         if (pattern == null) {
326             return toString();
327         }
328         return DateTimeFormat.forPattern(pattern).print(this);
329     }
330 
331     /**
332      * Output the instant using the specified format pattern.
333      *
334      * @param pattern  the pattern specification, null means use <code>toString</code>
335      * @param locale  Locale to use, null means default
336      * @return the formatted string, not null
337      * @see  org.joda.time.format.DateTimeFormat
338      */
339     public String toString(String pattern, Locale locale) throws IllegalArgumentException {
340         if (pattern == null) {
341             return toString();
342         }
343         return DateTimeFormat.forPattern(pattern).withLocale(locale).print(this);
344     }
