16  package org.joda.time.field;
17  
18  import org.joda.time.DateTimeFieldType;
19  import org.joda.time.DurationField;
20  
21  /**
22   * Precise datetime field, which has a precise unit duration field.
23   * <p>
24   * PreciseDurationDateTimeField is thread-safe and immutable, and its
25   * subclasses must be as well.
26   *
27   * @author Brian S O'Neill
28   * @since 1.0
29   */
30  public abstract class PreciseDurationDateTimeField extends BaseDateTimeField {
31  
32      @SuppressWarnings("unused")
33      private static final long serialVersionUID = 5004523158306266035L;
34  
35      /** The fractional unit in millis */
36      final long iUnitMillis;
37  
38      private final DurationField iUnitField;
39  
40      /**
41       * Constructor.
42       * 
43       * @param type  the field type
44       * @param unit  precise unit duration, like "days()".
45       * @throws IllegalArgumentException if duration field is imprecise
46       * @throws IllegalArgumentException if unit milliseconds is less than one
47       */
48      public PreciseDurationDateTimeField(DateTimeFieldType type, DurationField unit) {
49          super(type);
50  
51          if (!unit.isPrecise()) {
52              throw new IllegalArgumentException("Unit duration field must be precise");
53          }
54  
55          iUnitMillis = unit.getUnitMillis();
56          if (iUnitMillis < 1) {
57              throw new IllegalArgumentException("The unit milliseconds must be at least 1");
58          }
59  
60          iUnitField = unit;
61      }
62  
63      /**
64       * Returns false by default.
65       */
66      public boolean isLenient() {
67          return false;
68      }
69  
70      /**
71       * Set the specified amount of units to the specified time instant.
72       * 
73       * @param instant  the milliseconds from 1970-01-01T00:00:00Z to set in
74       * @param value  value of units to set.
75       * @return the updated time instant.
76       * @throws IllegalArgumentException if value is too large or too small.
77       */
78      public long set(long instant, int value) {
79          FieldUtils.verifyValueBounds(this, value, getMinimumValue(),
80                                       getMaximumValueForSet(instant, value));
81          return instant + (value - get(instant)) * iUnitMillis;
82      }
83  
84      /**
85       * This method assumes that this field is properly rounded on
86       * 1970-01-01T00:00:00. If the rounding alignment differs, override this
87       * method as follows:
88       * <pre>
89       * return super.roundFloor(instant + ALIGNMENT_MILLIS) - ALIGNMENT_MILLIS;
90       * </pre>
91       */
92      public long roundFloor(long instant) {
93          if (instant >= 0) {
94              return instant - instant % iUnitMillis;
95          } else {
96              instant += 1;
97              return instant - instant % iUnitMillis - iUnitMillis;
98          }
99      }
100 
101     /**
102      * This method assumes that this field is properly rounded on
103      * 1970-01-01T00:00:00. If the rounding alignment differs, override this
104      * method as follows:
105      * <pre>
106      * return super.roundCeiling(instant + ALIGNMENT_MILLIS) - ALIGNMENT_MILLIS;
107      * </pre>
108      */
109     public long roundCeiling(long instant) {
110         if (instant > 0) {
111             instant -= 1;
112             return instant - instant % iUnitMillis + iUnitMillis;
113         } else {
114             return instant - instant % iUnitMillis;
115         }
116     }
117 
118     /**
119      * This method assumes that this field is properly rounded on
120      * 1970-01-01T00:00:00. If the rounding alignment differs, override this
121      * method as follows:
122      * <pre>
123      * return super.remainder(instant + ALIGNMENT_MILLIS);
124      * </pre>
125      */
126     public long remainder(long instant) {
127         if (instant >= 0) {
128             return instant % iUnitMillis;
129         } else {
130             return (instant + 1) % iUnitMillis + iUnitMillis - 1;
131         }
132     }
133 
134     /**
135      * Returns the duration per unit value of this field. For example, if this
136      * field represents "minute of hour", then the duration field is minutes.
137      *
138      * @return the duration of this field, or UnsupportedDurationField if field
139      * has no duration
140      */
141     public DurationField getDurationField() {
142         return iUnitField;
143     }
144 
145     /**
146      * Get the minimum value for the field.
147      * 
148      * @return the minimum value
149      */
150     public int getMinimumValue() {
151         return 0;
152     }
153 
154     public final long getUnitMillis() {
155         return iUnitMillis;
156     }
157 
158     /**
159      * Called by the set method to get the maximum allowed value. By default,
160      * returns getMaximumValue(instant). Override to provide a faster
161      * implementation.
162      */
163     protected int getMaximumValueForSet(long instant, int value) {
164         return getMaximumValue(instant);
165     }
166 
167 }
