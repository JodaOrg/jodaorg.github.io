45  /**
46   * DateTimeZone represents a time zone.
47   * <p>
48   * A time zone is a system of rules to convert time from one geographic 
49   * location to another. For example, Paris, France is one hour ahead of
50   * London, England. Thus when it is 10:00 in London, it is 11:00 in Paris.
51   * <p>
52   * All time zone rules are expressed, for historical reasons, relative to
53   * Greenwich, London. Local time in Greenwich is referred to as Greenwich Mean
54   * Time (GMT).  This is similar, but not precisely identical, to Universal 
55   * Coordinated Time, or UTC. This library only uses the term UTC.
56   * <p>
57   * Using this system, America/Los_Angeles is expressed as UTC-08:00, or UTC-07:00
58   * in the summer. The offset -08:00 indicates that America/Los_Angeles time is
59   * obtained from UTC by adding -08:00, that is, by subtracting 8 hours.
60   * <p>
61   * The offset differs in the summer because of daylight saving time, or DST.
62   * The following definitions of time are generally used:
63   * <ul>
64   * <li>UTC - The reference time.
65   * <li>Standard Time - The local time without a daylight saving time offset.
66   * For example, in Paris, standard time is UTC+01:00.
67   * <li>Daylight Saving Time - The local time with a daylight saving time 
68   * offset. This offset is typically one hour, but not always. It is typically
69   * used in most countries away from the equator.  In Paris, daylight saving 
70   * time is UTC+02:00.
71   * <li>Wall Time - This is what a local clock on the wall reads. This will be
72   * either Standard Time or Daylight Saving Time depending on the time of year
73   * and whether the location uses Daylight Saving Time.
74   * </ul>
75   * <p>
76   * Unlike the Java TimeZone class, DateTimeZone is immutable. It also only
77   * supports long format time zone ids. Thus EST and ECT are not accepted.
78   * However, the factory that accepts a TimeZone will attempt to convert from
79   * the old short id to a suitable long id.
80   * <p>
81   * DateTimeZone is thread-safe and immutable, and all subclasses must be as
82   * well.
83   * 
84   * @author Brian S O'Neill
85   * @author Stephen Colebourne
86   * @since 1.0
87   */
88  public abstract class DateTimeZone implements Serializable {

121     /**
122      * Gets the default time zone.
123      * <p>
124      * The default time zone is derived from the system property {@code user.timezone}.
125      * If that is {@code null} or is not a valid identifier, then the value of the
126      * JDK {@code TimeZone} default is converted. If that fails, {@code UTC} is used.
127      * <p>
128      * NOTE: If the {@code java.util.TimeZone} default is updated <i>after</i> calling this
129      * method, then the change will not be picked up here.
130      * 
131      * @return the default datetime zone object
132      */
133     public static DateTimeZone getDefault() {
134         DateTimeZone zone = cDefault;
135         if (zone == null) {
136             synchronized(DateTimeZone.class) {
137                 zone = cDefault;
138                 if (zone == null) {
139                     DateTimeZone temp = null;
140                     try {
141                         try {
142                             String id = System.getProperty("user.timezone");
143                             if (id != null) {  // null check avoids stack overflow
144                                 temp = forID(id);
145                             }
146                         } catch (RuntimeException ex) {
147                             // ignored
148                         }
149                         if (temp == null) {
150                             temp = forTimeZone(TimeZone.getDefault());
151                         }
152                     } catch (IllegalArgumentException ex) {
153                         // ignored
154                     }
155                     if (temp == null) {
156                         temp = UTC;
157                     }
158                     cDefault = zone = temp;
159                 }
160             }
161         }
162         return zone;
163     }

188     /**
189      * Gets a time zone instance for the specified time zone id.
190      * <p>
191      * The time zone id may be one of those returned by getAvailableIDs.
192      * Short ids, as accepted by {@link java.util.TimeZone}, are not accepted.
193      * All IDs must be specified in the long format.
194      * The exception is UTC, which is an acceptable id.
195      * <p>
196      * Alternatively a locale independent, fixed offset, datetime zone can
197      * be specified. The form <code>[+-]hh:mm</code> can be used.
198      * 
199      * @param id  the ID of the datetime zone, null means default
200      * @return the DateTimeZone object for the ID
201      * @throws IllegalArgumentException if the ID is not recognised
202      */
203     @FromString
204     public static DateTimeZone forID(String id) {
205         if (id == null) {
206             return getDefault();
207         }
208         if (id.equals("UTC")) {
209             return DateTimeZone.UTC;
210         }
211         DateTimeZone zone = cProvider.getZone(id);
212         if (zone != null) {
213             return zone;
214         }
215         if (id.startsWith("+") || id.startsWith("-")) {
216             int offset = parseOffset(id);
217             if (offset == 0L) {
218                 return DateTimeZone.UTC;
219             } else {
220                 id = printOffset(offset);
221                 return fixedOffsetZone(id, offset);
222             }
223         }
224         throw new IllegalArgumentException("The datetime zone id '" + id + "' is not recognised");
225     }

241     /**
242      * Gets a time zone instance for the specified offset to UTC in hours and minutes.
243      * This method assumes 60 minutes in an hour, and standard length minutes.
244      * <p>
245      * This factory is a convenient way of constructing zones with a fixed offset.
246      * The hours value must be in the range -23 to +23.
247      * The minutes value must be in the range -59 to +59.
248      * The following combinations of sign for the hour and minute are possible:
249      * <pre>
250      *  Hour    Minute    Example    Result
251      * 
252      *  +ve     +ve       (2, 15)    +02:15
253      *  +ve     zero      (2, 0)     +02:00
254      *  +ve     -ve       (2, -15)   IllegalArgumentException
255      * 
256      *  zero    +ve       (0, 15)    +00:15
257      *  zero    zero      (0, 0)     +00:00
258      *  zero    -ve       (0, -15)   -00:15
259      * 
260      *  -ve     +ve       (-2, 15)   -02:15
261      *  -ve     zero      (-2, 0)    -02:00
262      *  -ve     -ve       (-2, -15)  -02:15
263      * </pre>
264      * Note that in versions before 2.3, the minutes had to be zero or positive.
265      * 
266      * @param hoursOffset  the offset in hours from UTC, from -23 to +23
267      * @param minutesOffset  the offset in minutes from UTC, from -59 to +59
268      * @return the DateTimeZone object for the offset
269      * @throws IllegalArgumentException if any value is out of range, the minutes are negative
270      *  when the hours are positive, or the resulting offset exceeds +/- 23:59:59.000
271      */
272     public static DateTimeZone forOffsetHoursMinutes(int hoursOffset, int minutesOffset) throws IllegalArgumentException {
273         if (hoursOffset == 0 && minutesOffset == 0) {
274             return DateTimeZone.UTC;
275         }
276         if (hoursOffset < -23 || hoursOffset > 23) {
277             throw new IllegalArgumentException("Hours out of range: " + hoursOffset);
278         }
279         if (minutesOffset < -59 || minutesOffset > 59) {
280             throw new IllegalArgumentException("Minutes out of range: " + minutesOffset);
281         }
282         if (hoursOffset > 0 && minutesOffset < 0) {
283             throw new IllegalArgumentException("Positive hours must not have negative minutes: " + minutesOffset);
284         }
285         int offset = 0;
286         try {
287             int hoursInMinutes = hoursOffset * 60;
288             if (hoursInMinutes < 0) {
289                 minutesOffset = hoursInMinutes - Math.abs(minutesOffset);
290             } else {
291                 minutesOffset = hoursInMinutes + minutesOffset;
292             }
293             offset = FieldUtils.safeMultiply(minutesOffset, DateTimeConstants.MILLIS_PER_MINUTE);
294         } catch (ArithmeticException ex) {
295             throw new IllegalArgumentException("Offset is too large");
296         }
297         return forOffsetMillis(offset);
298     }

301      * Gets a time zone instance for the specified offset to UTC in milliseconds.
302      *
303      * @param millisOffset  the offset in millis from UTC, from -23:59:59.999 to +23:59:59.999
304      * @return the DateTimeZone object for the offset
305      */
306     public static DateTimeZone forOffsetMillis(int millisOffset) {
307         if (millisOffset < -MAX_MILLIS || millisOffset > MAX_MILLIS) {
308             throw new IllegalArgumentException("Millis out of range: " + millisOffset);
309         }
310         String id = printOffset(millisOffset);
311         return fixedOffsetZone(id, millisOffset);
312     }

314     /**
315      * Gets a time zone instance for a JDK TimeZone.
316      * <p>
317      * DateTimeZone only accepts a subset of the IDs from TimeZone. The
318      * excluded IDs are the short three letter form (except UTC). This 
319      * method will attempt to convert between time zones created using the
320      * short IDs and the full version.
321      * <p>
322      * This method is not designed to parse time zones with rules created by
323      * applications using <code>SimpleTimeZone</code> directly.
324      * 
325      * @param zone  the zone to convert, null means default
326      * @return the DateTimeZone object for the zone
327      * @throws IllegalArgumentException if the zone is not recognised
328      */
329     public static DateTimeZone forTimeZone(TimeZone zone) {
330         if (zone == null) {
331             return getDefault();
332         }
333         final String id = zone.getID();
334         if (id == null) {
335             throw new IllegalArgumentException("The TimeZone id must not be null");
336         }
337         if (id.equals("UTC")) {
338             return DateTimeZone.UTC;
339         }
340 
341         // Convert from old alias before consulting provider since they may differ.
342         DateTimeZone dtz = null;
343         String convId = getConvertedId(id);
344         if (convId != null) {
345             dtz = cProvider.getZone(convId);
346         }
347         if (dtz == null) {
348             dtz = cProvider.getZone(id);
349         }
350         if (dtz != null) {
351             return dtz;
352         }
353 
354         // Support GMT+/-hh:mm formats
355         if (convId == null) {
356             convId = id;
357             if (convId.startsWith("GMT+") || convId.startsWith("GMT-")) {
358                 convId = convId.substring(3);
359                 int offset = parseOffset(convId);
360                 if (offset == 0L) {
361                     return DateTimeZone.UTC;
362                 } else {
363                     convId = printOffset(offset);
364                     return fixedOffsetZone(convId, offset);
365                 }
366             }
367         }
368         throw new IllegalArgumentException("The datetime zone id '" + id + "' is not recognised");
369     }

372     /**
373      * Gets the zone using a fixed offset amount.
374      * 
375      * @param id  the zone id
376      * @param offset  the offset in millis
377      * @return the zone
378      */
379     private static synchronized DateTimeZone fixedOffsetZone(String id, int offset) {
380         if (offset == 0) {
381             return DateTimeZone.UTC;
382         }
383         if (iFixedOffsetCache == null) {
384             iFixedOffsetCache = new HashMap<String, SoftReference<DateTimeZone>>();
385         }
386         DateTimeZone zone;
387         Reference<DateTimeZone> ref = iFixedOffsetCache.get(id);
388         if (ref != null) {
389             zone = ref.get();
390             if (zone != null) {
391                 return zone;
392             }
393         }
394         zone = new FixedDateTimeZone(id, null, offset, offset);
395         iFixedOffsetCache.put(id, new SoftReference<DateTimeZone>(zone));
396         return zone;
397     }

628     private static int parseOffset(String str) {
629         // Can't use a real chronology if called during class
630         // initialization. Offset parser doesn't need it anyhow.
631         Chronology chrono = new BaseChronology() {
632             private static final long serialVersionUID = -3128740902654445468L;
633             public DateTimeZone getZone() {
634                 return null;
635             }
636             public Chronology withUTC() {
637                 return this;
638             }
639             public Chronology withZone(DateTimeZone zone) {
640                 return this;
641             }
642             public String toString() {
643                 return getClass().getName();
644             }
645         };
646         return -(int) offsetFormatter().withChronology(chrono).parseMillis(str);
647     }

649     /**
650      * Formats a timezone offset string.
651      * <p>
652      * This method is kept separate from the formatting classes to speed and
653      * simplify startup and classloading.
654      * 
655      * @param offset  the offset in milliseconds
656      * @return the time zone string
657      */
658     private static String printOffset(int offset) {
659         StringBuffer buf = new StringBuffer();
660         if (offset >= 0) {
661             buf.append('+');
662         } else {
663             buf.append('-');
664             offset = -offset;
665         }
666 
667         int hours = offset / DateTimeConstants.MILLIS_PER_HOUR;
668         FormatUtils.appendPaddedInteger(buf, hours, 2);
669         offset -= hours * (int) DateTimeConstants.MILLIS_PER_HOUR;
670 
671         int minutes = offset / DateTimeConstants.MILLIS_PER_MINUTE;
672         buf.append(':');
673         FormatUtils.appendPaddedInteger(buf, minutes, 2);
674         offset -= minutes * DateTimeConstants.MILLIS_PER_MINUTE;
675         if (offset == 0) {
676             return buf.toString();
677         }
678 
679         int seconds = offset / DateTimeConstants.MILLIS_PER_SECOND;
680         buf.append(':');
681         FormatUtils.appendPaddedInteger(buf, seconds, 2);
682         offset -= seconds * DateTimeConstants.MILLIS_PER_SECOND;
683         if (offset == 0) {

875     /**
876      * Gets the millisecond offset to subtract from local time to get UTC time.
877      * This offset can be used to undo adding the offset obtained by getOffset.
878      *
879      * <pre>
880      * millisLocal == millisUTC   + getOffset(millisUTC)
881      * millisUTC   == millisLocal - getOffsetFromLocal(millisLocal)
882      * </pre>
883      *
884      * NOTE: After calculating millisLocal, some error may be introduced. At
885      * offset transitions (due to DST or other historical changes), ranges of
886      * local times may map to different UTC times.
887      * <p>
888      * For overlaps (where the local time is ambiguous), this method returns the
889      * offset applicable before the gap. The effect of this is that any instant
890      * calculated using the offset from an overlap will be in "summer" time.
891      * <p>
892      * For gaps, this method returns the offset applicable before the gap, ie "winter" offset.
893      * However, the effect of this is that any instant calculated using the offset
894      * from a gap will be after the gap, in "summer" time.
895      * <p>
896      * For example, consider a zone with a gap from 01:00 to 01:59:<br />
897      * Input: 00:00 (before gap) Output: Offset applicable before gap  DateTime: 00:00<br />
898      * Input: 00:30 (before gap) Output: Offset applicable before gap  DateTime: 00:30<br />
899      * Input: 01:00 (in gap)     Output: Offset applicable before gap  DateTime: 02:00<br />
900      * Input: 01:30 (in gap)     Output: Offset applicable before gap  DateTime: 02:30<br />
901      * Input: 02:00 (after gap)  Output: Offset applicable after gap   DateTime: 02:00<br />
902      * Input: 02:30 (after gap)  Output: Offset applicable after gap   DateTime: 02:30<br />
903      * <p>
904      * NOTE: Prior to v2.0, the DST overlap behaviour was not defined and varied by hemisphere.
905      * Prior to v1.5, the DST gap behaviour was also not defined.
906      * In v2.4, the documentation was clarified again.
907      *
908      * @param instantLocal  the millisecond instant, relative to this time zone, to get the offset for
909      * @return the millisecond offset to subtract from local time to get UTC time
910      */
911     public int getOffsetFromLocal(long instantLocal) {
912         // get the offset at instantLocal (first estimate)
913         final int offsetLocal = getOffset(instantLocal);
914         // adjust instantLocal using the estimate and recalc the offset
915         final long instantAdjusted = instantLocal - offsetLocal;
916         final int offsetAdjusted = getOffset(instantAdjusted);
917         // if the offsets differ, we must be near a DST boundary
918         if (offsetLocal != offsetAdjusted) {
919             // we need to ensure that time is always after the DST gap
920             // this happens naturally for positive offsets, but not for negative
921             if ((offsetLocal - offsetAdjusted) < 0) {
922                 // if we just return offsetAdjusted then the time is pushed
923                 // back before the transition, whereas it should be
924                 // on or after the transition
925                 long nextLocal = nextTransition(instantAdjusted);
926                 if (nextLocal == (instantLocal - offsetLocal)) {
927                     nextLocal = Long.MAX_VALUE;
928                 }
929                 long nextAdjusted = nextTransition(instantLocal - offsetAdjusted);
930                 if (nextAdjusted == (instantLocal - offsetAdjusted)) {
931                     nextAdjusted = Long.MAX_VALUE;
932                 }
933                 if (nextLocal != nextAdjusted) {
934                     return offsetLocal;
935                 }
936             }
937         } else if (offsetLocal >= 0) {
938             long prev = previousTransition(instantAdjusted);
939             if (prev < instantAdjusted) {
940                 int offsetPrev = getOffset(prev);
941                 int diff = offsetPrev - offsetLocal;
942                 if (instantAdjusted - prev <= diff) {
943                     return offsetPrev;
944                 }
945             }
946         }
947         return offsetAdjusted;
948     }

975      * where the calculation was done using a simple local zone.

978      *
979      * @param instantLocal  the local instant to convert to UTC

981      * @param originalInstantUTC  the original instant that the calculation is based on

989         long instantUTC = instantLocal - offsetOriginal;

991         if (offsetLocalFromOriginal == offsetOriginal) {
992             return instantUTC;
993         }
994         return convertLocalToUTC(instantLocal, strict);

997     /**
998      * Converts a local instant to a standard UTC instant with the same
999      * local time. This conversion is used after performing a calculation
1000      * where the calculation was done using a simple local zone.
1001      *
1002      * @param instantLocal  the local instant to convert to UTC
1003      * @param strict  whether the conversion should reject non-existent local times
1004      * @return the UTC instant with the same local time, 
1005      * @throws ArithmeticException if the result overflows a long
1006      * @throws IllegalInstantException if the zone has no equivalent local time
1007      * @since 1.5
1008      */
1009     public long convertLocalToUTC(long instantLocal, boolean strict) {
1010         // get the offset at instantLocal (first estimate)
1011         int offsetLocal = getOffset(instantLocal);
1012         // adjust instantLocal using the estimate and recalc the offset
1013         int offset = getOffset(instantLocal - offsetLocal);
1014         // if the offsets differ, we must be near a DST boundary
1015         if (offsetLocal != offset) {
1016             // if strict then always check if in DST gap
1017             // otherwise only check if zone in Western hemisphere (as the
1018             // value of offset is already correct for Eastern hemisphere)
1019             if (strict || offsetLocal < 0) {
1020                 // determine if we are in the DST gap
1021                 long nextLocal = nextTransition(instantLocal - offsetLocal);
1022                 if (nextLocal == (instantLocal - offsetLocal)) {
1023                     nextLocal = Long.MAX_VALUE;
1024                 }
1025                 long nextAdjusted = nextTransition(instantLocal - offset);
1026                 if (nextAdjusted == (instantLocal - offset)) {
1027                     nextAdjusted = Long.MAX_VALUE;
1028                 }
1029                 if (nextLocal != nextAdjusted) {
1030                     // yes we are in the DST gap
1031                     if (strict) {
1032                         // DST gap is not acceptable
1033                         throw new IllegalInstantException(instantLocal, getID());
1034                     } else {
1035                         // DST gap is acceptable, but for the Western hemisphere
1036                         // the offset is wrong and will result in local times
1037                         // before the cutover so use the offsetLocal instead
1038                         offset = offsetLocal;
1039                     }
1040                 }
1041             }
1042         }
1043         // check for overflow
1044         long instantUTC = instantLocal - offset;
1045         // If there is a sign change, but the two values have different signs...
1046         if ((instantLocal ^ instantUTC) < 0 && (instantLocal ^ offset) < 0) {
1047             throw new ArithmeticException("Subtracting time zone offset caused overflow");
1048         }
1049         return instantUTC;
1050     }

1202         long instantBefore = instant - 3 * DateTimeConstants.MILLIS_PER_HOUR;
1203         long instantAfter = instant + 3 * DateTimeConstants.MILLIS_PER_HOUR;

1224         } else {
