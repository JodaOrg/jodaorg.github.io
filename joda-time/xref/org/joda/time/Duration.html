24  /**
25   * An immutable duration specifying a length of time in milliseconds.
26   * <p>
27   * A duration is defined by a fixed number of milliseconds.
28   * There is no concept of fields, such as days or seconds, as these fields can vary in length.
29   * <p>
30   * A duration may be converted to a {@link Period} to obtain field values.
31   * This conversion will typically cause a loss of precision.
32   * <p>
33   * Duration is thread-safe and immutable.
34   *
35   * @author Brian S O'Neill
36   * @author Stephen Colebourne
37   * @since 1.0
38   */
39  public final class Duration
40          extends BaseDuration
41          implements ReadableDuration, Serializable {

64      /**
65       * Create a duration with the specified number of days assuming that
66       * there are the standard number of milliseconds in a day.
67       * <p>
68       * This method assumes that there are 24 hours in a day,
69       * 60 minutes in an hour, 60 seconds in a minute and 1000 milliseconds in
70       * a second. This will be true for most days, however days with Daylight
71       * Savings changes will not have 24 hours, so use this method with care.
72       * <p>
73       * A Duration is a representation of an amount of time. If you want to express
74       * the concept of 'days' you should consider using the {@link Days} class.
75       *
76       * @param days  the number of standard days in this duration
77       * @return the duration, never null
78       * @throws ArithmeticException if the days value is too large
79       * @since 1.6
80       */
81      public static Duration standardDays(long days) {
82          if (days == 0) {
83              return ZERO;
84          }
85          return new Duration(FieldUtils.safeMultiply(days, DateTimeConstants.MILLIS_PER_DAY));
86      }

379     /**
380      * Returns a new duration with this length plus that specified multiplied by the scalar.
381      * This instance is immutable and is not altered.
382      * <p>
383      * If the addition is zero, this instance is returned.
384      * 
385      * @param durationToAdd  the duration to add to this one
386      * @param scalar  the amount of times to add, such as -1 to subtract once
387      * @return the new duration instance
388      */
389     public Duration withDurationAdded(long durationToAdd, int scalar) {
390         if (durationToAdd == 0 || scalar == 0) {
391             return this;
392         }
393         long add = FieldUtils.safeMultiply(durationToAdd, scalar);
394         long duration = FieldUtils.safeAdd(getMillis(), add);
395         return new Duration(duration);
396     }
