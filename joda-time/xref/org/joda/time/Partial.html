31  /**
32   * Partial is an immutable partial datetime supporting any set of datetime fields.
33   * <p>
34   * A Partial instance can be used to hold any combination of fields.
35   * The instance does not contain a time zone, so any datetime is local.
36   * <p>
37   * A Partial can be matched against an instant using {@link #isMatch(ReadableInstant)}.
38   * This method compares each field on this partial with those of the instant
39   * and determines if the partial matches the instant.
40   * Given this definition, an empty Partial instance represents any datetime
41   * and always matches.
42   * <p>
43   * Calculations on Partial are performed using a {@link Chronology}.
44   * This chronology is set to be in the UTC time zone for all calculations.
45   * <p>
46   * Each individual field can be queried in two ways:
47   * <ul>
48   * <li><code>get(DateTimeFieldType.monthOfYear())</code>
49   * <li><code>property(DateTimeFieldType.monthOfYear()).get()</code>
50   * </ul>
51   * The second technique also provides access to other useful methods on the
52   * field:
53   * <ul>
54   * <li>numeric value - <code>monthOfYear().get()</code>
55   * <li>text value - <code>monthOfYear().getAsText()</code>
56   * <li>short text value - <code>monthOfYear().getAsShortText()</code>
57   * <li>maximum/minimum values - <code>monthOfYear().getMaximumValue()</code>
58   * <li>add/subtract - <code>monthOfYear().addToCopy()</code>
59   * <li>set - <code>monthOfYear().setCopy()</code>
60   * </ul>
61   * <p>
62   * Partial is thread-safe and immutable, provided that the Chronology is as well.
63   * All standard Chronology classes supplied are thread-safe and immutable.
64   *
65   * @author Stephen Colebourne
66   * @since 1.1
67   */
68  public final class Partial
69          extends AbstractPartial
70          implements ReadablePartial, Serializable {

178     /**
179      * Constructs a Partial with the specified fields and values.
180      * The fields must be specified in the order largest to smallest.
181      * <p>
182      * The constructor uses the specified chronology.
183      * 
184      * @param types  the types to create the partial from, not null
185      * @param values  the values to store, not null
186      * @param chronology  the chronology, null means ISO
187      * @throws IllegalArgumentException if the types or values are invalid
188      */
189     public Partial(DateTimeFieldType[] types, int[] values, Chronology chronology) {
190         super();
191         chronology = DateTimeUtils.getChronology(chronology).withUTC();
192         iChronology = chronology;
193         if (types == null) {
194             throw new IllegalArgumentException("Types array must not be null");
195         }
196         if (values == null) {
197             throw new IllegalArgumentException("Values array must not be null");
198         }
199         if (values.length != types.length) {
200             throw new IllegalArgumentException("Values array must be the same length as the types array");
201         }
202         if (types.length == 0) {
203             iTypes = types;
204             iValues = values;
205             return;
206         }
207         for (int i = 0; i < types.length; i++) {
208             if (types[i] == null) {
209                 throw new IllegalArgumentException("Types array must not contain null: index " + i);
210             }
211         }
212         DurationField lastUnitField = null;
213         for (int i = 0; i < types.length; i++) {
214             DateTimeFieldType loopType = types[i];
215             DurationField loopUnitField = loopType.getDurationType().getField(iChronology);
216             if (i > 0) {
217                 if (loopUnitField.isSupported() == false) {
218                     if (lastUnitField.isSupported()) {
219                         throw new IllegalArgumentException("Types array must be in order largest-smallest: " +
220                                         types[i - 1].getName() + " < " + loopType.getName());
221                     } else {
222                         throw new IllegalArgumentException("Types array must not contain duplicate unsupported: " +
223                                         types[i - 1].getName() + " and " + loopType.getName());
224                     }
225                 }
226                 int compare = lastUnitField.compareTo(loopUnitField);
227                 if (compare < 0) {
228                     throw new IllegalArgumentException("Types array must be in order largest-smallest: " +
229                             types[i - 1].getName() + " < " + loopType.getName());
230                 } else if (compare == 0 && lastUnitField.equals(loopUnitField)) {
231                     if (types[i - 1].getRangeDurationType() == null) {
232                         if (loopType.getRangeDurationType() == null) {
233                             throw new IllegalArgumentException("Types array must not contain duplicate: " +
234                                             types[i - 1].getName() + " and " + loopType.getName());
235                         }
236                     } else {
237                         if (loopType.getRangeDurationType() == null) {
238                             throw new IllegalArgumentException("Types array must be in order largest-smallest: " +
239                                     types[i - 1].getName() + " < " + loopType.getName());
240                         }
241                         DurationField lastRangeField = types[i - 1].getRangeDurationType().getField(iChronology);
242                         DurationField loopRangeField = loopType.getRangeDurationType().getField(iChronology);
243                         if (lastRangeField.compareTo(loopRangeField) < 0) {
244                             throw new IllegalArgumentException("Types array must be in order largest-smallest: " +
245                                     types[i - 1].getName() + " < " + loopType.getName());
246                         }
247                         if (lastRangeField.compareTo(loopRangeField) == 0) {
248                             throw new IllegalArgumentException("Types array must not contain duplicate: " +
249                                             types[i - 1].getName() + " and " + loopType.getName());
250                         }
251                     }
252                 }
253             }
254             lastUnitField = loopUnitField;
255         }
256         
257         iTypes = (DateTimeFieldType[]) types.clone();
258         chronology.validate(this, values);
259         iValues = (int[]) values.clone();
260     }

262     /**
263      * Constructs a Partial by copying all the fields and types from
264      * another partial.
265      * <p>
266      * This is most useful when copying from a YearMonthDay or TimeOfDay.
267      */
268     public Partial(ReadablePartial partial) {
269         super();
270         if (partial == null) {
271             throw new IllegalArgumentException("The partial must not be null");
272         }
273         iChronology = DateTimeUtils.getChronology(partial.getChronology()).withUTC();
274         iTypes = new DateTimeFieldType[partial.size()];
275         iValues = new int[partial.size()];
276         for (int i = 0; i < partial.size(); i++) {
277             iTypes[i] = partial.getFieldType(i);
278             iValues[i] = partial.getValue(i);
279         }
280     }

396     /**
397      * Creates a new Partial instance with the specified chronology.
398      * This instance is immutable and unaffected by this method call.
399      * <p>
400      * This method retains the values of the fields, thus the result will
401      * typically refer to a different instant.
402      * <p>
403      * The time zone of the specified chronology is ignored, as Partial
404      * operates without a time zone.
405      *
406      * @param newChronology  the new chronology, null means ISO
407      * @return a copy of this datetime with a different chronology
408      * @throws IllegalArgumentException if the values are invalid for the new chronology
409      */
410     public Partial withChronologyRetainFields(Chronology newChronology) {
411         newChronology = DateTimeUtils.getChronology(newChronology);
412         newChronology = newChronology.withUTC();
413         if (newChronology == getChronology()) {
414             return this;
415         } else {
416             Partial newPartial = new Partial(newChronology, iTypes, iValues);
417             newChronology.validate(newPartial, iValues);
418             return newPartial;
419         }
420     }

423     /**
424      * Gets a copy of this date with the specified field set to a new value.
425      * <p>
426      * If this partial did not previously support the field, the new one will.
427      * Contrast this behaviour with {@link #withField(DateTimeFieldType, int)}.
428      * <p>
429      * For example, if the field type is <code>dayOfMonth</code> then the day
430      * would be changed/added in the returned instance.
431      *
432      * @param fieldType  the field type to set, not null
433      * @param value  the value to set
434      * @return a copy of this instance with the field set
435      * @throws IllegalArgumentException if the value is null or invalid
436      */
437     public Partial with(DateTimeFieldType fieldType, int value) {
438         if (fieldType == null) {
439             throw new IllegalArgumentException("The field type must not be null");
440         }
441         int index = indexOf(fieldType);
442         if (index == -1) {
443             DateTimeFieldType[] newTypes = new DateTimeFieldType[iTypes.length + 1];
444             int[] newValues = new int[newTypes.length];
445             
446             // find correct insertion point to keep largest-smallest order
447             int i = 0;
448             DurationField unitField = fieldType.getDurationType().getField(iChronology);
449             if (unitField.isSupported()) {
450                 for (; i < iTypes.length; i++) {
451                     DateTimeFieldType loopType = iTypes[i];
452                     DurationField loopUnitField = loopType.getDurationType().getField(iChronology);
453                     if (loopUnitField.isSupported()) {
454                         int compare = unitField.compareTo(loopUnitField);
455                         if (compare > 0) {
456                             break;
457                         } else if (compare == 0) {
458                             if (fieldType.getRangeDurationType() == null) {
459                                 break;
460                             }
461                             if (loopType.getRangeDurationType() == null) {
462                                 continue;
463                             }
464                             DurationField rangeField = fieldType.getRangeDurationType().getField(iChronology);
465                             DurationField loopRangeField = loopType.getRangeDurationType().getField(iChronology);
466                             if (rangeField.compareTo(loopRangeField) > 0) {
467                                 break;
468                             }
469                         }
470                     }
471                 }
472             }
473             System.arraycopy(iTypes, 0, newTypes, 0, i);
474             System.arraycopy(iValues, 0, newValues, 0, i);
475             newTypes[i] = fieldType;
476             newValues[i] = value;
477             System.arraycopy(iTypes, i, newTypes, i + 1, newTypes.length - i - 1);
478             System.arraycopy(iValues, i, newValues, i + 1, newValues.length - i - 1);
479             // use public constructor to ensure full validation
480             // this isn't overly efficient, but is safe
481             Partial newPartial = new Partial(newTypes, newValues, iChronology);
482             iChronology.validate(newPartial, newValues);
483             return newPartial;
484         }
485         if (value == getValue(index)) {
486             return this;
487         }
488         int[] newValues = getValues();
489         newValues = getField(index).set(this, index, newValues, value);
490         return new Partial(this, newValues);
491     }

494      * Gets a copy of this date with the specified field removed.
495      * <p>
496      * If this partial did not previously support the field, no error occurs.
497      *
498      * @param fieldType  the field type to remove, may be null
499      * @return a copy of this instance with the field removed
500      */
501     public Partial without(DateTimeFieldType fieldType) {
502         int index = indexOf(fieldType);
503         if (index != -1) {
504             DateTimeFieldType[] newTypes = new DateTimeFieldType[size() - 1];
505             int[] newValues = new int[size() - 1];
506             System.arraycopy(iTypes, 0, newTypes, 0, index);
507             System.arraycopy(iTypes, index + 1, newTypes, index, newTypes.length - index);
508             System.arraycopy(iValues, 0, newValues, 0, index);
509             System.arraycopy(iValues, index + 1, newValues, index, newValues.length - index);
510             Partial newPartial = new Partial(iChronology, newTypes, newValues);
511             iChronology.validate(newPartial, newValues);
512             return newPartial;
513         }
514         return this;
515     }

590     /**
591      * Gets a copy of this Partial with the specified period added.
592      * <p>
593      * If the addition is zero, then <code>this</code> is returned.
594      * Fields in the period that aren't present in the partial are ignored.
595      * <p>
596      * This method is typically used to add multiple copies of complex
597      * period instances. Adding one field is best achieved using the method
598      * {@link #withFieldAdded(DurationFieldType, int)}.
599      * 
600      * @param period  the period to add to this one, null means zero
601      * @param scalar  the amount of times to add, such as -1 to subtract once
602      * @return a copy of this instance with the period added
603      * @throws ArithmeticException if the new datetime exceeds the capacity
604      */
605     public Partial withPeriodAdded(ReadablePeriod period, int scalar) {
606         if (period == null || scalar == 0) {
607             return this;
608         }
609         int[] newValues = getValues();
610         for (int i = 0; i < period.size(); i++) {
611             DurationFieldType fieldType = period.getFieldType(i);
612             int index = indexOf(fieldType);
613             if (index >= 0) {
614                 newValues = getField(index).add(this, index, newValues,
615                         FieldUtils.safeMultiply(period.getValue(i), scalar));
616             }
617         }
618         return new Partial(this, newValues);
619     }

723         if (f == null) {
724             if (size() == 0) {
725                 return null;
726             }
727             f = new DateTimeFormatter[2];
728             try {
729                 List<DateTimeFieldType> list = new ArrayList<DateTimeFieldType>(Arrays.asList(iTypes));
730                 f[0] = ISODateTimeFormat.forFields(list, true, false);
731                 if (list.size() == 0) {
732                     f[1] = f[0];
733                 }

736             }
737             iFormatter = f;
