36  /**

40   * reflecting the intention that they be defined in the same package.

45   * @author Stephen Colebourne

149     public long getDateTimeMillis(
150             int year, int monthOfYear, int dayOfMonth, int millisOfDay)
151             throws IllegalArgumentException {
152         Chronology base;
153         if ((base = getBase()) != null) {
154             return base.getDateTimeMillis(year, monthOfYear, dayOfMonth, millisOfDay);
155         }
156 
157         FieldUtils.verifyValueBounds
158             (DateTimeFieldType.millisOfDay(), millisOfDay, 0, DateTimeConstants.MILLIS_PER_DAY - 1);
159         return getDateMidnightMillis(year, monthOfYear, dayOfMonth) + millisOfDay;
160     }

162     public long getDateTimeMillis(
163             int year, int monthOfYear, int dayOfMonth,
164             int hourOfDay, int minuteOfHour, int secondOfMinute, int millisOfSecond)
165             throws IllegalArgumentException {
166         Chronology base;
167         if ((base = getBase()) != null) {
168             return base.getDateTimeMillis(year, monthOfYear, dayOfMonth,
169                                           hourOfDay, minuteOfHour, secondOfMinute, millisOfSecond);
170         }
171 
172         FieldUtils.verifyValueBounds(DateTimeFieldType.hourOfDay(), hourOfDay, 0, 23);
173         FieldUtils.verifyValueBounds(DateTimeFieldType.minuteOfHour(), minuteOfHour, 0, 59);
174         FieldUtils.verifyValueBounds(DateTimeFieldType.secondOfMinute(), secondOfMinute, 0, 59);
175         FieldUtils.verifyValueBounds(DateTimeFieldType.millisOfSecond(), millisOfSecond, 0, 999);
176 
177         return getDateMidnightMillis(year, monthOfYear, dayOfMonth)
178             + hourOfDay * DateTimeConstants.MILLIS_PER_HOUR
179             + minuteOfHour * DateTimeConstants.MILLIS_PER_MINUTE
180             + secondOfMinute * DateTimeConstants.MILLIS_PER_SECOND
181             + millisOfSecond;
182     }

190      * Checks if this chronology instance equals another.
191      * 
192      * @param obj  the object to compare to
193      * @return true if equal
194      * @since 1.6
195      */
196     public boolean equals(Object obj) {
197         if (this == obj) {
198             return true;
199         }
200         if (obj != null && getClass() == obj.getClass()) {
201             BasicChronology chrono = (BasicChronology) obj;
202             return getMinimumDaysInFirstWeek() == chrono.getMinimumDaysInFirstWeek() &&
203                     getZone().equals(chrono.getZone());
204         }
205         return false;
206     }

220     /**
221      * Gets a debugging toString.
222      * 
223      * @return a debugging string
224      */
225     public String toString() {
226         StringBuilder sb = new StringBuilder(60);
227         String name = getClass().getName();
228         int index = name.lastIndexOf('.');
229         if (index >= 0) {
230             name = name.substring(index + 1);
231         }
232         sb.append(name);
233         sb.append('[');
234         DateTimeZone zone = getZone();
235         if (zone != null) {
236             sb.append(zone.getID());
237         }
238         if (getMinimumDaysInFirstWeek() != 4) {
239             sb.append(",mdfw=");
240             sb.append(getMinimumDaysInFirstWeek());
241         }
242         sb.append(']');
243         return sb.toString();
244     }

246     protected void assemble(Fields fields) {
247         // First copy fields that are the same for all Gregorian and Julian
248         // chronologies.
249 
250         fields.millis = cMillisField;
251         fields.seconds = cSecondsField;
252         fields.minutes = cMinutesField;
253         fields.hours = cHoursField;
254         fields.halfdays = cHalfdaysField;
255         fields.days = cDaysField;
256         fields.weeks = cWeeksField;
257 
258         fields.millisOfSecond = cMillisOfSecondField;
259         fields.millisOfDay = cMillisOfDayField;
260         fields.secondOfMinute = cSecondOfMinuteField;
261         fields.secondOfDay = cSecondOfDayField;
262         fields.minuteOfHour = cMinuteOfHourField;
263         fields.minuteOfDay = cMinuteOfDayField;
264         fields.hourOfDay = cHourOfDayField;
265         fields.hourOfHalfday = cHourOfHalfdayField;
266         fields.clockhourOfDay = cClockhourOfDayField;
267         fields.clockhourOfHalfday = cClockhourOfHalfdayField;
268         fields.halfdayOfDay = cHalfdayOfDayField;
269 
270         // Now create fields that have unique behavior for Gregorian and Julian
271         // chronologies.
272 
273         fields.year = new BasicYearDateTimeField(this);
274         fields.yearOfEra = new GJYearOfEraDateTimeField(fields.year, this);
275 
276         // Define one-based centuryOfEra and yearOfCentury.
277         DateTimeField field = new OffsetDateTimeField(
278             fields.yearOfEra, 99);
279         fields.centuryOfEra = new DividedDateTimeField(
280             field, DateTimeFieldType.centuryOfEra(), 100);
281         
282         field = new RemainderDateTimeField(
283             (DividedDateTimeField) fields.centuryOfEra);
284         fields.yearOfCentury = new OffsetDateTimeField(
285             field, DateTimeFieldType.yearOfCentury(), 1);
286 
287         fields.era = new GJEraDateTimeField(this);
288         fields.dayOfWeek = new GJDayOfWeekDateTimeField(this, fields.days);
289         fields.dayOfMonth = new BasicDayOfMonthDateTimeField(this, fields.days);
290         fields.dayOfYear = new BasicDayOfYearDateTimeField(this, fields.days);
291         fields.monthOfYear = new GJMonthOfYearDateTimeField(this);
292         fields.weekyear = new BasicWeekyearDateTimeField(this);
293         fields.weekOfWeekyear = new BasicWeekOfWeekyearDateTimeField(this, fields.weeks);
294         
295         field = new RemainderDateTimeField(
296             fields.weekyear, DateTimeFieldType.weekyearOfCentury(), 100);
297         fields.weekyearOfCentury = new OffsetDateTimeField(
298             field, DateTimeFieldType.weekyearOfCentury(), 1);
299         
300         // The remaining (imprecise) durations are available from the newly
301         // created datetime fields.
302 
303         fields.years = fields.year.getDurationField();
304         fields.centuries = fields.centuryOfEra.getDurationField();
305         fields.months = fields.monthOfYear.getDurationField();
306         fields.weekyears = fields.weekyear.getDurationField();
307     }

329     /**
330      * Get the number of weeks in the year.
331      *
332      * @param year  the year to use
333      * @return number of weeks in the year
334      */
335     int getWeeksInYear(int year) {
336         long firstWeekMillis1 = getFirstWeekOfYearMillis(year);
337         long firstWeekMillis2 = getFirstWeekOfYearMillis(year + 1);
338         return (int) ((firstWeekMillis2 - firstWeekMillis1) / DateTimeConstants.MILLIS_PER_WEEK);
339     }

341     /**
342      * Get the millis for the first week of a year.
343      *
344      * @param year  the year to use
345      * @return millis
346      */
347     long getFirstWeekOfYearMillis(int year) {
348         long jan1millis = getYearMillis(year);
349         int jan1dayOfWeek = getDayOfWeek(jan1millis);
350         
351         if (jan1dayOfWeek > (8 - iMinDaysInFirstWeek)) {
352             // First week is end of previous year because it doesn't have enough days.
353             return jan1millis + (8 - jan1dayOfWeek)
354                 * (long)DateTimeConstants.MILLIS_PER_DAY;
355         } else {
356             // First week is start of this year because it has enough days.
357             return jan1millis - (jan1dayOfWeek - 1)
358                 * (long)DateTimeConstants.MILLIS_PER_DAY;
359         }
360     }

399     /**
400      * @param instant millis from 1970-01-01T00:00:00Z
401      */
402     int getYear(long instant) {
403         // Get an initial estimate of the year, and the millis value that
404         // represents the start of that year. Then verify estimate and fix if
405         // necessary.
406 
407         // Initial estimate uses values divided by two to avoid overflow.
408         long unitMillis = getAverageMillisPerYearDividedByTwo();
409         long i2 = (instant >> 1) + getApproxMillisAtEpochDividedByTwo();
410         if (i2 < 0) {
411             i2 = i2 - unitMillis + 1;
412         }
413         int year = (int) (i2 / unitMillis);
414 
415         long yearStart = getYearMillis(year);
416         long diff = instant - yearStart;
417 
418         if (diff < 0) {
419             year--;
420         } else if (diff >= DateTimeConstants.MILLIS_PER_DAY * 365L) {
421             // One year may need to be added to fix estimate.
422             long oneYear;
423             if (isLeapYear(year)) {
424                 oneYear = DateTimeConstants.MILLIS_PER_DAY * 366L;
425             } else {
426                 oneYear = DateTimeConstants.MILLIS_PER_DAY * 365L;
427             }
428 
429             yearStart += oneYear;
430 
431             if (yearStart <= instant) {
432                 // Didn't go too far, so actually add one year.
433                 year++;
434             }
435         }
436 
437         return year;
438     }

501     int getWeekyear(long instant) {
502         int year = getYear(instant);
503         int week = getWeekOfWeekyear(instant, year);
504         if (week == 1) {
505             return getYear(instant + DateTimeConstants.MILLIS_PER_WEEK);
506         } else if (week > 51) {
507             return getYear(instant - (2 * DateTimeConstants.MILLIS_PER_WEEK));
508         } else {

510         }

520     /**
521      * @param instant millis from 1970-01-01T00:00:00Z
522      * @param year precalculated year of millis
523      */
524     int getWeekOfWeekyear(long instant, int year) {
525         long firstWeekMillis1 = getFirstWeekOfYearMillis(year);
526         if (instant < firstWeekMillis1) {
527             return getWeeksInYear(year - 1);
528         }
529         long firstWeekMillis2 = getFirstWeekOfYearMillis(year + 1);
530         if (instant >= firstWeekMillis2) {
531             return 1;
532         }
533         return (int) ((instant - firstWeekMillis1) / DateTimeConstants.MILLIS_PER_WEEK) + 1;
534     }

536     /**
537      * @param instant millis from 1970-01-01T00:00:00Z
538      */
539     int getDayOfWeek(long instant) {
540         // 1970-01-01 is day of week 4, Thursday.
541 
542         long daysSince19700101;
543         if (instant >= 0) {
544             daysSince19700101 = instant / DateTimeConstants.MILLIS_PER_DAY;
545         } else {
546             daysSince19700101 = (instant - (DateTimeConstants.MILLIS_PER_DAY - 1))
547                 / DateTimeConstants.MILLIS_PER_DAY;
548             if (daysSince19700101 < -3) {
549                 return 7 + (int) ((daysSince19700101 + 4) % 7);
550             }
551         }
552 
553         return 1 + (int) ((daysSince19700101 + 3) % 7);
554     }

557      * @param instant millis from 1970-01-01T00:00:00Z
558      */
559     int getMillisOfDay(long instant) {
560         if (instant >= 0) {
561             return (int) (instant % DateTimeConstants.MILLIS_PER_DAY);
562         } else {
563             return (DateTimeConstants.MILLIS_PER_DAY - 1)
564                 + (int) ((instant + 1) % DateTimeConstants.MILLIS_PER_DAY);
565         }
566     }
