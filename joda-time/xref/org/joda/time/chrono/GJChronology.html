36  /**
37   * Implements the Gregorian/Julian calendar system which is the calendar system
38   * used in most of the world. Wherever possible, it is recommended to use the
39   * {@link ISOChronology} instead.
40   * <p>
41   * The Gregorian calendar replaced the Julian calendar, and the point in time
42   * when this chronology switches can be controlled using the second parameter
43   * of the getInstance method. By default this cutover is set to the date the
44   * Gregorian calendar was first instituted, October 15, 1582.
45   * <p>
46   * Before this date, this chronology uses the proleptic Julian calendar
47   * (proleptic means extending indefinitely). The Julian calendar has leap years
48   * every four years, whereas the Gregorian has special rules for 100 and 400
49   * years. A meaningful result will thus be obtained for all input values.
50   * However before 8 CE, Julian leap years were irregular, and before 45 BCE
51   * there was no Julian calendar.
52   * <p>
53   * This chronology differs from
54   * {@link java.util.GregorianCalendar GregorianCalendar} in that years
55   * in BCE are returned correctly. Thus year 1 BCE is returned as -1 instead of 1.
56   * The yearOfEra field produces results compatible with GregorianCalendar.
57   * <p>
58   * The Julian calendar does not have a year zero, and so year -1 is followed by
59   * year 1. If the Gregorian cutover date is specified at or before year -1
60   * (Julian), year zero is defined. In other words, the proleptic Gregorian
61   * chronology used by this class has a year zero.
62   * <p>
63   * To create a pure proleptic Julian chronology, use {@link JulianChronology},
64   * and to create a pure proleptic Gregorian chronology, use
65   * {@link GregorianChronology}.
66   * <p>
67   * GJChronology is thread-safe and immutable.
68   * 
69   * @author Brian S O'Neill
70   * @author Stephen Colebourne
71   * @since 1.0
72   */
73  public final class GJChronology extends AssembledChronology {

79       * Convert a datetime from one chronology to another.

81      private static long convertByYear(long instant, Chronology from, Chronology to) {
82          return to.getDateTimeMillis
83              (from.year().get(instant),
84               from.monthOfYear().get(instant),
85               from.dayOfMonth().get(instant),
86               from.millisOfDay().get(instant));
87      }

89      /**
90       * Convert a datetime from one chronology to another.
91       */
92      private static long convertByWeekyear(final long instant, Chronology from, Chronology to) {
93          long newInstant;
94          newInstant = to.weekyear().set(0, from.weekyear().get(instant));
95          newInstant = to.weekOfWeekyear().set(newInstant, from.weekOfWeekyear().get(instant));
96          newInstant = to.dayOfWeek().set(newInstant, from.dayOfWeek().get(instant));
97          newInstant = to.millisOfDay().set(newInstant, from.millisOfDay().get(instant));
98          return newInstant;
99      }

175     /**
176      * Factory method returns instances of the GJ cutover chronology. Any
177      * cutover date may be specified.
178      *
179      * @param zone  the time zone to use, null is default
180      * @param gregorianCutover  the cutover to use, null means default
181      * @param minDaysInFirstWeek  minimum number of days in first week of the year; default is 4
182      */
183     public static GJChronology getInstance(
184             DateTimeZone zone,
185             ReadableInstant gregorianCutover,
186             int minDaysInFirstWeek) {
187         
188         zone = DateTimeUtils.getZone(zone);
189         Instant cutoverInstant;
190         if (gregorianCutover == null) {
191             cutoverInstant = DEFAULT_CUTOVER;
192         } else {
193             cutoverInstant = gregorianCutover.toInstant();
194             LocalDate cutoverDate = new LocalDate(cutoverInstant.getMillis(), GregorianChronology.getInstance(zone));
195             if (cutoverDate.getYear() <= 0) {
196                 throw new IllegalArgumentException("Cutover too early. Must be on or after 0001-01-01.");
197             }
198         }
199 
200         GJCacheKey cacheKey = new GJCacheKey(zone, cutoverInstant, minDaysInFirstWeek);
201         GJChronology chrono = cCache.get(cacheKey);
202         if (chrono == null) {
203             if (zone == DateTimeZone.UTC) {
204                 chrono = new GJChronology
205                     (JulianChronology.getInstance(zone, minDaysInFirstWeek),
206                      GregorianChronology.getInstance(zone, minDaysInFirstWeek),
207                      cutoverInstant);
208             } else {
209                 chrono = getInstance(DateTimeZone.UTC, cutoverInstant, minDaysInFirstWeek);
210                 chrono = new GJChronology
211                     (ZonedChronology.getInstance(chrono, zone),
212                      chrono.iJulianChronology,
213                      chrono.iGregorianChronology,
214                      chrono.iCutoverInstant);
215             }
216             GJChronology oldChrono = cCache.putIfAbsent(cacheKey, chrono);
217             if (oldChrono != null) {
218                 chrono = oldChrono;
219             }
220         }
221         return chrono;
222     }

226      * cutover date may be specified.
227      *
228      * @param zone  the time zone to use, null is default
229      * @param gregorianCutover  the cutover to use

232     public static GJChronology getInstance(
233             DateTimeZone zone,
234             long gregorianCutover,
235             int minDaysInFirstWeek) {
236         
237         Instant cutoverInstant;
238         if (gregorianCutover == DEFAULT_CUTOVER.getMillis()) {
239             cutoverInstant = null;
240         } else {
241             cutoverInstant = new Instant(gregorianCutover);
242         }
243         return getInstance(zone, cutoverInstant, minDaysInFirstWeek);
244     }

317     public long getDateTimeMillis(int year, int monthOfYear, int dayOfMonth,
318                                   int millisOfDay)
319         throws IllegalArgumentException
320     {
321         Chronology base;
322         if ((base = getBase()) != null) {
323             return base.getDateTimeMillis(year, monthOfYear, dayOfMonth, millisOfDay);
324         }
325 
326         // Assume date is Gregorian.
327         long instant = iGregorianChronology.getDateTimeMillis
328             (year, monthOfYear, dayOfMonth, millisOfDay);
329         if (instant < iCutoverMillis) {
330             // Maybe it's Julian.
331             instant = iJulianChronology.getDateTimeMillis
332                 (year, monthOfYear, dayOfMonth, millisOfDay);
333             if (instant >= iCutoverMillis) {
334                 // Okay, it's in the illegal cutover gap.
335                 throw new IllegalArgumentException("Specified date does not exist");
336             }
337         }
338         return instant;
339     }

341     public long getDateTimeMillis(int year, int monthOfYear, int dayOfMonth,
342                                   int hourOfDay, int minuteOfHour,
343                                   int secondOfMinute, int millisOfSecond)
344         throws IllegalArgumentException
345     {
346         Chronology base;
347         if ((base = getBase()) != null) {
348             return base.getDateTimeMillis
349                 (year, monthOfYear, dayOfMonth,
350                  hourOfDay, minuteOfHour, secondOfMinute, millisOfSecond);
351         }
352 
353         // Assume date is Gregorian.
354         long instant;
355         try {
356             instant = iGregorianChronology.getDateTimeMillis
357                 (year, monthOfYear, dayOfMonth,
358                  hourOfDay, minuteOfHour, secondOfMinute, millisOfSecond);
359         } catch (IllegalFieldValueException ex) {
360             if (monthOfYear != 2 || dayOfMonth != 29) {
361                 throw ex;
362             }
363             instant = iGregorianChronology.getDateTimeMillis
364                 (year, monthOfYear, 28,
365                  hourOfDay, minuteOfHour, secondOfMinute, millisOfSecond);
366             if (instant >= iCutoverMillis) {
367                 throw ex;
368             }
369         }
370         if (instant < iCutoverMillis) {
371             // Maybe it's Julian.
372             instant = iJulianChronology.getDateTimeMillis
373                 (year, monthOfYear, dayOfMonth,
374                  hourOfDay, minuteOfHour, secondOfMinute, millisOfSecond);
375             if (instant >= iCutoverMillis) {
376                 // Okay, it's in the illegal cutover gap.
377                 throw new IllegalArgumentException("Specified date does not exist");
378             }
379         }
380         return instant;
381     }

434     /**
435      * Gets a debugging toString.
436      * 
437      * @return a debugging string
438      */
439     public String toString() {
440         StringBuffer sb = new StringBuffer(60);
441         sb.append("GJChronology");
442         sb.append('[');
443         sb.append(getZone().getID());
444         
445         if (iCutoverMillis != DEFAULT_CUTOVER.getMillis()) {
446             sb.append(",cutover=");
447             DateTimeFormatter printer;
448             if (withUTC().dayOfYear().remainder(iCutoverMillis) == 0) {
449                 printer = ISODateTimeFormat.date();
450             } else {
451                 printer = ISODateTimeFormat.dateTime();
452             }
453             printer.withChronology(withUTC()).printTo(sb, iCutoverMillis);
454         }
455         
456         if (getMinimumDaysInFirstWeek() != 4) {
457             sb.append(",mdfw=");
458             sb.append(getMinimumDaysInFirstWeek());
459         }
460         sb.append(']');
461         
462         return sb.toString();
463     }

465     protected void assemble(Fields fields) {
466         Object[] params = (Object[])getParam();
467 
468         JulianChronology julian = (JulianChronology)params[0];
469         GregorianChronology gregorian = (GregorianChronology)params[1];
470         Instant cutoverInstant = (Instant)params[2];
471         iCutoverMillis = cutoverInstant.getMillis();
472 
473         iJulianChronology = julian;
474         iGregorianChronology = gregorian;
475         iCutoverInstant = cutoverInstant;
476 
477         if (getBase() != null) {
478             return;
479         }
480 
481         if (julian.getMinimumDaysInFirstWeek() != gregorian.getMinimumDaysInFirstWeek()) {
482             throw new IllegalArgumentException();
483         }
484 
485         // Compute difference between the chronologies at the cutover instant
486         iGapDuration = iCutoverMillis - julianToGregorianByYear(iCutoverMillis);
487 
488         // Begin field definitions.
489 
490         // First just copy all the Gregorian fields and then override those
491         // that need special attention.
492         fields.copyFieldsFrom(gregorian);
493         
494         // Assuming cutover is at midnight, all time of day fields can be
495         // gregorian since they are unaffected by cutover.
496 
497         // Verify assumption.
498         if (gregorian.millisOfDay().get(iCutoverMillis) == 0) {
499             // Cutover is sometime in the day, so cutover fields are required
500             // for time of day.
501 
502             fields.millisOfSecond = new CutoverField(julian.millisOfSecond(), fields.millisOfSecond, iCutoverMillis);
503             fields.millisOfDay = new CutoverField(julian.millisOfDay(), fields.millisOfDay, iCutoverMillis);
504             fields.secondOfMinute = new CutoverField(julian.secondOfMinute(), fields.secondOfMinute, iCutoverMillis);
505             fields.secondOfDay = new CutoverField(julian.secondOfDay(), fields.secondOfDay, iCutoverMillis);
506             fields.minuteOfHour = new CutoverField(julian.minuteOfHour(), fields.minuteOfHour, iCutoverMillis);
507             fields.minuteOfDay = new CutoverField(julian.minuteOfDay(), fields.minuteOfDay, iCutoverMillis);
508             fields.hourOfDay = new CutoverField(julian.hourOfDay(), fields.hourOfDay, iCutoverMillis);
509             fields.hourOfHalfday = new CutoverField(julian.hourOfHalfday(), fields.hourOfHalfday, iCutoverMillis);
510             fields.clockhourOfDay = new CutoverField(julian.clockhourOfDay(), fields.clockhourOfDay, iCutoverMillis);
511             fields.clockhourOfHalfday = new CutoverField(julian.clockhourOfHalfday(),
512                                                          fields.clockhourOfHalfday, iCutoverMillis);
513             fields.halfdayOfDay = new CutoverField(julian.halfdayOfDay(), fields.halfdayOfDay, iCutoverMillis);
514         }
515 
516         // These fields just require basic cutover support.
517         {
518             fields.era = new CutoverField(julian.era(), fields.era, iCutoverMillis);
519         }
520 
521         // These fields are special because they have imprecise durations. The
522         // family of addition methods need special attention. Override affected
523         // duration fields as well.
524         {
525             fields.year = new ImpreciseCutoverField(
526                 julian.year(), fields.year, iCutoverMillis);
527             fields.years = fields.year.getDurationField();
528             fields.yearOfEra = new ImpreciseCutoverField(
529                 julian.yearOfEra(), fields.yearOfEra, fields.years, iCutoverMillis);
530             
531             fields.centuryOfEra = new ImpreciseCutoverField(
532                 julian.centuryOfEra(), fields.centuryOfEra, iCutoverMillis);
533             fields.centuries = fields.centuryOfEra.getDurationField();
534             
535             fields.yearOfCentury = new ImpreciseCutoverField(
536                 julian.yearOfCentury(), fields.yearOfCentury, fields.years, fields.centuries, iCutoverMillis);
537             
538             fields.monthOfYear = new ImpreciseCutoverField(
539                 julian.monthOfYear(), fields.monthOfYear, null, fields.years, iCutoverMillis);
540             fields.months = fields.monthOfYear.getDurationField();
541             
542             fields.weekyear = new ImpreciseCutoverField(
543                 julian.weekyear(), fields.weekyear, null, iCutoverMillis, true);
544             fields.weekyears = fields.weekyear.getDurationField();
545             fields.weekyearOfCentury = new ImpreciseCutoverField(
546                 julian.weekyearOfCentury(), fields.weekyearOfCentury, fields.weekyears, fields.centuries, iCutoverMillis);
547         }
548 
549         // DayOfYear and weekOfWeekyear require special handling since cutover
550         // year has fewer days and weeks. Extend the cutover to the start of
551         // the next year or weekyear. This keeps the sequence unbroken during
552         // the cutover year.
553 
554         {
555             long cutover = gregorian.year().roundCeiling(iCutoverMillis);
556             fields.dayOfYear = new CutoverField(
557                 julian.dayOfYear(), fields.dayOfYear, fields.years, cutover, false);
558         }
559 
560         {
561             long cutover = gregorian.weekyear().roundCeiling(iCutoverMillis);
562             fields.weekOfWeekyear = new CutoverField(
563                 julian.weekOfWeekyear(), fields.weekOfWeekyear, fields.weekyears, cutover, true);
564         }
565 
566         // These fields require basic cutover support, except they must link to
567         // imprecise durations.
568         {
569             CutoverField cf = new CutoverField
570                 (julian.dayOfMonth(), fields.dayOfMonth, iCutoverMillis);
571             cf.iRangeDurationField = fields.months;
572             fields.dayOfMonth = cf;
573         }
574     }
