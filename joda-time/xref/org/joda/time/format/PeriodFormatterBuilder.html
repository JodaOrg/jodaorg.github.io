39  /**
40   * Factory that creates complex instances of PeriodFormatter via method calls.
41   * <p>
42   * Period formatting is performed by the {@link PeriodFormatter} class.
43   * Three classes provide factory methods to create formatters, and this is one.
44   * The others are {@link PeriodFormat} and {@link ISOPeriodFormat}.
45   * <p>
46   * PeriodFormatterBuilder is used for constructing formatters which are then
47   * used to print or parse. The formatters are built by appending specific fields
48   * or other formatters to an instance of this builder.
49   * <p>
50   * For example, a formatter that prints years and months, like "15 years and 8 months",
51   * can be constructed as follows:
52   * <p>
53   * <pre>
54   * PeriodFormatter yearsAndMonths = new PeriodFormatterBuilder()
55   *     .printZeroAlways()
56   *     .appendYears()
57   *     .appendSuffix(" year", " years")
58   *     .appendSeparator(" and ")
59   *     .printZeroRarelyLast()
60   *     .appendMonths()
61   *     .appendSuffix(" month", " months")
62   *     .toFormatter();
63   * </pre>
64   * <p>
65   * PeriodFormatterBuilder itself is mutable and not thread-safe, but the
66   * formatters that it builds are thread-safe and immutable.
67   *
68   * @author Brian S O'Neill
69   * @since 1.0
70   * @see PeriodFormat
71   */

115     /**
116      * Constructs a PeriodFormatter using all the appended elements.
117      * <p>
118      * This is the main method used by applications at the end of the build
119      * process to create a usable formatter.
120      * <p>
121      * Subsequent changes to this builder do not affect the returned formatter.
122      * <p>
123      * The returned formatter may not support both printing and parsing.
124      * The methods {@link PeriodFormatter#isPrinter()} and
125      * {@link PeriodFormatter#isParser()} will help you determine the state
126      * of the formatter.
127      * 
128      * @return the newly created formatter
129      * @throws IllegalStateException if the builder can produce neither a printer nor a parser
130      */
131     public PeriodFormatter toFormatter() {
132         PeriodFormatter formatter = toFormatter(iElementPairs, iNotPrinter, iNotParser);
133         for (FieldFormatter fieldFormatter : iFieldFormatters) {
134             if (fieldFormatter != null) {
135                 fieldFormatter.finish(iFieldFormatters);
136             }
137         }
138         iFieldFormatters = (FieldFormatter[]) iFieldFormatters.clone();
139         return formatter;
140     }

383     /**
384      * Append a field prefix which applies only to the next appended field.
385      * If the field is not printed, neither is the prefix.
386      * <p>
387      * The value is converted to String. During parsing, the prefix is selected based
388      * on the match with the regular expression. The index of the first regular
389      * expression that matches value converted to String nominates the prefix. If
390      * none of the regular expressions match the value converted to String then the
391      * last prefix is selected.
392      * <p>
393      * An example usage for English might look like this:
394      * 
395      * <pre>
396      * appendPrefix(new String[] { &quot;&circ;1$&quot;, &quot;.*&quot; }, new String[] { &quot; year&quot;, &quot; years&quot; })
397      * </pre>
398      * 
399      * <p>
400      * Please note that for languages with simple mapping (singular and plural prefix
401      * only - like the one above) the {@link #appendPrefix(String, String)} method
402      * will produce in a slightly faster formatter and that
403      * {@link #appendPrefix(String[], String[])} method should be only used when the
404      * mapping between values and prefixes is more complicated than the difference between
405      * singular and plural.
406      *
407      * @param regularExpressions  an array of regular expressions, at least one
408      *  element, length has to match the length of prefixes parameter
409      * @param prefixes  an array of prefixes, at least one element, length has to
410      *  match the length of regularExpressions parameter
411      * @return this PeriodFormatterBuilder
412      * @throws IllegalStateException if no field exists to append to
413      * @see #appendPrefix
414      * @since 2.5
415      */
416     public PeriodFormatterBuilder appendPrefix(String[] regularExpressions, String[] prefixes) {
417         if (regularExpressions == null || prefixes == null ||
418                 regularExpressions.length < 1 || regularExpressions.length != prefixes.length) {
419             throw new IllegalArgumentException();
420         }
421         return appendPrefix(new RegExAffix(regularExpressions, prefixes));
422     }

425      * Append a field prefix which applies only to the next appended field. If
426      * the field is not printed, neither is the prefix.
427      *
428      * @param prefix custom prefix
429      * @return this PeriodFormatterBuilder
430      * @see #appendSuffix
431      */
432     private PeriodFormatterBuilder appendPrefix(PeriodFieldAffix prefix) {
433         if (prefix == null) {
434             throw new IllegalArgumentException();
435         }

437             prefix = new CompositeAffix(iPrefix, prefix);

675     /**
676      * Append a field suffix which applies only to the last appended field. If
677      * the field is not printed, neither is the suffix.
678      *
679      * @param suffix custom suffix
680      * @return this PeriodFormatterBuilder
681      * @throws IllegalStateException if no field exists to append to
682      * @see #appendPrefix
683      */
684     private PeriodFormatterBuilder appendSuffix(PeriodFieldAffix suffix) {
685         final Object originalPrinter;
686         final Object originalParser;
687         if (iElementPairs.size() > 0) {
688             originalPrinter = iElementPairs.get(iElementPairs.size() - 2);
689             originalParser = iElementPairs.get(iElementPairs.size() - 1);
690         } else {
691             originalPrinter = null;
692             originalParser = null;
693         }
694 
695         if (originalPrinter == null || originalParser == null ||
696                 originalPrinter != originalParser ||
697                 !(originalPrinter instanceof FieldFormatter)) {
698             throw new IllegalStateException("No field to apply suffix to");
699         }
700 
701         clearPrefix();
702         FieldFormatter newField = new FieldFormatter((FieldFormatter) originalPrinter, suffix);
703         iElementPairs.set(iElementPairs.size() - 2, newField);
704         iElementPairs.set(iElementPairs.size() - 1, newField);
705         iFieldFormatters[newField.getFieldType()] = newField;
706         
707         return this;
708     }

823     private PeriodFormatterBuilder appendSeparator(String text, String finalText,
824                                                    String[] variants,
825                                                    boolean useBefore, boolean useAfter) {
826         if (text == null || finalText == null) {
827             throw new IllegalArgumentException();
828         }
829 
830         clearPrefix();
831         
832         // optimise zero formatter case
833         List<Object> pairs = iElementPairs;
834         if (pairs.size() == 0) {
835             if (useAfter && useBefore == false) {
836                 Separator separator = new Separator(
837                         text, finalText, variants,
838                         Literal.EMPTY, Literal.EMPTY, useBefore, useAfter);
839                 append0(separator, separator);
840             }
841             return this;
842         }
843         
844         // find the last separator added
845         int i;
846         Separator lastSeparator = null;
847         for (i=pairs.size(); --i>=0; ) {
848             if (pairs.get(i) instanceof Separator) {
849                 lastSeparator = (Separator) pairs.get(i);
850                 pairs = pairs.subList(i + 1, pairs.size());
851                 break;
852             }
853             i--;  // element pairs
854         }
855         
856         // merge formatters
857         if (lastSeparator != null && pairs.size() == 0) {
858             throw new IllegalStateException("Cannot have two adjacent separators");
859         } else {
860             Object[] comp = createComposite(pairs);
861             pairs.clear();
862             Separator separator = new Separator(
863                     text, finalText, variants,
864                     (PeriodPrinter) comp[0], (PeriodParser) comp[1],
865                     useBefore, useAfter);
866             pairs.add(separator);
867             pairs.add(separator);
868         }
869         
870         return this;
871     }

890     private static PeriodFormatter toFormatter(List<Object> elementPairs, boolean notPrinter, boolean notParser) {
891         if (notPrinter && notParser) {
892             throw new IllegalStateException("Builder has created neither a printer nor a parser");
893         }
894         int size = elementPairs.size();
895         if (size >= 2 && elementPairs.get(0) instanceof Separator) {
896             Separator sep = (Separator) elementPairs.get(0);
897             if (sep.iAfterParser == null && sep.iAfterPrinter == null) {
898                 PeriodFormatter f = toFormatter(elementPairs.subList(2, size), notPrinter, notParser);
899                 sep = sep.finish(f.getPrinter(), f.getParser());
900                 return new PeriodFormatter(sep, sep);
901             }
902         }
903         Object[] comp = createComposite(elementPairs);
904         if (notPrinter) {
905             return new PeriodFormatter(null, (PeriodParser) comp[1]);
906         } else if (notParser) {
907             return new PeriodFormatter((PeriodPrinter) comp[0], null);
908         } else {
909             return new PeriodFormatter((PeriodPrinter) comp[0], (PeriodParser) comp[1]);
910         }
911     }

919             default:
920                 Composite comp = new Composite(elementPairs);
921                 return new Object[] {comp, comp};
922         }

967      * An affix that can be ignored.
968      */
969     static abstract class IgnorableAffix implements PeriodFieldAffix {
970         private volatile String[] iOtherAffixes;
971 
972         public void finish(Set<PeriodFieldAffix> periodFieldAffixesToIgnore) {
973             if (iOtherAffixes == null) {
974                 // Calculate the shortest affix in this instance.
975                 int shortestAffixLength = Integer.MAX_VALUE;
976                 String shortestAffix = null;
977                 for (String affix : getAffixes()) {
978                     if (affix.length() < shortestAffixLength) {
979                         shortestAffixLength = affix.length();
980                         shortestAffix = affix;
981                     }
982                 }
983                 
984                 // Pick only affixes that are longer than the shortest affix in this instance.
985                 // This will reduce the number of parse operations and thus speed up the PeriodFormatter.
986                 // also need to pick affixes that differ only in case (but not those that are identical)
987                 Set<String> affixesToIgnore = new HashSet<String>();
988                 for (PeriodFieldAffix periodFieldAffixToIgnore : periodFieldAffixesToIgnore) {
989                     if (periodFieldAffixToIgnore != null) {
990                         for (String affixToIgnore : periodFieldAffixToIgnore.getAffixes()) {
991                             if (affixToIgnore.length() > shortestAffixLength ||
992                                     (affixToIgnore.equalsIgnoreCase(shortestAffix) && !affixToIgnore.equals(shortestAffix))) {
993                                 affixesToIgnore.add(affixToIgnore);
994                             }
995                         }                       
996                     }
997                 }
998                 iOtherAffixes = affixesToIgnore.toArray(new String[affixesToIgnore.size()]);
999             }
1000         }
1001 
1002         /**
1003          * Checks if there is a match among the other affixes (stored internally) 
1004          * that is longer than the passed value (textLength).
1005          * 
1006          * @param textLength  the length of the match
1007          * @param periodStr  the Period string that will be parsed
1008          * @param position  the position in the Period string at which the parsing should be started.
1009          * @return true if the other affixes (stored internally) contain a match 
1010          *  that is longer than the textLength parameter, false otherwise
1011          */
1012         protected boolean matchesOtherAffix(int textLength, String periodStr, int position) {
1013             if (iOtherAffixes != null) {
1014                 // ignore case when affix length differs
1015                 // match case when affix length is same
1016                 for (String affixToIgnore : iOtherAffixes) {
1017                     int textToIgnoreLength = affixToIgnore.length();
1018                     if ((textLength < textToIgnoreLength && periodStr.regionMatches(true, position, affixToIgnore, 0, textToIgnoreLength)) ||
1019                             (textLength == textToIgnoreLength && periodStr.regionMatches(false, position, affixToIgnore, 0, textToIgnoreLength))) {
1020                         return true;
1021                     }
1022                 }
1023             }
1024             return false;
1025         }
1026     }

1511         public int parseInto(
1512                 ReadWritablePeriod period, String text, 
1513                 int position, Locale locale) {
1514 
1515             boolean mustParse = (iPrintZeroSetting == PRINT_ZERO_ALWAYS);
1516 
1517             // Shortcut test.
1518             if (position >= text.length()) {
1519                 return mustParse ? ~position : position;
1520             }
1521 
1522             if (iPrefix != null) {
1523                 position = iPrefix.parse(text, position);
1524                 if (position >= 0) {
1525                     // If prefix is found, then the parse must finish.
1526                     mustParse = true;
1527                 } else {
1528                     // Prefix not found, so bail.
1529                     if (!mustParse) {
1530                         // It's okay because parsing of this field is not
1531                         // required. Don't return an error. Fields down the
1532                         // chain can continue on, trying to parse.
1533                         return ~position;
1534                     }
1535                     return position;
1536                 }
1537             }
1538 
1539             int suffixPos = -1;
1540             if (iSuffix != null && !mustParse) {
1541                 // Pre-scan the suffix, to help determine if this field must be
1542                 // parsed.
1543                 suffixPos = iSuffix.scan(text, position);
1544                 if (suffixPos >= 0) {
1545                     // If suffix is found, then parse must finish.
1546                     mustParse = true;
1547                 } else {
1548                     // Suffix not found, so bail.
1549                     if (!mustParse) {
1550                         // It's okay because parsing of this field is not
1551                         // required. Don't return an error. Fields down the
1552                         // chain can continue on, trying to parse.
1553                         return ~suffixPos;
1554                     }
1555                     return suffixPos;
1556                 }
1557             }
1558 
1559             if (!mustParse && !isSupported(period.getPeriodType(), iFieldType)) {
1560                 // If parsing is not required and the field is not supported,
1561                 // exit gracefully so that another parser can continue on.
1562                 return position;
1563             }
1564 
1565             int limit;
1566             if (suffixPos > 0) {
1567                 limit = Math.min(iMaxParsedDigits, suffixPos - position);
1568             } else {
1569                 limit = Math.min(iMaxParsedDigits, text.length() - position);
1570             }
1571 
1572             // validate input number
1573             int length = 0;
1574             int fractPos = -1;
1575             boolean hasDigits = false;
1576             while (length < limit) {
1577                 char c = text.charAt(position + length);
1578                 // leading sign
1579                 if (length == 0 && (c == '-' || c == '+') && !iRejectSignedValues) {
1580                     boolean negative = c == '-';
1581 
1582                     // Next character must be a digit.
1583                     if (length + 1 >= limit || 
1584                         (c = text.charAt(position + length + 1)) < '0' || c > '9')
1585                     {
1586                         break;
1587                     }
1588 
1589                     if (negative) {
1590                         length++;
1591                     } else {
1592                         // Skip the '+' for parseInt to succeed.
1593                         position++;
1594                     }
1595                     // Expand the limit to disregard the sign character.
1596                     limit = Math.min(limit + 1, text.length() - position);
1597                     continue;
1598                 }
1599                 // main number
1600                 if (c >= '0' && c <= '9') {
1601                     hasDigits = true;
1602                 } else {
1603                     if ((c == '.' || c == ',')
1604                          && (iFieldType == SECONDS_MILLIS || iFieldType == SECONDS_OPTIONAL_MILLIS)) {
1605                         if (fractPos >= 0) {
1606                             // can't have two decimals
1607                             break;
1608                         }
1609                         fractPos = position + length + 1;
1610                         // Expand the limit to disregard the decimal point.
1611                         limit = Math.min(limit + 1, text.length() - position);
1612                     } else {
1613                         break;
1614                     }
1615                 }
1616                 length++;
1617             }
1618 
1619             if (!hasDigits) {
1620                 return ~position;
1621             }
1622 
1623             if (suffixPos >= 0 && position + length != suffixPos) {
1624                 // If there are additional non-digit characters before the
1625                 // suffix is reached, then assume that the suffix found belongs
1626                 // to a field not yet reached. Return original position so that
1627                 // another parser can continue on.
1628                 return position;
1629             }
1630 
1631             if (iFieldType != SECONDS_MILLIS && iFieldType != SECONDS_OPTIONAL_MILLIS) {
1632                 // Handle common case.
1633                 setFieldValue(period, iFieldType, parseInt(text, position, length));
1634             } else if (fractPos < 0) {
1635                 setFieldValue(period, SECONDS, parseInt(text, position, length));
1636                 setFieldValue(period, MILLIS, 0);
1637             } else {
1638                 int wholeValue = parseInt(text, position, fractPos - position - 1);
1639                 setFieldValue(period, SECONDS, wholeValue);
1640 
1641                 int fractLen = position + length - fractPos;
1642                 int fractValue;
1643                 if (fractLen <= 0) {
1644                     fractValue = 0;
1645                 } else {
1646                     if (fractLen >= 3) {
1647                         fractValue = parseInt(text, fractPos, 3);
1648                     } else {
1649                         fractValue = parseInt(text, fractPos, fractLen);
1650                         if (fractLen == 1) {
1651                             fractValue *= 100;
1652                         } else {
1653                             fractValue *= 10;
1654                         }
1655                     }
1656                     if (wholeValue < 0) {
1657                         fractValue = -fractValue;
1658                     }
1659                 }
1660 
1661                 setFieldValue(period, MILLIS, fractValue);
1662             }
1663                 
1664             position += length;
1665 
1666             if (position >= 0 && iSuffix != null) {
1667                 position = iSuffix.parse(text, position);
1668             }
1669                 
1670             return position;
1671         }

1674          * @param text text to parse
1675          * @param position position in text
1676          * @param length exact count of characters to parse
1677          * @return parsed int value
1678          */
1679         private int parseInt(String text, int position, int length) {
1680             if (length >= 10) {
1681                 // Since value may exceed max, use stock parser which checks for this.
1682                 return Integer.parseInt(text.substring(position, position + length));
1683             }
1684             if (length <= 0) {
1685                 return 0;
1686             }
1687             int value = text.charAt(position++);
1688             length--;
1689             boolean negative;
1690             if (value == '-') {
1691                 if (--length < 0) {
1692                     return 0;
1693                 }
1694                 negative = true;
1695                 value = text.charAt(position++);
1696             } else {
1697                 negative = false;
1698             }
1699             value -= '0';
1700             while (length-- > 0) {
1701                 value = ((value << 3) + (value << 1)) + text.charAt(position++) - '0';
1702             }
1703             return negative ? -value : value;
1704         }
