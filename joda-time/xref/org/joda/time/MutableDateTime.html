32  /**
33   * MutableDateTime is the standard implementation of a modifiable datetime class.
34   * It holds the datetime as milliseconds from the Java epoch of 1970-01-01T00:00:00Z.
35   * <p>
36   * This class uses a Chronology internally. The Chronology determines how the
37   * millisecond instant value is converted into the date time fields.
38   * The default Chronology is <code>ISOChronology</code> which is the agreed
39   * international standard and compatible with the modern Gregorian calendar.
40   * <p>
41   * Each individual field can be accessed in two ways:
42   * <ul>
43   * <li><code>getHourOfDay()</code>
44   * <li><code>hourOfDay().get()</code>
45   * </ul>
46   * The second technique also provides access to other useful methods on the
47   * field:
48   * <ul>
49   * <li>get numeric value
50   * <li>set numeric value
51   * <li>add to numeric value
52   * <li>add to numeric value wrapping with the field
53   * <li>get text value
54   * <li>get short text value
55   * <li>set text value
56   * <li>field maximum value
57   * <li>field minimum value
58   * </ul>
59   *
60   * <p>
61   * MutableDateTime is mutable and not thread-safe, unless concurrent threads
62   * are not invoking mutator methods.
63   *
64   * @author Guy Allard
65   * @author Brian S O'Neill
66   * @author Stephen Colebourne
67   * @author Mike Schrag
68   * @since 1.0
69   * @see DateTime
70   */
71  public class MutableDateTime
72          extends BaseDateTime
73          implements ReadWritableDateTime, Cloneable, Serializable {

404     /**
405      * Sets the status of rounding to use the specified field and ROUND_FLOOR mode.
406      * A null field will disable rounding.
407      * Once set, the instant is then rounded using the new field and mode.
408      * <p>
409      * Enabling rounding will cause all subsequent calls to {@link #setMillis(long)}
410      * to be rounded. This can be used to control the precision of the instant,
411      * for example by setting a rounding field of minuteOfDay, the seconds and
412      * milliseconds will always be zero.
413      *
414      * @param field rounding field or null to disable
415      */
416     public void setRounding(DateTimeField field) {
417         setRounding(field, MutableDateTime.ROUND_FLOOR);
418     }
419 
420     /**
421      * Sets the status of rounding to use the specified field and mode.
422      * A null field or mode of ROUND_NONE will disable rounding.
423      * Once set, the instant is then rounded using the new field and mode.
424      * <p>
425      * Enabling rounding will cause all subsequent calls to {@link #setMillis(long)}
426      * to be rounded. This can be used to control the precision of the instant,
427      * for example by setting a rounding field of minuteOfDay, the seconds and
428      * milliseconds will always be zero.
429      *
430      * @param field  rounding field or null to disable
431      * @param mode  rounding mode or ROUND_NONE to disable
432      * @throws IllegalArgumentException if mode is unknown, no exception if field is null
433      */
434     public void setRounding(DateTimeField field, int mode) {
435         if (field != null && (mode < ROUND_NONE || mode > ROUND_HALF_EVEN)) {
436             throw new IllegalArgumentException("Illegal rounding mode: " + mode);
437         }
438         iRoundingField = (mode == ROUND_NONE ? null : field);
439         iRoundingMode = (field == null ? ROUND_NONE : mode);
440         setMillis(getMillis());
441     }

443     //-----------------------------------------------------------------------
444     /**
445      * Set the milliseconds of the datetime.
446      * <p>
447      * All changes to the millisecond field occurs via this method.
448      *
449      * @param instant  the milliseconds since 1970-01-01T00:00:00Z to set the
450      * datetime to
451      */
452     public void setMillis(long instant) {
453         switch (iRoundingMode) {
454             case ROUND_NONE:
455                 break;
456             case ROUND_FLOOR:
457                 instant = iRoundingField.roundFloor(instant);
458                 break;
459             case ROUND_CEILING:
460                 instant = iRoundingField.roundCeiling(instant);
461                 break;
462             case ROUND_HALF_FLOOR:
463                 instant = iRoundingField.roundHalfFloor(instant);
464                 break;
465             case ROUND_HALF_CEILING:
466                 instant = iRoundingField.roundHalfCeiling(instant);
467                 break;
468             case ROUND_HALF_EVEN:
469                 instant = iRoundingField.roundHalfEven(instant);
470                 break;
471         }
472         super.setMillis(instant);
473     }

566     /**
567      * Sets the time zone of the datetime, changing the chronology and field values.
568      * <p>
569      * Changing the zone using this method retains the millisecond instant.
570      * The millisecond instant is adjusted in the new zone to compensate.
571      * 
572      * chronology. Setting the time zone does not affect the millisecond value
573      * of this instant.
574      * <p>
575      * If the chronology already has this time zone, no change occurs.
576      *
577      * @param newZone  the time zone to use, null means default zone
578      * @see #setZoneRetainFields
579      */
580     public void setZone(DateTimeZone newZone) {
581         newZone = DateTimeUtils.getZone(newZone);
582         Chronology chrono = getChronology();
583         if (chrono.getZone() != newZone) {
584             setChronology(chrono.withZone(newZone));  // set via this class not super
585         }
586     }

588     /**
589      * Sets the time zone of the datetime, changing the chronology and millisecond.
590      * <p>
591      * Changing the zone using this method retains the field values.
592      * The millisecond instant is adjusted in the new zone to compensate.
593      * <p>
594      * If the chronology already has this time zone, no change occurs.
595      *
596      * @param newZone  the time zone to use, null means default zone
597      * @see #setZone
598      */
599     public void setZoneRetainFields(DateTimeZone newZone) {
600         newZone = DateTimeUtils.getZone(newZone);
601         DateTimeZone originalZone = DateTimeUtils.getZone(getZone());
602         if (newZone == originalZone) {
603             return;
604         }
605         
606         long millis = originalZone.getMillisKeepLocal(newZone, getMillis());
607         setChronology(getChronology().withZone(newZone));  // set via this class not super
608         setMillis(millis);
609     }
