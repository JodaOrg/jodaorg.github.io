Dependency Information

Apache Maven

<dependency>
  <groupId>joda-time</groupId>
  <artifactId>joda-time</artifactId>
  <version>2.10.6</version>
</dependency>

Apache Buildr

'joda-time:joda-time:jar:2.10.6'

Apache Ivy

<dependency org="joda-time" name="joda-time" rev="2.10.6">
  <artifact name="joda-time" type="jar" />
</dependency>

Groovy Grape

@Grapes(
@Grab(group='joda-time', module='joda-time', version='2.10.6')
)

Gradle/Grails

compile 'joda-time:joda-time:2.10.6'

Scala SBT

libraryDependencies += "joda-time" % "joda-time" % "2.10.6"

Leiningen
