16  package org.joda.collect.grid;
17  
18  import java.util.Comparator;
19  
20  import org.joda.collect.grid.Grid.Cell;
21  
22  import com.google.common.base.Objects;
23  
24  /**
25   * Abstract implementation of the {@code Grid.Cell} data structure.
26   * 
27   * @param <V> the type of the value
28   * @author Stephen Colebourne
29   */
30  abstract class AbstractCell<V> implements Cell<V> {
31  
32      /**
33       * Compare by row then column.
34       * 
35       * @param <R> the type of the value
36       * @return the comparator, not null
37       */
38      @SuppressWarnings({ "unchecked", "rawtypes" })
39      static final <R> Comparator<Cell<R>> comparator() {
40          return (Comparator<Cell<R>>) (Comparator) COMPARATOR;
41      }
42      /**
43       * Compare by row then column.
44       */
45      private static final Comparator<Cell<?>> COMPARATOR = new Comparator<Cell<?>>() {
46          @Override
47          public int compare(Cell<?> cell1, Cell<?> cell2) {
48              int thisRow = cell1.getRow();
49              int otherRow = cell2.getRow();
50              int cmp = (thisRow < otherRow ? -1 : (thisRow > otherRow ? 1 : 0));
51              if (cmp == 0) {
52                  int thisCol = cell1.getColumn();
53                  int otherCol = cell2.getColumn();
54                  cmp = (thisCol < otherCol ? -1 : (thisCol > otherCol ? 1 : 0));
55              }
56              return cmp;
57          }
58      };
59  
60      //-----------------------------------------------------------------------
61      /**
62       * Restricted constructor.
63       */
64      AbstractCell() {
65      }
66  
67      //-----------------------------------------------------------------------
68      @Override
69      public boolean equalRowColumn(int row, int column) {
70          return row == getRow() && column == getColumn();
71      }
72  
73      @Override
74      public boolean equalValue(Object value) {
75          return Objects.equal(value, getValue());
76      }
77  
78      //-----------------------------------------------------------------------
79      @Override
80      public boolean equals(Object obj) {
81          if (obj == this) {
82              return true;
83          }
84          if (obj instanceof Cell) {
85              Cell<?> other = (Cell<?>) obj;
86              return getRow() == other.getRow() && getColumn() == other.getColumn()
87                      && Objects.equal(getValue(), other.getValue());
88          }
89          return false;
90      }
91  
92      @Override
93      public int hashCode() {
94          return getRow() ^ Integer.rotateLeft(getColumn(), 16) ^ getValue().hashCode();
95      }
96  
97      @Override
98      public String toString() {
99          return "(" + getRow() + "," + getColumn() + ")=" + getValue();
100     }
101 
102 }
