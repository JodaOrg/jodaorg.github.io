16  package org.joda.collect.grid;
17  
18  import java.util.List;
19  import java.util.Set;
20  
21  import com.google.common.collect.ImmutableCollection;

23  /**
24   * A data structure representing a grid keyed by {@code int} row and {@code int} column.
25   * <p>
26   * A grid has a fixed number of rows and columns, but not all cells must be occupied.
27   * Dense and sparse implementations are provided that handle high and low numbers of cells
28   * relative to the potential capacity.
29   * 
30   * @param <V> the type of the value
31   * @author Stephen Colebourne
32   */
33  public interface Grid<V> {
34  
35      /**
36       * Gets the number of rows in the grid.
37       * <p>
38       * A grid has a fixed number of rows and columns, but not all cells must be occupied.
39       * This returns the row capacity, not the number of occupied rows.
40       * It is guaranteed that {@link #contains(int, int)} will return {@code false}
41       * for indices larger than the row count.
42       * 
43       * @return the number of rows, zero or greater
44       */
45      int rowCount();
46  
47      /**
48       * Gets the number of columns in the grid.
49       * <p>
50       * A grid has a fixed number of rows and columns, but not all cells must be occupied.
51       * This returns the column capacity, not the number of occupied columns.
52       * It is guaranteed that {@link #contains(int, int)} will return {@code false}
53       * for indices larger than the column count.
54       * 
55       * @return the number of columns, zero or greater
56       */
57      int columnCount();
58  
59      /**
60       * Checks if the specified row-column exists.
61       * <p>
62       * This simply checks that the row and column indices are between
63       * zero and the row and column counts.
64       * 
65       * @param row  the row
66       * @param column  the column
67       * @return true if the row-column exists
68       */
69      boolean exists(int row, int column);
70  
71      //-----------------------------------------------------------------------
72      /**
73       * Checks if the grid is full.
74       * <p>
75       * A full grid has a cell at every combination of row and column.
76       * 
77       * @return true if full
78       */
79      boolean isFull();
80  
81      /**
82       * Checks if the grid is empty.
83       * 
84       * @return true if empty
85       */
86      boolean isEmpty();
87  
88      /**
89       * Gets the number of cells that are present.
90       * 
91       * @return the size of the set of cells
92       */
93      int size();
94  
95      /**
96       * Checks if a value is present at the specified row-column.
97       * <p>
98       * If either index does not exist, false is returned.
99       * 
100      * @param row  the row
101      * @param column  the column
102      * @return true if there is a value at the row-column
103      */
104     boolean contains(int row, int column);
105 
106     /**
107      * Checks if the specified value is contained in the grid.
108      * 
109      * @param valueToFind  the value to find, null returns false
110      * @return true if the grid contains the value
111      */
112     boolean containsValue(Object valueToFind);
113 
114     /**
115      * Gets the value at the specified row-column.
116      * <p>
117      * If either index does not exist, null is returned.
118      * 
119      * @param row  the row
120      * @param column  the column
121      * @return the value at the row-column, null if not found
122      */
123     V get(int row, int column);
124 
125     /**
126      * Gets the cell at the specified row-column.
127      * <p>
128      * If either index does not exist, null is returned.
129      * 
130      * @param row  the row
131      * @param column  the column
132      * @return the cell at the row-column, null if not found
133      */
134     Cell<V> cell(int row, int column);
135 
136     //-----------------------------------------------------------------------
137     /**
138      * Checks if this grid equals another grid.
139      * <p>
140      * Two grids are equal if they are the same size and contain the same set of cells.
141      * 
142      * @param obj  the object to compare to, null returns false
143      * @return true if equal
144      */
145     @Override
146     boolean equals(Object obj);
147 
148     /**
149      * Gets a suitable hash code.
150      * <p>
151      * The hash code is {@code rowCount ^ Integer.rotateLeft(columnCount, 16) ^ cells.hashCode()}.
152      * 
153      * @return the hash code
154      */
155     @Override
156     int hashCode();
157 
158     //-----------------------------------------------------------------------
159     /**
160      * Clears the grid.
161      * <p>
162      * The grid will be empty after calling this method.
163      * 
164      * @throws UnsupportedOperationException if read-only
165      */
166     void clear();
167 
168     /**
169      * Puts a value into this grid.
170      * <p>
171      * The value at the specified row-column is set.
172      * Any previous value at the row-column is replaced.
173      * <p>
174      * If either index does not exist, {@code IndexOutOfBoundsException} is thrown.
175      * 
176      * @param row  the row, zero or greater
177      * @param column  the column, zero or greater
178      * @param value  the value to put into the grid, not null
179      * @throws IndexOutOfBoundsException if either index does not exist
180      * @throws UnsupportedOperationException if read-only
181      */
182     void put(int row, int column, V value);
183 
184     /**
185      * Puts all cells from a grid into this grid.
186      * <p>
187      * The value at the specified row-column is set.
188      * Any previous value at the row-column is replaced.
189      * 
190      * @param grid  the grid to put into this grid, not null
191      * @throws IndexOutOfBoundsException if a cell has an invalid index
192      * @throws UnsupportedOperationException if read-only
193      */
194     void putAll(Grid<? extends V> grid);
195 
196     /**
197      * Removes the value at the specified row-column.
198      * <p>
199      * If either index does not exist, no action occurs and false is returned.
200      * 
201      * @param row  the row
202      * @param column  the column
203      * @return true if the grid is altered
204      * @throws UnsupportedOperationException if read-only
205      */
206     boolean remove(int row, int column);
207 
208     //-----------------------------------------------------------------------
209     /**
210      * Gets the complete set of cells.
211      * <p>
212      * If the grid is mutable then cells may be added or removed from the set.
213      * The cells are returned in order, looping around rows, then columns.
214      * <p>
215      * The cell returned from the set iterator may be a mutable {@code Cell}
216      * implementation that cannot be stored beyond the lifetime of an iteration.
217      * 
218      * @return the set of all cells, not null
219      */
220     Set<Cell<V>> cells();
221 
222     /**
223      * Gets the values in order through rows, then columns.
224      * <p>
225      * The returned data structure is an ordered collection.
226      * The values are returned in order, looping around rows, then columns.
227      * 
228      * @return the collection of all values, not null
229      */
230     ImmutableCollection<V> values();
231 
232     /**
233      * Gets the columns of a single row.
234      * <p>
235      * The list will contain all columns from zero to {@code columnCount}.
236      * Where there is no value for a cell, the list will contain null.
237      * <p>
238      * The returned list is immutable, except for {@link List#set(int, Object)},
239      * which adds, updates or deletes from the underlying grid.
240      * 
241      * @param row  the row, zero or greater
242      * @return the columns of the specified row, not null
243      * @throws IndexOutOfBoundsException if the row is invalid
244      */
245     List<V> row(int row);
246 
247     /**
248      * Gets the entire grid of values, by row then column.
249      * <p>
250      * The outer list contains all rows from zero to {@code rowCount}.
251      * Each inner list contains all columns from zero to {@code columnCount}.
252      * Where there is no value for a cell, the value is null.
253      * <p>
254      * The returned list is immutable, except for the {@link List#set(int, Object)}
255      * method on the inner list, which adds, updates or deletes from the underlying grid.
256      * 
257      * @return the entire grid, by row then column, not null
258      */
259     List<List<V>> rows();
260 
261     /**
262      * Gets the rows of a single column.
263      * <p>
264      * The list will contain all rows from zero to {@code rowCount}.
265      * Where data is not present, the list will contain null.
266      * <p>
267      * The returned list is immutable, except for {@link List#set(int, Object)},
268      * which adds, updates or deletes from the underlying grid.
269      * 
270      * @param column  the column, zero or greater
271      * @return the rows of the specified column, not null
272      * @throws IndexOutOfBoundsException if the column is invalid
273      */
274     List<V> column(int column);
275 
276     /**
277      * Gets the entire grid of values, by column then row.
278      * <p>
279      * The outer list contains all columns from zero to {@code columnCount}.
280      * Each inner list contains all rows from zero to {@code rowCount}.
281      * Where there is no value for a cell, the value is null.
282      * <p>
283      * The returned list is immutable, except for the {@link List#set(int, Object)}
284      * method on the inner list, which adds, updates or deletes from the underlying grid.
285      * 
286      * @return the entire grid, by row then column, not null
287      */
288     List<List<V>> columns();
289 
290     //-----------------------------------------------------------------------
291     /**
292      * A cell within the grid compared only using row and column.
293      * 
294      * @param <V> the type of the value
295      */
296     public interface Cell<V> {
297 
298         /**
299          * Gets the row index.
300          * 
301          * @return the row, zero or greater
302          */
303         int getRow();
304 
305         /**
306          * Gets the column index.
307          * 
308          * @return the column, zero or greater
309          */
310         int getColumn();
311 
312         /**
313          * Gets the value of the cell.
314          * 
315          * @return the cell value, not null
316          */
317         V getValue();
318 
319         /**
320          * Checks if the row-column of this cell matches the specified row and column.
321          * 
322          * @param row  the row to check
323          * @param column  the column to check
324          * @return true if equal
325          */
326         boolean equalRowColumn(int row, int column);
327 
328         /**
329          * Checks if the value of this cell matches the specified value.
330          * 
331          * @param value  the row to check, null returns false
332          * @return true if equal
333          */
334         boolean equalValue(Object value);
335 
336         /**
337          * Checks if this cell equals another cell.
338          * <p>
339          * Two cells are equal if they have equal row, column and value.
340          * 
341          * @param obj  the object to compare to, null returns false
342          * @return true if equal
343          */
344         @Override
345         boolean equals(Object obj);
346 
347         /**
348          * Gets a suitable hash code.
349          * <p>
350          * The hash code is {@code row ^ Integer.rotateLeft(column, 16) ^ value.hashCode()}.
351          * 
352          * @return the hash code
353          */
354         @Override
355         int hashCode();
356     }
