75      /**
76       * Creates a {@code DenseGrid} copying from another grid.
77       *
78       * @param <V> the type of the value
79       * @param grid  the grid to copy, not null
80       * @return the mutable grid, not null
81       */
82      @SuppressWarnings("unchecked")
83      public static <V> DenseGrid<V> create(Grid<? extends V> grid) {
84          if (grid == null) {
85              throw new IllegalArgumentException("Grid must not be null");
86          }
87          if (grid instanceof DenseImmutableGrid) {
88              return new DenseGrid<V>((DenseImmutableGrid<V>) grid);
89          }
90          DenseGrid<V> created = DenseGrid.create(grid.rowCount(), grid.columnCount());
91          created.putAll(grid);
92          return created;
93      }

95      /**
96       * Creates a {@code DenseGrid} copying from an array.
97       * <p>
98       * The row count and column count are derived from the maximum size of the array.
99       * The grid is initialized from the non-null values.
100      *
101      * @param <V> the type of the value
102      * @param array  the array, first by row, then by column
103      * @return the mutable grid, not null
104      */
105     public static <V> DenseGrid<V> create(V[][] array) {
106         if (array == null) {
107             throw new IllegalArgumentException("Array must not be null");
108         }
109         int rowCount = array.length;
110         if (rowCount == 0) {
111             return new DenseGrid<V>(0, 0);
112         }
113         int columnCount = array[0].length;
114         for (int i = 1; i < rowCount; i++) {
115             columnCount = Math.max(columnCount, array[i].length);
116         }
117         DenseGrid<V> created = DenseGrid.create(rowCount, columnCount);
118         for (int row = 0; row < array.length; row++) {
119             V[] rowValues = array[row];
120             for (int column = 0; column < rowValues.length; column++) {
121                 V cellValue = rowValues[column];
122                 if (cellValue != null) {
123                     created.values[row * columnCount + column] = cellValue;
124                     created.size++;
125                 }
126             }
127         }
128         return created;
129     }

212     static class Cells<V> extends AbstractSet<Cell<V>> {
213         private final DenseGrid<V> grid;

215         public Cells(DenseGrid<V> grid) {
216             this.grid = grid;

219         @Override
220         public int size() {
221             return grid.size;
222         }

226             Cell<?> cell = (Cell<?>) obj;
227             return Objects.equal(cell.getValue(), grid.get(cell.getRow(), cell.getColumn()));
228         }

230         @Override
231         public Iterator<Cell<V>> iterator() {
232             return new Iterator<Cell<V>>() {
233                 private MutableCell<V> cell = new MutableCell<V>();
234                 private int count;
235                 private int current = -1;
236 
237                 @Override
238                 public boolean hasNext() {
239                     return (count < grid.size);
240                 }
241                 @Override
242                 public Cell<V> next() {
243                     if (hasNext() == false) {
244                         throw new NoSuchElementException("No more elements");
245                     }
246                     current++;
247                     for ( ; current < grid.values.length; current++) {
248                         if (grid.values[current] != null) {
249                             break;
250                         }
251                     }
252                     V value = grid.values[current];
253                     count++;
254                     cell.set(current / grid.columnCount, current % grid.columnCount, value);
255                     return cell;
256                 }
257                 @Override
258                 public void remove() {
259                     if (current < 0) {
260                         throw new IllegalStateException("Unable to remove, next() not called yet");
261                     }
262                     if (grid.values[current] == null) {
263                         throw new IllegalStateException("Unable to remove, element has been removed");
264                     }
265                     grid.values[current] = null;
266                     grid.size--;
267                     count--;
268                 }
269             };
270         }
271 
272         @Override
273         public boolean add(Cell<V> cell) {
274             Preconditions.checkArgument(cell != null, "Cell must not be null");
275             int oldSize = grid.size;
276             grid.put(cell.getRow(), cell.getColumn(), cell.getValue());
277             return grid.size > oldSize;
278         }
279 
280         @Override
281         public boolean remove(Object obj) {
282             Cell<?> cell = (Cell<?>) obj;
283             return grid.remove(cell.getRow(), cell.getColumn());
284         }

358     static class Inner<V> extends AbstractList<V> {
359         private final DenseGrid<V> grid;
360         private final int size;
361         private final int offset;
362         private final int gap;
363 
364         Inner(DenseGrid<V> grid, int offset, int size, int gap) {
365             this.grid = grid;
366             this.size = size;
367             this.offset = offset;
368             this.gap = gap;
369         }
370 
371         @Override
372         public int size() {
373             return size;
374         }
375 
376         @Override
377         public V get(int index) {
378             Preconditions.checkElementIndex(index, size);
379             int arrayIndex = index * gap + offset;
380             return grid.values[arrayIndex];
381         }
382 
383         @Override
384         public V set(int index, V newValue) {
385             Preconditions.checkElementIndex(index, size);
386             int arrayIndex = index * gap + offset;
387             V old = grid.values[arrayIndex];
388             grid.values[arrayIndex] = newValue;
389             if (old == null && newValue != null) {
390                 grid.size++;
391             } else if (old != null && newValue == null) {
392                 grid.size--;
393             }
394             return old;
395         }
396     }

405     @Override
406     public void put(int row, int column, V value) {
407         if (exists(row, column) == false) {
408             throw new IndexOutOfBoundsException("Invalid row-column: " + row + "," + column);
409         }
410         if (value == null) {
411             throw new IllegalArgumentException("Value must not be null");
412         }
413         Object current = values[row * columnCount + column];
414         values[row * columnCount + column] = value;
415         if (current == null) {
416             size++;
417         }
418     }

420     @Override
421     public void putAll(Grid<? extends V> grid) {
422         if (grid == null) {
423             throw new IllegalArgumentException("Grid must nor be null");
424         }
425         for (Cell<? extends V> cell : grid.cells()) {
426             put(cell.getRow(), cell.getColumn(), cell.getValue());
427         }
428     }

430     @Override
431     public boolean remove(int row, int column) {
432         V current = get(row, column);
433         if (current != null) {
434             values[row * columnCount + column] = null;
435             size--;
436             return true;
437         }
438         return false;

464     @Override
465     public int hashCode() {
466         int hash = 0;
467         for (int i = 0; i < values.length; i++) {
468             Object value = values[i];
469             if (value != null) {
470                 int row = i / columnCount;
471                 int column = i % columnCount;
472                 hash += (row ^ Integer.rotateLeft(column, 16) ^ value.hashCode());
473             }
474         }
475         return rowCount ^ Integer.rotateLeft(columnCount, 16) ^ hash;
476     }
