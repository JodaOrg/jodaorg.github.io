209     /**
210      * Resolves the field type.
211      */
212     public void resolveType() {
213         if (getTypeStyle() == null) {
214             setTypeStyle("");
215         }
216         final String fieldType = getFieldType();
217         String rawType = fieldType;
218         String generics = "";
219         if (rawType.contains("<")) {
220             rawType = rawType.substring(0, fieldType.indexOf('<'));
221             generics = fieldType.substring(fieldType.indexOf('<'));
222         }
223         if (getTypeStyle().equals("smart")) {
224             setType(fieldType);
225         } else if (getTypeStyle().length() > 0) {
226             if (getTypeStyle().contains("<>")) {
227                 setType(getTypeStyle().replace("<>", generics));
228             } else {
229                 setType(getTypeStyle() + generics);
230             }
231         } else {
232             setType(fieldType);
233         }
234     }

481     /**
482      * Resolves the getter generator.
483      */
484     public void resolveGetterGen() {
485         if (getGetStyle() == null) {
486             setGetStyle("");
487         }
488         String style = getGetStyle();
489         String access = "public";
490         if (style.equals("private")) {
491             style = "smart";
492             access = "private";
493         }
494         if (style.equals("get")) {
495             getterGen = GetterGen.GetGetterGen.PUBLIC;
496         } else if (style.equals("is")) {
497             getterGen = GetterGen.IsGetterGen.PUBLIC;
498         } else if (style.equals("smart")) {
499             if (bean.isImmutable()) {
500                 String clone = config.getImmutableGetClones().get(getFieldTypeRaw());
501                 if ("clone".equals(clone)) {
502                     getterGen = GetterGen.CloneGetterGen.of(access);
503                 } else if ("cloneCast".equals(clone)) {
504                     getterGen = GetterGen.CloneCastGetterGen.of(access);
505                 } else if (getType().equals("boolean")) {
506                     getterGen = GetterGen.IsGetterGen.of(access);
507                 } else {
508                     getterGen = GetterGen.GetGetterGen.of(access);
509                 }
510             } else if (getType().equals("boolean")) {
511                 getterGen = GetterGen.IsGetterGen.of(access);
512             } else {
513                 getterGen = GetterGen.GetGetterGen.of(access);
514             }
515         } else if (style.equals("")) {
516             getterGen = GetterGen.NoGetterGen.INSTANCE;
517         } else if (style.equals("clone")) {
518             getterGen = GetterGen.CloneGetterGen.PUBLIC;
519         } else if (style.equals("cloneCast")) {
520             getterGen = GetterGen.CloneCastGetterGen.PUBLIC;
521         } else if (style.equals("manual")) {
522             getterGen = GetterGen.ManualGetterGen.INSTANCE;
523         } else {
524             throw new RuntimeException("Unable to locate setter generator '" + style + "'" +
525                     " in " + getBean().getTypeRaw() + "." + getPropertyName());
526         }
527     }

537     //-----------------------------------------------------------------------
538     /**
539      * Resolves the setter generator.
540      */
541     public void resolveSetterGen() {
542         if (getSetStyle() == null) {
543             setSetStyle("");
544         }
545         String style = getSetStyle().replace("\\n", "\n");
546         String access = "public";
547         if (style.equals("private")) {
548             style = "smart";
549             access = "private";
550         }
551         if (style.equals("set")) {
552             setterGen = SetterGen.SetSetterGen.PUBLIC;
553         } else if (style.equals("setClearAddAll")) {
554             setterGen = new SetterGen.PatternSetterGen("$field.clear();\n$field.addAll($value);");
555         } else if (style.equals("setClearPutAll")) {
556             setterGen = new SetterGen.PatternSetterGen("$field.clear();\n$field.putAll($value);");
557         } else if (style.equals("smart")) {
558             if (isDerived()) {
559                 setterGen = SetterGen.NoSetterGen.INSTANCE;
560             } else if (isFinal()) {
561                 if (isCollectionType()) {
562                     setterGen = new SetterGen.PatternSetterGen("$field.clear();\n$field.addAll($value);", access);
563                 } else if (isMapType()) {
564                     setterGen = new SetterGen.PatternSetterGen("$field.clear();\n$field.putAll($value);", access);
565                 } else {
566                     setterGen = SetterGen.NoSetterGen.INSTANCE;
567                 }
568             } else {
569                 setterGen = SetterGen.SetSetterGen.of(access);
570             }
571         } else if (style.equals("")) {
572             setterGen = SetterGen.NoSetterGen.INSTANCE;
573         } else if (style.equals("manual")) {
574             setterGen = SetterGen.NoSetterGen.INSTANCE;
575         } else if (style.contains("$field") || style.contains("$value")) {
576             if (style.contains("$field") || style.contains("\n")) {
577                 setterGen = new SetterGen.PatternSetterGen(style);
578             } else {
579                 setterGen = new SetterGen.PatternSetterGen("$field = " + style);
580             }
581         } else {
582             throw new RuntimeException("Unable to locate setter generator '" + style + "'" +
583                     " in " + getBean().getTypeRaw() + "." + getPropertyName());
584         }
585     }

607     /**
608      * Resolves the copy generator.
609      */
610     public void resolveCopyGen() {
611         if (getBean().isMutable() && getBean().isBuilderScopePublic() == false) {
612             return;  // no copying
613         }
614         if (config.getInvalidImmutableTypes().contains(getFieldTypeRaw())) {
615             throw new RuntimeException("Invalid collection type for immutable bean: " + getFieldTypeRaw() +
616                     " in " + getBean().getTypeRaw() + "." + getPropertyName());
617         }
618         if (isDerived()) {
619             copyGen = CopyGen.NoCopyGen.INSTANCE;
620         } else {
621             CopyGen copier = config.getCopyGenerators().get(getFieldTypeRaw());
622             if (copier != null) {
623                 copyGen = copier;
624             } else {
625                 String clone = config.getImmutableGetClones().get(getFieldTypeRaw());
626                 if (clone != null) {
627                     if (clone.equals("clone")) {
628                         copyGen = CopyGen.CLONE;
629                     } else {
630                         copyGen = CopyGen.CLONE_CAST;
631                     }
632                 } else {
633                     copyGen = CopyGen.ASSIGN;
634                 }
635             }
636         }
637     }

712     /**
713      * Gets the read-write flag.
714      * 
715      * @return the read write
716      */
717     public PropertyStyle getStyle() {
718         if (isDerived()) {
719             return PropertyStyle.DERIVED;
720         }
721         if (getBean().isImmutable()) {
722             return PropertyStyle.IMMUTABLE;
723         }
724         if (getGetStyle().length() > 0 && getSetStyle().length() > 0 && (getSetterGen().isSetterGenerated(this) || getSetStyle().equals("manual"))) {
725             return PropertyStyle.READ_WRITE;
726         }
727         if (getGetStyle().length() > 0) {
728             if (bean.isBuilderScopePublic()) {
729                 return PropertyStyle.READ_ONLY_BUILDABLE;
730             } else {
731                 return PropertyStyle.READ_ONLY;
732             }
733         }
734         if (getSetStyle().length() > 0) {
735             return PropertyStyle.WRITE_ONLY;
736         }
737         throw new RuntimeException("Property must have a getter or setter: " +
738                 getBean().getTypeRaw() + "." + getPropertyName());
739     }
