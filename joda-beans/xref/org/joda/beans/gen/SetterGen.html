18  import java.util.ArrayList;
19  import java.util.Collections;
20  import java.util.List;

22  /**
23   * A generator of set methods.
24   * 
25   * @author Stephen Colebourne
26   */
27  abstract class SetterGen {
28  
29      /**
30       * Checks if a setter method is possible.
31       * 
32       * @param prop  the property data, not null
33       * @return true if a setter is possible
34       */
35      abstract boolean isSetterGenerated(GeneratableProperty prop);
36  
37      /**
38       * Generates the setter method.
39       * 
40       * @param indent  the indent to use, not null
41       * @param prop  the property data, not null
42       * @return the generated code, not null
43       */
44      abstract List<String> generateSetter(String indent, GeneratableProperty prop);
45  
46      /**
47       * Generates the setter method invocation.
48       * This is just the method name.
49       * 
50       * @param prop  the property data, not null
51       * @param newValue  the new value, not null
52       * @return the generated code, null if no setter
53       */
54      String generateSetInvoke(GeneratableProperty prop, String newValue) {
55          return "set" + prop.getUpperName() + "(" + newValue + ")";
56      }
57  
58      //-----------------------------------------------------------------------
59      static final class SetSetterGen extends SetterGen {
60          static final SetSetterGen PUBLIC = new SetSetterGen("public");
61          static final SetSetterGen PRIVATE = new SetSetterGen("private");
62          private final String access;
63          static SetSetterGen of(String access) {
64              return (access.equals("private") ? PRIVATE : PUBLIC);
65          }
66          private SetSetterGen(String access) {
67              this.access = access;
68          }
69          @Override
70          boolean isSetterGenerated(GeneratableProperty prop) {
71              return true;
72          }
73          @Override
74          List<String> generateSetter(String indent, GeneratableProperty prop) {
75              List<String> list = new ArrayList<String>();
76              list.add("\t/**");
77              list.add("\t * Sets " + prop.getFirstComment());
78              for (String comment : prop.getComments()) {
79                  list.add("\t * " + comment);
80              }
81              list.add("\t * @param " + prop.getPropertyName() + "  the new value of the property" + prop.getNotNullJavadoc());
82              list.add("\t */");
83              if (prop.isOverrideSet()) {
84                  list.add("\t@Override");
85              }
86              if (prop.isDeprecated()) {
87                  list.add("\t@Deprecated");
88              }
89              list.add("\t" + access + " void set" + prop.getUpperName() + "(" + prop.getType() +  " " + prop.getPropertyName() + ") {");
90              if (prop.isValidated()) {
91                  list.add("\t\t" + prop.getValidationMethodName() + "(" + prop.getPropertyName() + ", \"" + prop.getPropertyName() + "\");");
92              }
93              list.add("\t\tthis." + prop.getFieldName() + " = " + prop.getPropertyName() + ";");
94              list.add("\t}");
95              list.add("");
96              return list;
97          }
98      }

100     static class PatternSetterGen extends SetterGen {
101         private final String setPattern;
102         private final String access;
103         PatternSetterGen(String setPattern) {
104             this(setPattern, "public");
105         }
106         PatternSetterGen(String setPattern, String access) {
107             this.setPattern = setPattern;
108             this.access = access;
109         }
110         @Override
111         boolean isSetterGenerated(GeneratableProperty prop) {
112             return true;
113         }
114         @Override
115         List<String> generateSetter(String indent, GeneratableProperty prop) {
116             List<String> list = new ArrayList<String>();
117             list.add(indent + "/**");
118             list.add(indent + " * Sets " + prop.getFirstComment());
119             for (String comment : prop.getComments()) {
120                 list.add(indent + " * " + comment);
121             }
122             list.add(indent + " * @param " + prop.getPropertyName() + "  the new value of the property" + prop.getNotNullJavadoc());
123             list.add(indent + " */");
124             if (prop.isOverrideSet()) {
125                 list.add("\t@Override");
126             }
127             if (prop.isDeprecated()) {
128                 list.add(indent + "@Deprecated");
129             }
130             list.add(indent + access + " void set" + prop.getUpperName() + "(" + prop.getType() +  " " + prop.getPropertyName() + ") {");
131             if (prop.isValidated()) {
132                 list.add("\t\t" + prop.getValidationMethodName() + "(" + prop.getPropertyName() + ", \"" + prop.getPropertyName() + "\");");
133             }
134             final String[] split = setPattern.split("\n");
135             for (String line : split) {
136                 line = line.replace("$field", "this." + prop.getFieldName());
137                 line = line.replace("$value", prop.getPropertyName());
138                 line = line.replace("<>", prop.getTypeGenerics());
139                 if (split.length == 1 && line.endsWith(";") == false) {
140                     line += ";";
141                 }
142                 list.add(indent + "\t" + line);
143             }
144             list.add(indent + "}");
145             list.add("");
146             return list;
147         }
148     }

150     static class NoSetterGen extends SetterGen {

152         @Override

154             return false;
