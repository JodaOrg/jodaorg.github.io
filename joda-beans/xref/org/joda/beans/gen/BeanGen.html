64      /** Pattern to find bean type. */
65      private static final Pattern BEAN_TYPE = Pattern.compile(".*class +(" +

94      /** The style pattern. */

127     /**
128      * Constructor.
129      * @param content  the content to process, not null
130      * @param config  the config to use, not null
131      */
132     BeanGen(List<String> content, BeanGenConfig config) {
133         this.content = content;
134         this.config = config;
135         int beanDefIndex = parseBeanDefinition();
136         if (beanDefIndex >= 0) {
137             this.data = new GeneratableBean();
138             this.data.getCurrentImports().addAll(parseImports(beanDefIndex));
139             this.data.setImportInsertLocation(parseImportLocation(beanDefIndex));
140             this.data.setBeanStyle(parseBeanStyle(beanDefIndex));
141             if (data.isBeanStyleValid() == false) {
142                 throw new RuntimeException("Invalid bean style: " + data.getBeanStyle());
143             }
144             this.data.setBeanBuilderScope(parseBeanBuilderScope(beanDefIndex));
145             if (data.isBeanBuilderScopeValid() == false) {
146                 throw new RuntimeException("Invalid bean builder scope: " + data.getBeanStyle());
147             }
148             this.data.setImmutableConstructor(parseImmutableConstructor(beanDefIndex));
149             this.data.setConstructable(parseConstructable(beanDefIndex));
150             this.data.setTypeParts(parseBeanType(beanDefIndex));
151             this.data.setSuperTypeParts(parseBeanSuperType(beanDefIndex));
152             if (parseBeanHierarchy(beanDefIndex).equals("immutable")) {
153                 this.data.setImmutable(true);
154                 this.data.setConstructorStyle(CONSTRUCTOR_BY_BUILDER);
155             } else if (this.data.getImmutableConstructor() == CONSTRUCTOR_NONE) {
156                 if (data.isImmutable()) {
157                     if (data.isTypeFinal()) {
158                         this.data.setConstructorStyle(CONSTRUCTOR_BY_ARGS);
159                     } else {
160                         this.data.setConstructorStyle(CONSTRUCTOR_BY_BUILDER);
161                     }
162                 } else {
163                     this.data.setConstructorStyle(CONSTRUCTOR_BY_BUILDER);
164                 }
165             } else {
166                 this.data.setConstructorStyle(this.data.getImmutableConstructor());
167             }
168             if (data.isImmutable()) {
169                 this.data.setImmutableValidator(parseImmutableValidator(beanDefIndex));
170             }
171             this.properties = parseProperties(data);
172             this.autoStartIndex = parseStartAutogen();
173             this.autoEndIndex = parseEndAutogen();
174             this.insertRegion = content.subList(autoStartIndex + 1, autoEndIndex);
175             this.data.setManualClone(parseManualClone(beanDefIndex));
176             this.data.setManualEqualsHashCode(parseManualEqualsHashCode(beanDefIndex));
177             this.data.setManualToStringCode(parseManualToStringCode(beanDefIndex));
178             if (data.isImmutable()) {
179                 for (PropertyGen prop : properties) {
180                     if (prop.getData().isDerived() == false && prop.getData().isFinal() == false) {
181                         throw new RuntimeException("ImmutableBean must have final properties: " + data.getTypeRaw() + "." + prop.getData().getFieldName());
182                     }
183                 }
184             } else if (data.getImmutableConstructor() > CONSTRUCTOR_NONE) {
185                 throw new RuntimeException("Mutable beans must not specify @ImmutableConstructor: " + data.getTypeRaw());
186             }
187         } else {
188             this.autoStartIndex = -1;
189             this.autoEndIndex = -1;
190             this.insertRegion = null;
191             this.data = null;
192             this.properties = null;
193         }
194     }

196     //-----------------------------------------------------------------------
197     void process() {
198         if (insertRegion != null) {
199             data.ensureImport(BeanDefinition.class);
200             if (properties.size() > 0) {
201                 data.ensureImport(PropertyDefinition.class);
202             }
203             removeOld();
204             if (data.isRootClass() && data.isExtendsDirectBean()) {
205                 data.ensureImport(DirectBean.class);
206             }
207             insertRegion.add("\t///CLOVER:OFF");
208             generateMeta();
209             generateImmutableBuilderMethod();
210             generateArgBasedConstructor();
211             generateBuilderBasedConstructor();
212             generateMetaBean();
213             generatePropertyByName();
214             generatePropertyNames();
215             generateGettersSetters();
216             generateSeparator();
217             generateImmutableToBuilder();
218             generateClone();
219             generateEquals();
220             generateHashCode();
221             generateToString();
222             generateMetaClass();
223             generateBuilderClass();
224             insertRegion.add("\t///CLOVER:ON");
225             resolveImports();
226             resolveIndents();
227         }
228     }

230     private void resolveImports() {
231         if (data.getNewImports().size() > 0) {
232             int pos = data.getImportInsertLocation() + 1;
233             for (String imp : data.getNewImports()) {
234                 content.add(pos++, "import " + imp + ";");
235             }
236         }
237     }

239     private void resolveIndents() {
240         for (ListIterator<String> it = content.listIterator(); it.hasNext(); ) {
241             it.set(it.next().replace("\t", config.getIndent()));
242         }
243     }

246     private int parseBeanDefinition() {
247         for (int index = 0; index < content.size(); index++) {
248             String line = content.get(index).trim();
249             if (line.startsWith("@BeanDefinition")) {
250                 return index;
251             }
252         }
253         return -1;
254     }

256     private Set<String> parseImports(int defLine) {
257         Set<String> imports = new HashSet<String>();
258         for (int index = 0; index < defLine; index++) {
259             if (content.get(index).startsWith("import ")) {
260                 String imp = content.get(index).substring(7).trim();
261                 imp = imp.substring(0, imp.indexOf(';'));
262                 if (imp.endsWith(".*") == false) {
263                     imports.add(imp);
264                 }
265             }
266         }
267         return imports;
268     }

270     private int parseImportLocation(int defLine) {
271         int location = 0;
272         for (int index = 0; index < defLine; index++) {
273             if (content.get(index).startsWith("import ") || content.get(index).startsWith("package ")) {
274                 location = index;
275             }
276         }
277         return location;

281         String line = content.get(defLine).trim();
282         Matcher matcher = STYLE_PATTERN.matcher(line);
283         if (matcher.matches()) {

316     private String[] parseBeanType(int defLine) {
317         Matcher matcher = BEAN_TYPE.matcher("");
318         for (int index = defLine; index < content.size(); index++) {
319             String line = content.get(index);
320             matcher.reset(line);
321             if (matcher.matches()) {
322                 int startName = matcher.start(1);
323                 String fnl = line.substring(0, startName).contains(" final ") ? " final " : null;
324                 return new String[] {fnl, matcher.group(1), matcher.group(2), matcher.group(3), matcher.group(4),
325                         matcher.group(5), matcher.group(6), matcher.group(7), matcher.group(8)};
326             }
327             if (line.contains(AUTOGENERATED_START_TEXT)) {
328                 break;
329             }
330         }
331         throw new RuntimeException("Unable to locate bean class name");
332     }

334     private String[] parseBeanSuperType(int defLine) {
335         // need to start searching beyond the full type to avoid 'extends' having two meanings
336         String fullType = data.getType();
337         // search for implements
338         Matcher matcherImplements = SUPER_IMPL_TYPE.matcher("");
339         boolean matchedType = false;
340         for (int index = defLine; index < content.size(); index++) {
341             String line = content.get(index);
342             if (matchedType == false) {
343                 if (line.contains(fullType) == false) {
344                     continue;
345                 }
346                 matchedType = true;
347                 line = line.substring(line.indexOf(fullType) + fullType.length());
348             }
349             matcherImplements.reset(line);
350             if (matcherImplements.matches()) {
351                 return new String[] {matcherImplements.group(1)};
352             }
353             if (line.contains(AUTOGENERATED_START_TEXT)) {
354                 break;
355             }
356         }
357         // search for extends
358         Matcher matcherExtends = SUPER_TYPE.matcher("");
359         matchedType = false;
360         for (int index = defLine; index < content.size(); index++) {
361             String line = content.get(index);
362             if (matchedType == false) {
363                 if (line.contains(fullType) == false) {
364                     continue;
365                 }
366                 matchedType = true;
367                 line = line.substring(line.indexOf(fullType) + fullType.length());
368             }
369             matcherExtends.reset(line);
370             if (matcherExtends.matches()) {
371                 return new String[] {matcherExtends.group(1), matcherExtends.group(2), matcherExtends.group(3), matcherExtends.group(4), matcherExtends.group(5)};
372             }
373             if (line.contains(AUTOGENERATED_START_TEXT)) {
374                 break;
375             }
376         }
377         throw new RuntimeException("Unable to locate bean superclass");
378     }

380     private int parseImmutableConstructor(int defLine) {
381         int found = CONSTRUCTOR_NONE;
382         for (int index = defLine; index < content.size(); index++) {
383             if (content.get(index).trim().equals("@ImmutableConstructor")) {
384                 if (found > 0) {
385                     throw new RuntimeException("Only one @ImmutableConstructor may be specified");
386                 }
387                 found = CONSTRUCTOR_BY_ARGS;
388                 if (index + 1 < content.size()) {
389                     String nextLine = content.get(index + 1);
390                     if (nextLine.contains("Builder ") || nextLine.contains("Builder<")) {
391                         found = CONSTRUCTOR_BY_BUILDER;
392                     }
393                 }
394             }
395         }
396         return found;
397     }

399     private String parseImmutableValidator(int defLine) {
400         boolean found = false;
401         for (int index = defLine; index < content.size(); index++) {
402             if (content.get(index).trim().equals("@ImmutableValidator")) {
403                 if (found) {
404                     throw new RuntimeException("Only one @ImmutableValidator may be specified");
405                 }
406                 found = true;
407                 if (index + 1 < content.size()) {
408                     String nextLine = content.get(index + 1);
409                     Matcher matcher = VALIDATOR_PATTERN.matcher(nextLine);
410                     if (matcher.matches()) {
411                         return matcher.group(1);
412                     }
413                     throw new RuntimeException("@ImmutableValidator method must be private void and no-args");
414                 }
415             }
416         }
417         return null;
418     }

420     private List<PropertyGen> parseProperties(GeneratableBean data) {
421         List<PropertyGen> props = new ArrayList<PropertyGen>();
422         for (int index = 0; index < content.size(); index++) {
423             String line = content.get(index).trim();
424             if (line.startsWith("@PropertyDefinition")) {
425                 PropertyGen prop = new PropertyGen(this, content, index, false);
426                 props.add(prop);
427                 data.getProperties().add(prop.getData());
428             } else if (line.startsWith("@DerivedProperty")) {
429                 PropertyGen prop = new PropertyGen(this, content, index, true);
430                 props.add(prop);
431                 data.getProperties().add(prop.getData());
432             }
433         }
434         return props;
435     }

437     private int parseStartAutogen() {
438         for (int index = 0; index < content.size(); index++) {
439             String line = content.get(index).trim();
440             if (line.contains(" AUTOGENERATED START ")) {
441                 content.set(index, AUTOGENERATED_START);
442                 return index;
443             }
444         }
445         for (int index = content.size() - 1; index >= 0; index--) {
446             String line = content.get(index).trim();
447             if (line.equals("}")) {
448                 content.add(index, AUTOGENERATED_START);
449                 return index;
450             }
451             if (line.length() > 0) {
452                 break;
453             }
454         }
455         throw new RuntimeException("Unable to locate start autogeneration point");
456     }

458     private int parseEndAutogen() {
459         for (int index = autoStartIndex; index < content.size(); index++) {
460             String line = content.get(index).trim();
461             if (line.contains(" AUTOGENERATED END ")) {
462                 content.set(index, AUTOGENERATED_END);
463                 return index;
464             }
465         }
466         content.add(autoStartIndex + 1, AUTOGENERATED_END);
467         return autoStartIndex + 1;
468     }

474     private boolean parseManualClone(int defLine) {
475         for (int index = defLine; index < autoStartIndex; index++) {
476             String line = content.get(index).trim();
477             if (line.startsWith("public ") && line.endsWith(" clone() {")) {
478                 return true;
479             }
480         }
481         for (int index = autoEndIndex; index < content.size(); index++) {
482             String line = content.get(index).trim();
483             if (line.startsWith("public ") && line.endsWith(" clone() {")) {
484                 return true;
485             }
486         }
487         return false;
488     }

490     private boolean parseManualEqualsHashCode(int defLine) {
491         for (int index = defLine; index < autoStartIndex; index++) {
492             String line = content.get(index).trim();
493             if (line.equals("public int hashCode() {") || (line.startsWith("public boolean equals(") && line.endsWith(") {"))) {
494                 return true;
495             }
496         }
497         for (int index = autoEndIndex; index < content.size(); index++) {
498             String line = content.get(index).trim();
499             if (line.equals("public int hashCode() {") || (line.startsWith("public boolean equals(") && line.endsWith(") {"))) {
500                 return true;
501             }
502         }
503         return false;
504     }

506     private boolean parseManualToStringCode(int defLine) {
507         for (int index = defLine; index < autoStartIndex; index++) {
508             String line = content.get(index).trim();
509             if (line.equals("public String toString() {")) {
510                 return true;
511             }
512         }
513         for (int index = autoEndIndex; index < content.size(); index++) {
514             String line = content.get(index).trim();
515             if (line.equals("public String toString() {")) {
516                 return true;
517             }
518         }
519         return false;

537     private void generateImmutableBuilderMethod() {
538         if (data.isConstructable() &&
539                 ((data.isImmutable() && data.isEffectiveBuilderScopePublic()) || (data.isMutable() && data.isBuilderScopePublic()))) {
540             insertRegion.add("\t/**");
541             insertRegion.add("\t * Returns a builder used to create an instance of the bean.");
542             if (data.isTypeGeneric()) {
543                 for (int j = 0; j < data.getTypeGenericCount(); j++) {
544                     insertRegion.add("\t * @param " + data.getTypeGenericName(j, true) + "  the type");
545                 }
546             }
547             insertRegion.add("\t * @return the builder, not null");
548             insertRegion.add("\t */");
549             if (data.isTypeGeneric()) {
550                 insertRegion.add("\tpublic static " + data.getTypeGeneric(true) + " " + data.getTypeRaw() + ".Builder" + data.getTypeGenericName(true) + " builder() {");
551             } else {
552                 insertRegion.add("\tpublic static " + data.getTypeRaw() + ".Builder builder() {");
553             }
554             insertRegion.add("\t\treturn new " + data.getTypeRaw() + ".Builder" + data.getTypeGenericName(true) + "();");
555             insertRegion.add("\t}");
556             insertRegion.add("");
557         }
558     }

560     private void generateBuilderBasedConstructor() {
561         if (data.getConstructorStyle() == CONSTRUCTOR_BY_BUILDER && data.getImmutableConstructor() == CONSTRUCTOR_NONE && 
562                 ((data.isMutable() && data.isBuilderScopePublic()) || data.isImmutable())) {
563             List<PropertyGen> nonDerived = nonDerivedProperties();
564             String scope = (data.isTypeFinal() ? "private" : "protected");
565             // signature
566             insertRegion.add("\t/**");
567             insertRegion.add("\t * Restricted constructor.");
568             insertRegion.add("\t * @param builder  the builder to copy from, not null");
569             insertRegion.add("\t */");
570             insertRegion.add("\t" + scope + " " + data.getTypeRaw() + "(" + data.getTypeRaw() + ".Builder" + data.getTypeGenericName(true) + " builder) {");
571             // super
572             if (data.isSubClass()) {
573                 insertRegion.add("\t\tsuper(builder);");
574             }
575             // validate
576             for (PropertyGen prop : properties) {
577                 if (prop.getData().isValidated()) {
578                     insertRegion.add("\t\t" + prop.getData().getValidationMethodName() +
579                             "(builder." + prop.generateBuilderFieldName() +
580                             ", \"" + prop.getData().getPropertyName() + "\");");
581                 }
582             }
583             // assign
584             if (data.isImmutable()) {
585                 // assign
586                 for (int i = 0; i < nonDerived.size(); i++) {
587                     insertRegion.addAll(nonDerived.get(i).generateConstructorAssign("builder."));
588                 }
589             } else {
590                 for (int i = 0; i < nonDerived.size(); i++) {
591                     PropertyGen propGen = nonDerived.get(i);
592                     GeneratableProperty prop = propGen.getData();
593                     if (prop.isCollectionType()) {
594                         if (prop.isNotNull()) {
595                             insertRegion.add("\t\tthis." + prop.getPropertyName() + ".addAll(builder." + propGen.generateBuilderFieldName() + ");");
596                         } else {
597                             insertRegion.add("\t\tif (" + prop.getPropertyName() + " != null) {");
598                             insertRegion.add("\t\t\tthis." + prop.getPropertyName() + ".addAll(builder." + propGen.generateBuilderFieldName() + ");");
599                             insertRegion.add("\t\t}");
600                         }
601                     } else if (prop.isMapType()) {
602                         if (prop.isNotNull()) {
603                             insertRegion.add("\t\tthis." + prop.getPropertyName() + ".putAll(builder." + propGen.generateBuilderFieldName() + ");");
604                         } else {
605                             insertRegion.add("\t\tif (" + prop.getPropertyName() + " != null) {");
606                             insertRegion.add("\t\t\tthis." + prop.getPropertyName() + ".putAll(builder." + propGen.generateBuilderFieldName() + ");");
607                             insertRegion.add("\t\t}");
608                         }
609                     } else {
610                         insertRegion.add("\t\tthis." + prop.getPropertyName() + " = builder." + propGen.generateBuilderFieldName() + ";");
611                     }
612                 }
613             }
