34      /** The getter pattern. */
35      private static final Pattern GETTER_PATTERN = Pattern.compile(".*[ ,(]get[ ]*[=][ ]*[\"]([a-zA-Z-]*)[\"].*");
36      /** The setter pattern. */
37      private static final Pattern SETTER_PATTERN = Pattern.compile(".*[ ,(]set[ ]*[=][ ]*[\"]([ !#-~]*)[\"].*");

54      /**
55       * Constructor.
56       * @param bean  the bean generator
57       * @param content  the lines, not null
58       * @param lineIndex  the index of a PropertyDefinition
59       * @param derived  true if derived
60       */
61      public PropertyGen(BeanGen bean, List<String> content, int lineIndex, boolean derived) {
62          this.bean = bean;
63          this.propertyIndex = lineIndex;
64          this.annotationIndex = parseAnnotationStart(content, lineIndex);
65          this.fieldIndex = parseCodeIndex(content);
66          GeneratableProperty prop = new GeneratableProperty(bean.getData(), bean.getConfig());
67          if (derived) {
68              prop.setGetStyle("manual");
69              prop.setSetStyle("");
70              prop.setTypeStyle("");
71              prop.setDeprecated(parseDeprecated(content));
72              prop.setPropertyName(parseMethodNameAsPropertyName(content));
73              prop.setUpperName(makeUpperName(prop.getPropertyName()));
74              prop.setFieldType(parseMethodType(content));
75              prop.setInitializer(parseFieldInitializer(content));
76          } else {
77              prop.setGetStyle(parseGetStyle(content));
78              prop.setSetStyle(parseSetStyle(content));
79              prop.setTypeStyle(parseTypeStyle(content));
80              prop.setValidation(parseValidation(content));
81              prop.setDeprecated(parseDeprecated(content));
82              prop.setFieldName(parseFieldName(content));
83              prop.setPropertyName(makePropertyName(bean, prop.getFieldName()));
84              prop.setUpperName(makeUpperName(prop.getPropertyName()));
85              prop.setFinal(parseFinal(content));
86              prop.setFieldType(parseFieldType(content));
87              prop.setInitializer(parseFieldInitializer(content));
88          }
89          prop.resolveType();
90          prop.resolveGetterGen();
91          prop.resolveSetterGen();
92          prop.resolveCopyGen();
93          prop.resolveBuilderGen();
94          prop.resolveValidation();
95          List<String> comments = parseComment(content, prop.getPropertyName());
96          prop.setFirstComment(comments.get(0));
97          prop.getComments().addAll(comments.subList(1, comments.size()));
98          this.data = prop;
99      }

113     private int parseAnnotationStart(List<String> content, int lineIndex) {
114         while (lineIndex > 0 && content.get(lineIndex - 1).trim().startsWith("@")) {
115             lineIndex--;
116         }
117         return lineIndex;
118     }

120     private int parseCodeIndex(List<String> content) {
121         for (int index = propertyIndex; index < content.size(); index++) {
122             if (content.get(index).trim().startsWith("@") == false) {
123                 if (content.get(index).trim().length() == 0) {
124                     throw new RuntimeException("Unable to locate field for property at line " + (propertyIndex + 1) + ", found blank line");
125                 }
126                 return index;
127             }
128         }
129         throw new RuntimeException("Unable to locate field for property at line " + (propertyIndex + 1));
130     }

133         String line = content.get(propertyIndex).trim();
134         Matcher matcher = GETTER_PATTERN.matcher(line);
135         if (matcher.matches()) {

138         return "smart";

168     private boolean parseDeprecated(List<String> content) {
169         for (int index = annotationIndex; index < fieldIndex; index++) {
170             String line = content.get(index).trim();
171             if (line.equals("@Deprecated") || line.startsWith("@Deprecated ")) {
172                 return true;
173             }
174         }
175         return false;
176     }

178     //-----------------------------------------------------------------------
179     private String parseFieldName(List<String> content) {
180         String line = parseFieldDefinition(content);
181         String[] parts = line.split(" ");
182         String last = parts[parts.length - 1];
183         if (last.endsWith(";") && last.length() > 1) {
184             return last.substring(0, last.length() - 1);
185         }
186         throw new RuntimeException("Unable to locate field name at line " + (propertyIndex + 1));
187     }

189     private boolean parseFinal(List<String> content) {
190         String line = parseFieldDefinition(content);
191         String[] parts = line.split(" ");
192         if (parts.length < 2) {
193             throw new RuntimeException("Unable to locate field type at line " + (propertyIndex + 1));
194         }
195         if (parts[0].equals("final") || parts[1].equals("final") ||
196                 (parts.length >= 3 && parts[2].equals("final"))) {
197             return true;
198         }
199         return false;
200     }

202     private String parseFieldType(List<String> content) {
203         String line = parseFieldDefinition(content);
204         String[] parts = line.split(" ");
205         if (parts.length < 2) {
206             throw new RuntimeException("Unable to locate field type at line " + (propertyIndex + 1));
207         }
208         int partsPos = parts.length - 2;
209         String type = parts[partsPos];
210         while (true) {
211             int open = 0, openPos = 0, close = 0, closePos = 0;
212             while ((openPos = type.indexOf('<', openPos)) >= 0) {
213                 open++;
214                 openPos++;
215             }
216             while ((closePos = type.indexOf('>', closePos)) >= 0) {
217                 close++;
218                 closePos++;
219             }
220             if (open == close) {
221                 break;
222             }
223             if (partsPos == 0) {
224                 throw new RuntimeException("Unable to locate field type at line " + (propertyIndex + 1) + ", mismatched generics");
225             }
226             partsPos--;
227             type = parts[partsPos] + " " + type;
228         }
229         return type;
230     }

232     private String parseFieldDefinition(List<String> content) {
233         String line = content.get(fieldIndex).trim();
234         if (line.contains("//")) {
235             line = line.substring(0, line.indexOf("//")).trim();
236         }
237         if (line.contains(" = ")) {
238             line = line.substring(0, line.indexOf(" = ")).trim() + ";";
239         }
240         return line;
241     }

243     private String parseFieldInitializer(List<String> content) {
244         String line = content.get(fieldIndex).trim();
245         if (line.contains("//")) {
246             line = line.substring(0, line.indexOf("//")).trim();
247         }
248         if (line.contains(" = ")) {
249             line = line.substring(line.indexOf(" = ") + 3).trim();
250             if (line.endsWith(";") == false) {
251                 throw new RuntimeException("Field line does not end with semi-colon");
252             }
253             return line.substring(0, line.length() - 1).trim();
254         }
255         return "";
256     }

259     private String parseMethodNameAsPropertyName(List<String> content) {
260         String[] parts = parseMethodDefinition(content);
261         if (parts[1].length() == 0 || Character.isUpperCase(parts[1].charAt(0)) == false) {
262             throw new RuntimeException("@DerivedProperty method name invalid'");
263         }
264         return Character.toLowerCase(parts[1].charAt(0)) + parts[1].substring(1);
265     }

272     private String[] parseMethodDefinition(List<String> content) {
273         String line = content.get(fieldIndex).trim();
274         if (line.startsWith("public ")) {
275             line = line.substring(7).trim();
276         } else if (line.startsWith("private ")) {
277             line = line.substring(8).trim();
278         } else if (line.startsWith("protected ")) {
279             line = line.substring(10).trim();
280         }
281         String lineEnd = "() {";
282         if (line.startsWith("abstract ")) {
283             line = line.substring(9).trim();
284             lineEnd = "();";
285         } else if (line.startsWith("final ")) {
286             line = line.substring(6).trim();
287         } else if (line.startsWith("static ")) {
288             throw new RuntimeException("@DerivedProperty method cannot be static");
289         }
290         int getIndex = line.indexOf(" get");
291         if (getIndex < 0) {
292             throw new RuntimeException("@DerivedProperty method must start with 'get'");
293         }
294         if (line.endsWith(lineEnd) == false) {
295             throw new RuntimeException("@DerivedProperty method must end with '" + lineEnd + "'");
296         }
297         line = line.substring(0, line.length() - lineEnd.length());
298         String[] split = new String[2];
299         split[0] = line.substring(0, getIndex).trim();
300         split[1] = line.substring(getIndex + 4).trim();
301         return split;
302     }

305     private List<String> parseComment(List<String> content, String propertyName) {
306         List<String> comments = new ArrayList<String>();
307         String commentEnd = content.get(annotationIndex - 1).trim();
308         if (commentEnd.equals("*/")) {
309             int startCommentIndex = -1;
310             for (int index = annotationIndex - 1; index >= 0; index--) {
311                 String line = content.get(index).trim();
312                 if (line.equals("/**")) {
313                     startCommentIndex = index + 1;
314                     break;
315                 }
316             }
317             if (startCommentIndex == -1) {
318                 throw new RuntimeException("Unable to locate comment start at line " + annotationIndex);
319             }
320             if (startCommentIndex < annotationIndex - 1) {
321                 for (int i = startCommentIndex; i < annotationIndex - 1; i++) {
322                     String commentLine = content.get(i).trim();
323                     if (commentLine.startsWith("*")) {
324                         commentLine = commentLine.substring(1).trim();
325                     }
326                     if (commentLine.startsWith("@return") == false && commentLine.startsWith("@param") == false &&
327                             commentLine.startsWith("@throws") == false && commentLine.startsWith("@exception") == false) {
328                         comments.add(commentLine);
329                     }
330                 }
331                 String firstLine = comments.get(0);
332                 if (firstLine.length() > 0) {
333                     comments.set(0, firstLine.substring(0, 1).toLowerCase() + firstLine.substring(1));
334                 } else {
335                     comments.remove(0);
336                 }
337             }
338         } else if (commentEnd.startsWith("/**") && commentEnd.endsWith("*/")) {
339             int startPos = commentEnd.indexOf("/**") + 3;
340             int endPos = commentEnd.lastIndexOf("*/");
341             String comment = commentEnd.substring(startPos, endPos).trim();
342             if (comment.length() > 0) {
343                 comments.add(comment.substring(0, 1).toLowerCase() + comment.substring(1));
344             }
345         }
346         if (comments.size() == 0) {
347             comments.add("the " + propertyName + ".");
348         }
349         return comments;
350     }

358     List<String> generateMetaPropertyConstant() {
359         data.getBean().ensureImport(MetaProperty.class);
360         data.getBean().ensureImport(DirectMetaProperty.class);
361         List<String> list = new ArrayList<String>();
362         list.add("\t\t/**");
363         list.add("\t\t * The meta-property for the {@code " + data.getPropertyName() + "} property.");
364         list.add("\t\t */");
365         if (data.isBeanGenericType()) {
366             list.add("\t\t@SuppressWarnings({\"unchecked\", \"rawtypes\" })");
367             list.add("\t\tprivate final MetaProperty<" + propertyType() + "> " + metaFieldName() +
368                 " = (DirectMetaProperty) DirectMetaProperty.of" + readWrite() + "(");
369             list.add("\t\t\t\tthis, \"" + data.getPropertyName() + "\", " +
370                 data.getBean().getTypeRaw() + ".class, " + actualType() + ");");
371         } else {
372             String propertyType = propertyType();
373             if (propertyType.length() == 1) {
374                 propertyType = "Object";
375             }
376             if (data.isGenericParamType()) {
377                 list.add("\t\t@SuppressWarnings({\"unchecked\", \"rawtypes\" })");
378             }
379             list.add("\t\tprivate final MetaProperty<" + propertyType + "> " + metaFieldName() +
380                 " = DirectMetaProperty.of" + readWrite() + "(");
381             list.add("\t\t\t\tthis, \"" + data.getPropertyName() + "\", " +
382                 data.getBean().getTypeRaw() + ".class, " + actualType() + ");");
383         }
384         return list;
385     }

531     private String readWrite() {
532         switch (data.getStyle()) {
533             case READ_WRITE:
534                 return "ReadWrite";
535             case READ_ONLY:
536                 return "ReadOnly";
537             case WRITE_ONLY:
538                 return "WriteOnly";
539             case DERIVED:
540                 return "Derived";
541             case READ_ONLY_BUILDABLE:
542                 return "ReadOnlyBuildable";
543             case IMMUTABLE:
544                 return "Immutable";
545             default:
546                 break;
547         }
548         throw new RuntimeException("Invalid style");
549     }

551     private String actualType() {
552         String pt = propertyType();
553         if (pt.equals(data.getType())) {
554             int genericStart = pt.indexOf('<');
555             if (genericStart >= 0) {
556                 return "(Class) " + pt.substring(0, genericStart) + ".class";
557             }
558             if (data.getType().length() == 1) {
559                 return "Object.class";
560             }
561             if (data.isGenericArrayType()) {
562                 return "Object[].class";
563             }
564             return pt + ".class";
565         }
566         return pt + ".TYPE";
567     }

570         String pt = propertyType();

572             return "(" + pt + ") ";

581     private String propertyType(String type) {
582         if (type.equals("boolean")) {
583             return "Boolean";
584         }
585         if (type.equals("byte")) {
586             return "Byte";
587         }
588         if (type.equals("short")) {
589             return "Short";
590         }
591         if (type.equals("char")) {
592             return "Character";
593         }
594         if (type.equals("int")) {
595             return "Integer";
596         }
597         if (type.equals("long")) {
598             return "Long";
599         }
600         if (type.equals("float")) {
601             return "Float";
602         }
603         if (type.equals("double")) {
604             return "Double";
605         }
606         return type;
607     }
