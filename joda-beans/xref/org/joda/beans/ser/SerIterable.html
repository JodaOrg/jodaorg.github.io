16  package org.joda.beans.ser;
17  
18  import java.util.List;
19  
20  /**
21   * An abstraction over collections, lists, sets and maps.
22   * <p>
23   * This is a plugin point that can handle Guava collections.
24   *
25   * @author Stephen Colebourne
26   */
27  public interface SerIterable {
28  
29      /**
30       * Obtains an iterator over the data.
31       * 
32       * @return the iterator, not null
33       */
34      SerIterator iterator();
35  
36      /**
37       * Adds an item to the builder.
38       * 
39       * @param key  the key, such as for a map, null if no key
40       * @param value  the value, such as for a map or list value, may be null
41       * @param count  the count, such as for a multiset, typically one or greater
42       */
43      void add(Object key, Object value, int count);
44  
45      /**
46       * Builds the final collection.
47       * 
48       * @return the build collection, not null
49       */
50      Object build();
51  
52      /**
53       * Is the iterable map-like, as opposed to collection-like.
54       * 
55       * @return true if map-like, false if collection-like
56       */
57      boolean isMapLike();
58  
59      /**
60       * Gets the type of the key.
61       * 
62       * @return the key type, null if no key
63       */
64      Class<?> keyType();
65  
66      /**
67       * Gets the type of the value.
68       * 
69       * @return the value type, not null
70       */
71      Class<?> valueType();
72  
73      /**
74       * Gets the generic parameters of the value type.
75       * 
76       * @return the generic parameters of the value type, not null
77       */
78      List<Class<?>> valueTypeTypes();
79  
80  }
