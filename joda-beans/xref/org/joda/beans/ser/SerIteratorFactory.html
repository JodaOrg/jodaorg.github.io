59      /**
60       * Creates an iterator wrapper for an arbitrary value.
61       * 
62       * @param value  the possible collection-like object, not null
63       * @return the iterator, null if not a collection-like type
64       */
65      public SerIterator create(final Object value) {
66          if (value instanceof Collection) {
67              return collection((Collection<?>) value, Object.class);
68          }
69          if (value instanceof Map) {
70              return map((Map<?, ?>) value, Object.class, Object.class);
71          }
72          if (value.getClass().isArray() && value.getClass().getComponentType().isPrimitive() == false) {
73              Object[] array = (Object[]) value;
74              return array(array, array.getClass().getComponentType());
75          }
76          return null;
77      }

79      /**
80       * Creates an iterator wrapper for a meta-property value.
81       * 
82       * @param value  the possible collection-like object, not null
83       * @param prop  the meta-property defining the value, not null
84       * @param beanClass  the class of the bean, not the meta-property, for better generics, not null
85       * @return the iterator, null if not a collection-like type
86       */
87      public SerIterator create(final Object value, final MetaProperty<?> prop, Class<?> beanClass) {
88          if (value instanceof Collection) {
89              Class<?> valueType = JodaBeanUtils.collectionType(prop, beanClass);
90              return collection((Collection<?>) value, valueType);
91          }
92          if (value instanceof Map) {
93              Class<?> keyType = JodaBeanUtils.mapKeyType(prop, beanClass);
94              Class<?> valueType = JodaBeanUtils.mapValueType(prop, beanClass);
95              return map((Map<?, ?>) value, keyType, valueType);
96          }
97          if (value.getClass().isArray() && value.getClass().getComponentType().isPrimitive() == false) {
98              Object[] array = (Object[]) value;
99              return array(array, array.getClass().getComponentType());
100         }
101         return null;
102     }

105     /**
106      * Creates an iterator wrapper for a meta-property value.
107      * 
108      * @param metaTypeDescription  the description of the collection type, not null
109      * @param settings  the settings object, not null
110      * @return the iterator, null if not a collection-like type
111      */
112     public SerIterable createIterable(final String metaTypeDescription, final JodaBeanSer settings) {
113         if (metaTypeDescription.equals("Set")) {
114             return set(Object.class);
115         }
116         if (metaTypeDescription.equals("List")) {
117             return list(Object.class);
118         }
119         if (metaTypeDescription.equals("Collection")) {
120             return list(Object.class);
121         }
122         if (metaTypeDescription.equals("Map")) {
123             return map(Object.class, Object.class);
124         }
125         if (metaTypeDescription.equals("Object[]")) {
126             return array(Object.class);
127         }
128         if (metaTypeDescription.endsWith("[]")) {
129             String clsStr = metaTypeDescription.substring(0, metaTypeDescription.length() - 2);
130             try {
131                 Class<?> cls = settings.decodeClass(clsStr, null);
132                 return array(cls);
133             } catch (ClassNotFoundException ex) {
134                 throw new RuntimeException(ex);
135             }
136         }
137         return null;
138     }

140     /**
141      * Creates an iterator wrapper for a meta-property value.
142      * 
143      * @param prop  the meta-property defining the value, not null
144      * @param beanClass  the class of the bean, not the meta-property, for better generics, not null
145      * @return the iterator, null if not a collection-like type
146      */
147     public SerIterable createIterable(final MetaProperty<?> prop, final Class<?> beanClass) {
148         if (NavigableSet.class.isAssignableFrom(prop.propertyType())) {
149             Class<?> valueType = JodaBeanUtils.collectionType(prop, beanClass);
150             return navigableSet(valueType);
151         }
152         if (SortedSet.class.isAssignableFrom(prop.propertyType())) {
153             Class<?> valueType = JodaBeanUtils.collectionType(prop, beanClass);
154             return sortedSet(valueType);
155         }
156         if (Set.class.isAssignableFrom(prop.propertyType())) {
157             Class<?> valueType = JodaBeanUtils.collectionType(prop, beanClass);
158             return set(valueType);
159         }
160         if (Collection.class.isAssignableFrom(prop.propertyType())) {  // includes List
161             Class<?> valueType = JodaBeanUtils.collectionType(prop, beanClass);
162             return list(valueType);
163         }
164         if (NavigableMap.class.isAssignableFrom(prop.propertyType())) {
165             Class<?> keyType = JodaBeanUtils.mapKeyType(prop, beanClass);
166             Class<?> valueType = JodaBeanUtils.mapValueType(prop, beanClass);
167             return navigableMap(keyType, valueType);
168         }
169         if (SortedMap.class.isAssignableFrom(prop.propertyType())) {
170             Class<?> keyType = JodaBeanUtils.mapKeyType(prop, beanClass);
171             Class<?> valueType = JodaBeanUtils.mapValueType(prop, beanClass);
172             return sortedMap(keyType, valueType);
173         }
174         if (Map.class.isAssignableFrom(prop.propertyType())) {
175             Class<?> keyType = JodaBeanUtils.mapKeyType(prop, beanClass);
176             Class<?> valueType = JodaBeanUtils.mapValueType(prop, beanClass);
177             return map(keyType, valueType);
178         }
179         if (prop.propertyType().isArray() && prop.propertyType().getComponentType().isPrimitive() == false) {
180             return array(prop.propertyType().getComponentType());
181         }
182         return null;
183     }

208             @Override

213             public Class<?> keyType() {

293     @SuppressWarnings("rawtypes")
294     public static final SerIterator collection(final Collection<?> coll, final Class<?> valueType) {
295         return new SerIterator() {
296             private final Iterator it = coll.iterator();
297             private Object current;
298 
299             @Override
300             public String metaTypeName() {
301                 if (coll instanceof Set) {

303                 }
304                 if (coll instanceof List) {
305                     return "List";
306                 }
307                 return "Collection";
308             }
309             @Override
310             public int size() {
311                 return coll.size();
312             }
313             @Override
314             public boolean hasNext() {
315                 return it.hasNext();
316             }
317             @Override
318             public void next() {
319                 current = it.next();
320             }
321             @Override
322             public Class<?> keyType() {
323                 return null;
324             }
325             @Override
326             public Object key() {
327                 return null;
328             }
329             @Override
330             public int count() {
331                 return 1;
332             }
333             @Override
334             public Class<?> valueType() {
335                 return valueType;
336             }
337             @Override

470      * @return the iterable, not null
471      */
472     public static final SerIterable array(final Class<?> valueType) {
473         final List<Object> list = new ArrayList<Object>();
474         return new SerIterable() {
475             @Override
476             public SerIterator iterator() {
477                 return array(build(), valueType);
478             }
479             @Override
480             public void add(Object key, Object value, int count) {
481                 if (key != null) {
482                     throw new IllegalArgumentException("Unexpected key");
483                 }
484                 if (count != 1) {
485                     throw new IllegalArgumentException("Unexpected count");
486                 }
487                 for (int i = 0; i < count; i++) {
488                     list.add(value);
489                 }
490             }
491             @Override
492             public Object[] build() {
493                 Object[] array = (Object[]) Array.newInstance(valueType, list.size());
494                 return list.toArray(array);
495             }
496             @Override
497             public Class<?> keyType() {
498                 return null;
499             }
500             @Override
501             public Class<?> valueType() {
502                 return valueType;
503             }
504         };

507     /**
508      * Gets an iterator wrapper for an array.
509      * 
510      * @param array  the array, not null
511      * @param valueType  the value type, not null
512      * @return the iterator, not null
513      */
514     public static final SerIterator array(final Object[] array, final Class<?> valueType) {
515         return new SerIterator() {
516             private int index = -1;
517 
518             @Override
519             public String metaTypeName() {
520                 if (valueType == Object.class) {
521                     return "Object[]";
522                 }
523                 if (valueType == String.class) {
524                     return "String[]";
525                 }
526                 return valueType.getName() + "[]";
527             }
528             @Override
529             public int size() {
530                 return array.length;
531             }
532             @Override
533             public boolean hasNext() {
534                 return (index + 1) < array.length;
535             }
536             @Override
537             public void next() {
538                 index++;
539             }
540             @Override
541             public Class<?> keyType() {
542                 return null;
543             }
544             @Override
545             public Object key() {
546                 return null;
547             }
548             @Override
549             public int count() {
550                 return 1;
551             }
552             @Override
553             public Class<?> valueType() {
554                 return valueType;
555             }
556             @Override
557             public Object value() {
558                 return array[index];
559             }
560         };
