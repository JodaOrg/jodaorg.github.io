197     /**
198      * Parses the root bean.
199      * 
200      * @param rootType  the root type, not null
201      * @return the bean, not null
202      * @throws Exception if an error occurs
203      */
204     private <T> T read(final Class<T> rootType) throws Exception {
205         StartElement start = advanceToStartElement();
206         if (start.getName().equals(BEAN_QNAME) == false) {
207             throw new IllegalArgumentException("Expected root element 'bean' but found '" + start.getName() + "'");
208         }
209         Attribute attr = start.getAttributeByName(TYPE_QNAME);
210         if (attr == null && rootType == Bean.class) {
211             throw new IllegalArgumentException("Root element attribute must specify '" + TYPE + "'");
212         }
213         Class<?> effectiveType = rootType;
214         if (attr != null) {
215             String typeStr = attr.getValue();
216             effectiveType = SerTypeMapper.decodeType(typeStr, settings, null, knownTypes);
217             if (rootType.isAssignableFrom(effectiveType) == false) {
218                 throw new IllegalArgumentException("Specified root type is incompatible with XML root type: " + rootType.getName() + " and " + effectiveType.getName());
219             }
220         }
221         if (Bean.class.isAssignableFrom(effectiveType) == false) {
222             throw new IllegalArgumentException("Root type is not a Joda-Bean: " + effectiveType.getName());
223         }
224         basePackage = effectiveType.getPackage().getName() + ".";
225         Object parsed = parseBean(effectiveType);
226         return rootType.cast(parsed);
227     }

229     /**
230      * Parses a logical bean in the input XML.
231      * <p>
232      * Return type allows for a non-bean to be returned.
233      * 
234      * @param beanType  the bean type, not null
235      * @return the bean, not null
236      */
237     private Object parseBean(final Class<?> beanType) throws Exception {
238         String propName = "";
239         try {
240             XMLEvent event = null;
241             // handle case where whole bean is Joda-Convert string
242             if (settings.getConverter().isConvertible(beanType)) {
243                 StringBuilder buf = new StringBuilder();
244                 while (reader.hasNext()) {
245                     event = nextEvent(">btxt ");
246                     if (event.isCharacters()) {
247                         buf.append(event.asCharacters().getData());
248                     } else if (event.isEndElement()) {
249                         return settings.getConverter().convertFromString(beanType, buf.toString());
250                     } else if (event.isStartElement()) {
251                         break;  // not serialized via Joda-Convert
252                     } else if (event.isEndDocument()) {
253                         throw new IllegalArgumentException("Unexpected end of document");
254                     }
255                 }
256             } else {
257                 event = nextEvent(">bean ");
258             }
259             // handle structured bean
260             SerDeserializer deser = settings.getDeserializers().findDeserializer(beanType);
261             MetaBean metaBean = deser.findMetaBean(beanType);
262             BeanBuilder<?> builder = deser.createBuilder(beanType, metaBean);
263             // handle beans with structure
264             while (event.isEndElement() == false) {
265                 if (event.isStartElement()) {
266                     StartElement start = event.asStartElement();
267                     propName = start.getName().getLocalPart();
268                     MetaProperty<?> metaProp = deser.findMetaProperty(beanType, metaBean, propName);
269                     if (metaProp == null) {
270                         int depth = 0;
271                         event = nextEvent(" skip ");
272                         while (event.isEndElement() == false || depth > 0) {
273                             if (event.isStartElement()) {
274                                 depth++;
275                             } else if (event.isEndElement()) {
276                                 depth--;
277                             }
278                             event = nextEvent(" skip ");
279                         }
280                         // skip elements
281                     } else {
282                         Class<?> childType = parseTypeAttribute(start, metaProp.propertyType());
283                         Object value;
284                         if (Bean.class.isAssignableFrom(childType)) {
285                             value = parseBean(childType);
286                         } else {
287                             SerIterable iterable = SerIteratorFactory.INSTANCE.createIterable(metaProp, beanType);
288                             if (iterable != null) {
289                                 value = parseIterable(start, iterable);
290                             } else {
291                                 // metatype
292                                 Attribute metaTypeAttr = start.getAttributeByName(METATYPE_QNAME);
293                                 if (metaTypeAttr != null) {
294                                     iterable = SerIteratorFactory.INSTANCE.createIterable(metaTypeAttr.getValue(), settings, knownTypes);
295                                     if (iterable == null) {
296                                         throw new IllegalArgumentException("Invalid metaType");
297                                     }
298                                     value = parseIterable(start, iterable);
299                                 } else {
300                                     String text = advanceAndParseText();
301                                     value = settings.getConverter().convertFromString(childType, text);
302                                 }
303                             }
304                         }
305                         deser.setValue(builder, metaProp, value);
306                     }
307                     propName = "";
308                 }
309                 event = nextEvent(".bean ");
310             }
311             return deser.build(beanType, builder);
312         } catch (Exception ex) {
313             throw new RuntimeException("Error parsing bean: " + beanType.getName() + "::" + propName + ", " + ex.getMessage(), ex);
314         }
315     }

317     /**
318      * Parses to a collection wrapper.
319      * 
320      * @param iterable  the iterable builder, not null
321      * @return the iterable, not null
322      */
323     private Object parseIterable(final StartElement iterableEvent, final SerIterable iterable) throws Exception {
324         Attribute rowsAttr = iterableEvent.getAttributeByName(ROWS_QNAME);
325         Attribute columnsAttr = iterableEvent.getAttributeByName(COLS_QNAME);
326         if (rowsAttr != null && columnsAttr != null) {
327             iterable.dimensions(new int[] {Integer.parseInt(rowsAttr.getValue()), Integer.parseInt(columnsAttr.getValue())});
328         }
329         XMLEvent event = nextEvent(">iter ");
330         while (event.isEndElement() == false) {
331             if (event.isStartElement()) {
332                 StartElement start = event.asStartElement();
333                 QName expectedType = iterable.category() == SerCategory.MAP ? ENTRY_QNAME : ITEM_QNAME;
334                 if (start.getName().equals(expectedType) == false) {
335                     throw new IllegalArgumentException("Expected '" + expectedType.getLocalPart() + "' but found '" + start.getName() + "'");
336                 }
337                 int count = 1;
338                 Object key = null;
339                 Object column = null;
340                 Object value = null;
341                 if (iterable.category() == SerCategory.COUNTED) {
342                     Attribute countAttr = start.getAttributeByName(COUNT_QNAME);
343                     if (countAttr != null) {
344                         count = Integer.parseInt(countAttr.getValue());
345                     }
346                     value = parseValue(iterable, start);
347                     
348                 } else if (iterable.category() == SerCategory.TABLE || iterable.category() == SerCategory.GRID) {
349                     Attribute rowAttr = start.getAttributeByName(ROW_QNAME);
350                     Attribute colAttr = start.getAttributeByName(COL_QNAME);
351                     if (rowAttr == null || colAttr == null) {
352                         throw new IllegalArgumentException("Unable to read table as row/col attribute missing");
353                     }
354                     String rowStr = rowAttr.getValue();
355                     if (iterable.keyType() != null) {
356                         key = settings.getConverter().convertFromString(iterable.keyType(), rowStr);
357                     } else {
358                         key = rowStr;
359                     }
360                     String colStr = colAttr.getValue();
361                     if (iterable.columnType() != null) {
362                         column = settings.getConverter().convertFromString(iterable.columnType(), colStr);
363                     } else {
364                         column = colStr;
365                     }
366                     value = parseValue(iterable, start);
367                     
368                 } else if (iterable.category() == SerCategory.MAP) {
369                     Attribute keyAttr = start.getAttributeByName(KEY_QNAME);
370                     if (keyAttr != null) {
371                         // item is value with a key attribute
372                         String keyStr = keyAttr.getValue();
373                         if (iterable.keyType() != null) {
374                             key = settings.getConverter().convertFromString(iterable.keyType(), keyStr);
375                         } else {
376                             key = keyStr;
377                         }
378                         value = parseValue(iterable, start);
379                         
380                     } else {
381                         // two items nested in this entry
382                         event = nextEvent(">>map ");
383                         int loop = 0;
384                         while (event.isEndElement() == false) {
385                             if (event.isStartElement()) {
386                                 start = event.asStartElement();
387                                 if (start.getName().equals(ITEM_QNAME) == false) {
388                                     throw new IllegalArgumentException("Expected 'item' but found '" + start.getName() + "'");
389                                 }
390                                 if (key == null) {
391                                     key = parseKey(iterable, start);
392                                 } else {
393                                     value = parseValue(iterable, start);
394                                 }
395                                 loop++;
396                             }
397                             event = nextEvent("..map ");
398                         }
399                         if (loop != 2) {
400                             throw new IllegalArgumentException("Expected 2 'item's but found " + loop);
401                         }
402                     }                    
403                     
404                 } else {  // COLLECTION
405                     value = parseValue(iterable, start);
406                 }
407                 iterable.add(key, column, value, count);
408             }
409             event = nextEvent(".iter ");
410         }
411         return iterable.build();
412     }

414     private Object parseKey(final SerIterable iterable, StartElement start) throws Exception {
415         // type
416         Class<?> childType = parseTypeAttribute(start, iterable.keyType());
417         if (Bean.class.isAssignableFrom(childType) || settings.getConverter().isConvertible(childType)) {
418             return parseBean(childType);
419         } else {
420             throw new IllegalArgumentException("Unable to read map as parsed key type is neither a bean nor a simple type: " + childType.getName());
421         }
422     }

424     private Object parseValue(final SerIterable iterable, StartElement start) throws Exception {
425         // null
426         Object value;
427         Attribute nullAttr = start.getAttributeByName(NULL_QNAME);
428         if (nullAttr != null) {
429             if (nullAttr.getValue().equals("true") == false) {
430                 throw new IllegalArgumentException("Unexpected value for null attribute");
431             }
432             advanceAndParseText();  // move to end tag and ignore any text
433             value = null;
434         } else {
435             // type
436             Class<?> childType = parseTypeAttribute(start, iterable.valueType());
437             if (Bean.class.isAssignableFrom(childType)) {
438                 value = parseBean(childType);
439             } else {
440                 // try deep generic parameters
441                 SerIterable childIterable = SerIteratorFactory.INSTANCE.createIterable(iterable);
442                 if (childIterable != null) {
443                     value = parseIterable(start, childIterable);
444                 } else {
445                     // metatype
446                     Attribute metaTypeAttr = start.getAttributeByName(METATYPE_QNAME);
447                     if (metaTypeAttr != null) {
448                         childIterable = SerIteratorFactory.INSTANCE.createIterable(metaTypeAttr.getValue(), settings, knownTypes);
449                         if (childIterable == null) {
450                             throw new IllegalArgumentException("Invalid metaType");
451                         }
452                         value = parseIterable(start, childIterable);
453                     } else {
454                         String text = advanceAndParseText();
455                         value = settings.getConverter().convertFromString(childType, text);
456                     }
457                 }
458             }
459         }
460         return value;
461     }

464     private Class<?> parseTypeAttribute(final StartElement start, final Class<?> defaultType) throws ClassNotFoundException {
465         Attribute typeAttr = start.getAttributeByName(TYPE_QNAME);
466         if (typeAttr == null) {
467             return (defaultType == Object.class ? String.class : defaultType);
468         }
469         String childTypeStr = typeAttr.getValue();
470         return SerTypeMapper.decodeType(childTypeStr, settings, basePackage, knownTypes);
471     }

473     // reader can be anywhere, but normally at StartDocument
474     private StartElement advanceToStartElement() throws Exception {
475         while (reader.hasNext()) {
476             XMLEvent event = nextEvent("advnc ");
477             if (event.isStartElement()) {
478                 return event.asStartElement();
479             }
480         }
481         throw new IllegalArgumentException("Unexpected end of document");
482     }

484     // reader must be at StartElement
485     private String advanceAndParseText() throws Exception {
486         StringBuilder buf = new StringBuilder();
487         while (reader.hasNext()) {
488             XMLEvent event = nextEvent("text  ");
489             if (event.isCharacters()) {
490                 buf.append(event.asCharacters().getData());
491             } else if (event.isEndElement()) {
492                 return buf.toString();
493             } else if (event.isStartElement()) {
494                 throw new IllegalArgumentException("Unexpected start tag");
495             }
496         }
497         throw new IllegalArgumentException("Unexpected end of document");
498     }
