106     /**
107      * Reads and parses to a bean.
108      * 
109      * @param <T>  the root type
110      * @param input  the input stream, not null
111      * @param rootType  the root type, not null
112      * @return the bean, not null
113      */
114     public <T> T read(final InputStream input, Class<T> rootType) {
115         if (input instanceof DataInputStream) {
116             this.input = (DataInputStream) input;
117         } else {
118             this.input = new DataInputStream(input);
119         }
120         try {
121             try {
122                 return parseRoot(rootType);
123             } finally {
124                 input.close();
125             }
126         } catch (RuntimeException ex) {
127             throw ex;
128         } catch (Exception ex) {
129             throw new RuntimeException(ex);
130         }
131     }

134     /**
135      * Parses the root bean.
136      * 
137      * @param rootType  the root type, not null
138      * @return the bean, not null
139      * @throws Exception if an error occurs
140      */
141     private <T> T parseRoot(final Class<T> declaredType) throws Exception {
142         // root array
143         int typeByte = input.readByte();
144         if (typeByte != MIN_FIX_ARRAY + 2) {
145             throw new IllegalArgumentException("Invalid binary data: Expected array, but was: 0x" + toHex(typeByte));
146         }
147         // version
148         typeByte = input.readByte();
149         if (typeByte != 1) {
150             throw new IllegalArgumentException("Invalid binary data: Expected version 1, but was: 0x" + toHex(typeByte));
151         }
152         // parse
153         Object parsed = parseObject(declaredType, null, null, null, true);
154         return declaredType.cast(parsed);
155     }

157     private Object parseBean(int propertyCount, Class<?> beanType) throws Exception {
158         String propName = "";
159         try {
160             SerDeserializer deser = settings.getDeserializers().findDeserializer(beanType);
161             MetaBean metaBean = deser.findMetaBean(beanType);
162             BeanBuilder<?> builder = deser.createBuilder(beanType, metaBean);
163             for (int i = 0; i < propertyCount; i++) {
164                 // property name
165                 propName = acceptString(input.readByte());
166                 MetaProperty<?> metaProp = deser.findMetaProperty(beanType, metaBean, propName);
167                 if (metaProp == null) {
168                     MsgPackInput.skipObject(input);
169                 } else {
170                     Object value = parseObject(metaProp.propertyType(), metaProp, beanType, null, false);
171                     deser.setValue(builder, metaProp, value);
172                 }
173                 propName = "";
174             }
175             return deser.build(beanType, builder);
176         } catch (Exception ex) {
177             throw new RuntimeException("Error parsing bean: " + beanType.getName() + "::" + propName + ", " + ex.getMessage(), ex);
178         }
179     }

181     private Object parseObject(Class<?> declaredType, MetaProperty<?> metaProp, Class<?> beanType, SerIterable parentIterable, boolean rootType) throws Exception {
182         // establish type
183         Class<?> effectiveType = declaredType;
184         String metaType = null;
185         int typeByte = input.readByte();
186         if (isMap(typeByte)) {
187             input.mark(8);
188             int mapSize = acceptMap(typeByte);
189             if (mapSize > 0) {
190                 int typeByteTemp = input.readByte();
191                 if (typeByteTemp == EXT_8) {
192                     int size = input.readUnsignedByte();
193                     typeByteTemp = input.readByte();
194                     if (typeByteTemp == JODA_TYPE_BEAN) {
195                         String typeStr = acceptStringBytes(size);
196                         effectiveType = SerTypeMapper.decodeType(typeStr, settings, basePackage, knownTypes);
197                         if (rootType) {
198                             if (Bean.class.isAssignableFrom(effectiveType) == false) {
199                                 throw new IllegalArgumentException("Root type is not a Joda-Bean: " + effectiveType.getName());
200                             }
201                             basePackage = effectiveType.getPackage().getName() + ".";
202                         }
203                         if (declaredType.isAssignableFrom(effectiveType) == false) {
204                             throw new IllegalArgumentException("Specified type is incompatible with declared type: " + declaredType.getName() + " and " + effectiveType.getName());
205                         }
206                         if (input.readByte() != NIL) {
207                             throw new IllegalArgumentException("Invalid binary data: Expected null after bean type");
208                         }
209                         return parseBean(mapSize - 1, effectiveType);
210                     } else if (typeByteTemp == JODA_TYPE_DATA) {
211                         if (mapSize != 1) {
212                             throw new IllegalArgumentException("Invalid binary data: Expected map size 1, but was: " + mapSize);
213                         }
214                         String typeStr = acceptStringBytes(size);
215                         effectiveType = SerTypeMapper.decodeType(typeStr, settings, basePackage, knownTypes);
216                         if (declaredType.isAssignableFrom(effectiveType) == false) {
217                             throw new IllegalArgumentException("Specified type is incompatible with declared type: " + declaredType.getName() + " and " + effectiveType.getName());
218                         }
219                         typeByte = input.readByte();
220                     } else if (typeByteTemp == JODA_TYPE_META) {
221                         if (mapSize != 1) {
222                             throw new IllegalArgumentException("Invalid binary data: Expected map size 1, but was: " + mapSize);
223                         }
224                         metaType = acceptStringBytes(size);
225                         typeByte = input.readByte();
226                     } else {
227                         input.reset();
228                     }
229                 } else {
230                     input.reset();
231                 }
232             } else {
233                 input.reset();
234             }
235         }
236         // parse based on type
237         if (typeByte == NIL) {
238             return null;
239         }
240         if (Bean.class.isAssignableFrom(effectiveType)) {
241             if (isMap(typeByte)) {
242                 int mapSize = acceptMap(typeByte);
243                 return parseBean(mapSize, effectiveType);
244             } else {
245                 return parseSimple(typeByte, effectiveType);
246             }
247         } else {
248             if (isMap(typeByte) || isArray(typeByte)) {
249                 SerIterable childIterable = null;
250                 if (metaProp != null) {
251                     childIterable = SerIteratorFactory.INSTANCE.createIterable(metaProp, beanType);
252                 } else if (parentIterable != null) {
253                     childIterable = SerIteratorFactory.INSTANCE.createIterable(parentIterable);
254                     if (childIterable == null && metaType != null) {
255                         childIterable = SerIteratorFactory.INSTANCE.createIterable(metaType, settings, knownTypes);
256                     }
257                 }
258                 if (childIterable == null) {
259                     throw new IllegalArgumentException("Invalid binary data: Invalid metaType: " + metaType);
260                 }
261                 return parseIterable(typeByte, childIterable);
262             } else {
263                 return parseSimple(typeByte, effectiveType);
264             }
265         }
266     }

268     private Object parseIterable(int typeByte, SerIterable iterable) throws Exception {
269         if (iterable.category() == SerCategory.MAP) {
270             return parseIterableMap(typeByte, iterable);
271         } else if (iterable.category() == SerCategory.COUNTED) {
272             return parseIterableCounted(typeByte, iterable);
273         } else if (iterable.category() == SerCategory.TABLE) {
274             return parseIterableTable(typeByte, iterable);
275         } else {
276             return parseIterableArray(typeByte, iterable);
277         }
278     }

280     private Object parseIterableMap(int typeByte, SerIterable iterable) throws Exception {
281         int size = acceptMap(typeByte);
282         for (int i = 0; i < size; i++) {
283             Object key = parseObject(iterable.keyType(), null, null, null, false);
284             Object value = parseObject(iterable.valueType(), null, null, iterable, false);
285             iterable.add(key, null, value, 1);
286         }
287         return iterable.build();
288     }

290     private Object parseIterableTable(int typeByte, SerIterable iterable) throws Exception {
291         int size = acceptArray(typeByte);
292         for (int i = 0; i < size; i++) {
293             if (acceptArray(input.readByte()) != 3) {
294                 throw new IllegalArgumentException("Table must have cell array size 3");
295             }
296             Object key = parseObject(iterable.keyType(), null, null, null, false);
297             Object column = parseObject(iterable.columnType(), null, null, null, false);
298             Object value = parseObject(iterable.valueType(), null, null, iterable, false);
299             iterable.add(key, column, value, 1);
300         }
301         return iterable.build();
302     }

304     private Object parseIterableCounted(int typeByte, SerIterable iterable) throws Exception {
305         int size = acceptMap(typeByte);
306         for (int i = 0; i < size; i++) {
307             Object value = parseObject(iterable.valueType(), null, null, iterable, false);
308             int count = acceptInteger(input.readByte());
309             iterable.add(null, null, value, count);
310         }
311         return iterable.build();
312     }

314     private Object parseIterableArray(int typeByte, SerIterable iterable) throws Exception {
315         int size = acceptArray(typeByte);
316         for (int i = 0; i < size; i++) {
317             iterable.add(null, null, parseObject(iterable.valueType(), null, null, iterable, false), 1);
318         }
319         return iterable.build();

322     private Object parseSimple(int typeByte, Class<?> type) throws Exception {
323         if (isString(typeByte)) {
324             String text = acceptString(typeByte);
325             if (type == String.class || type == Object.class) {
326                 return text;
327             }
328             return settings.getConverter().convertFromString(type, text);
329         }
330         if (isIntegral(typeByte)) {
331             long value = acceptLong(typeByte);
332             if (type == Long.class || type == long.class) {
333                 return Long.valueOf(value);
334                 
335             } else if (type == Short.class || type == short.class) {
336                 if (value < Short.MIN_VALUE || value > Short.MAX_VALUE) {
337                     throw new IllegalArgumentException("Invalid binary data: Expected byte, but was " + value);
338                 }
339                 return Short.valueOf((short) value);
340                 
341             } else if (type == Byte.class || type == byte.class) {
342                 if (value < Byte.MIN_VALUE || value > Byte.MAX_VALUE) {
343                     throw new IllegalArgumentException("Invalid binary data: Expected byte, but was " + value);
344                 }
345                 return Byte.valueOf((byte) value);
346                 
347             } else {
348                 if (value < Integer.MIN_VALUE || value > Integer.MAX_VALUE) {
349                     throw new IllegalArgumentException("Invalid binary data: Expected int, but was " + value);
350                 }
351                 return Integer.valueOf((int) value);
352             }
353         }
354         switch (typeByte) {
355             case TRUE:
356                 return Boolean.TRUE;
357             case FALSE:
358                 return Boolean.FALSE;
359             case FLOAT_32:
360                 return Float.valueOf(input.readFloat());
361             case FLOAT_64:
362                 return Double.valueOf(input.readDouble());
363             case BIN_8:
364             case BIN_16:
365             case BIN_32:
366                 return acceptBinary(typeByte);
367         }
368         throw new IllegalArgumentException("Invalid binary data: Expected " + type.getName() + ", but was: 0x" + toHex(typeByte));
369     }

372     private int acceptMap(int typeByte) throws IOException {
373         int size;
374         if (typeByte >= MIN_FIX_MAP && typeByte <= MAX_FIX_MAP) {
375             size = (typeByte - MIN_FIX_MAP);
376         } else if (typeByte == MAP_16) {
377             size = input.readUnsignedShort();
378         } else if (typeByte == MAP_32) {
379             size = input.readInt();
380             if (size < 0) {
381                 throw new IllegalArgumentException("Invalid binary data: Map too large");
382             }
383         } else {
384             throw new IllegalArgumentException("Invalid binary data: Expected map, but was: 0x" + toHex(typeByte));
385         }
386         return size;
387     }

425     private String acceptStringBytes(int size) throws IOException {
426         byte[] bytes = new byte[size];
427         input.readFully(bytes);
428         // inline common ASCII case for much better performance
429         char[] chars = new char[size];
430         for (int i = 0; i < size; i++) {
431             byte b = bytes[i];
432             if (b >= 0) {
433                 chars[i] = (char) b;
434             } else {
435                 return new String(bytes, UTF_8);
436             }
437         }
438         return new String(chars);
439     }

441     private byte[] acceptBinary(int typeByte) throws IOException {
442         int size;
443         if (typeByte == BIN_8) {
444             size = input.readUnsignedByte();
445         } else if (typeByte == BIN_16) {
446             size = input.readUnsignedShort();
447         } else if (typeByte == BIN_32) {
448             size = input.readInt();
449             if (size < 0) {
450                 throw new IllegalArgumentException("Invalid binary data: Binary too large");

453             throw new IllegalArgumentException("Invalid binary data: Expected binary, but was: 0x" + toHex(typeByte));
454         }
455         byte[] bytes = new byte[size];
456         input.readFully(bytes);
457         return bytes;

460     private int acceptInteger(int typeByte) throws IOException {
461         if (typeByte >= MIN_FIX_INT && typeByte <= MAX_FIX_INT) {
462             return typeByte;
463         }
464         switch (typeByte) {
465             case UINT_8:
466                 return input.readUnsignedByte();
467             case UINT_16:
468                 return input.readUnsignedShort();
469             case UINT_32: {
470                 int val = input.readInt();
471                 if (val < 0) {
472                     throw new IllegalArgumentException("Invalid binary data: Expected int, but was large unsigned int");
473                 }
474                 return val;
475             }
476             case UINT_64: {
477                 long val = input.readLong();
478                 if (val < 0 || val > Integer.MAX_VALUE) {
479                     throw new IllegalArgumentException("Invalid binary data: Expected int, but was large unsigned int");
480                 }
481                 return (int) val;
482             }
483             case SINT_8:
484                 return input.readByte();
485             case SINT_16:
486                 return input.readShort();
487             case SINT_32:
488                 return input.readInt();
489             case SINT_64: {
490                 long val = input.readLong();
491                 if (val < Integer.MIN_VALUE || val > Integer.MAX_VALUE) {
492                     throw new IllegalArgumentException("Invalid binary data: Expected int, but was large signed int");
493                 }
494                 return (int) val;
495             }
496         }
497         throw new IllegalArgumentException("Invalid binary data: Expected int, but was: 0x" + toHex(typeByte));
498     }

500     private long acceptLong(int typeByte) throws IOException {
501         if (typeByte >= MIN_FIX_INT && typeByte <= MAX_FIX_INT) {
502             return typeByte;
503         }
504         switch (typeByte) {
505             case UINT_8:
506                 return input.readUnsignedByte();
507             case UINT_16:
508                 return input.readUnsignedShort();
509             case UINT_32: {
510                 return ((long) input.readInt()) & 0xFFFFFFFFL;
511             }
512             case UINT_64: {
513                 long val = input.readLong();
514                 if (val < 0) {
515                     throw new IllegalArgumentException("Invalid binary data: Expected long, but was large unsigned int");
516                 }
517                 return val;
518             }
519             case SINT_8:
520                 return input.readByte();
521             case SINT_16:
522                 return input.readShort();
523             case SINT_32:
524                 return input.readInt();
525             case SINT_64: {
526                 return input.readLong();
527             }
528         }
529         throw new IllegalArgumentException("Invalid binary data: Expected long, but was: 0x" + toHex(typeByte));
530     }
