18  import java.io.IOException;
19  import java.io.Reader;

21  /**
22   * Reader of JSON data.
23   */
24  final class JsonInput {

26      /** encoding JSON */
27      private static final String[] REPLACE = new String[128];
28      static {
29          for (int i = 0; i < 32; i++) {
30              REPLACE[i] = String.format("\\u%04x", i);
31          }
32          REPLACE['\b'] = "\\b";
33          REPLACE['\t'] = "\\t";
34          REPLACE['\n'] = "\\n";
35          REPLACE['\f'] = "\\f";
36          REPLACE['\r'] = "\\r";
37          REPLACE['"'] = "\\\"";
38          REPLACE['\\'] = "\\\\";
39          REPLACE[127] = "\\u007f";
40      }

43       * The reader.
44       */
45      private final Reader input;
46      /**
47       * The reused string buffer.
48       */
49      private final StringBuilder buf = new StringBuilder(32);
50      /**
51       * The last parsed integral number.

53      private long integral;

57      private double floating;

68       * Creates an instance that parses JSON.

72      JsonInput(Reader input) {

76      //-----------------------------------------------------------------------
77      /**
78       * Writes a JSON null.
79       * 
80       * @throws IOException if an error occurs
81       */
82      JsonEvent readEvent() throws IOException {
83          char next = readNext();
84          // whitespace
85          while (next == ' ' || next == '\t' || next == '\n' || next == '\r') {
86              next = readNext();
87          }
88          // identify token
89          switch (next) {
90              case '{':
91                  return JsonEvent.OBJECT;
92              case '}':
93                  return JsonEvent.OBJECT_END;
94              case '[':
95                  return JsonEvent.ARRAY;
96              case ']':
97                  return JsonEvent.ARRAY_END;
98              case '"':
99                  return JsonEvent.STRING;
100             case '-':
101             case '0':
102             case '1':
103             case '2':
104             case '3':
105             case '4':
106             case '5':
107             case '6':
108             case '7':
109             case '8':
110             case '9':
111                 return acceptNumber(next);
112             case 'n':
113                 return acceptNull();
114             case 't':
115                 return acceptTrue();
116             case 'f':
117                 return acceptFalse();
118             case ',':
119                 return JsonEvent.COMMA;
120             case ':':
121                 return JsonEvent.COLON;
122             default:
123                 throw new IllegalArgumentException("Invalid JSON data: Expected JSON character but found '" + next + "'");
124         }
125     }

127     // store peeked value for later use
128     void pushBack(char ch) throws IOException {
129         cachedNext = ch;
130     }

132     // store peeked value for later use
133     void pushBackObjectKey(String objectKey) throws IOException {

137     JsonEvent ensureEvent(JsonEvent event, JsonEvent expected) throws IOException {
138         if (event != expected) {
139             throw new IllegalArgumentException("Invalid JSON data: Expected " + expected + " but found " + event);
140         }
141         return event;
142     }

144     JsonEvent acceptEvent(JsonEvent expected) throws IOException {
145         return ensureEvent(readEvent(), expected);
146     }

148     //-----------------------------------------------------------------------
149     // expect object key and parse it
150     String acceptObjectKey(JsonEvent event) throws IOException {
151         ensureEvent(event, JsonEvent.STRING);
152         return parseObjectKey();
153     }

155     // opening quite already consumed
156     String parseObjectKey() throws IOException {
157         if (cachedObjectKey != null) {
158             String key = cachedObjectKey;
159             cachedObjectKey = null;
160             return key;
161         }
162         String str = parseString();
163         acceptEvent(JsonEvent.COLON);
164         return str;
165     }

167     //-----------------------------------------------------------------------
168     // expect string and parse it
169     String acceptString() throws IOException {
170         acceptEvent(JsonEvent.STRING);
171         return parseString();
172     }

174     // opening quite already consumed
175     String parseString() throws IOException {
176         buf.setLength(0);
177         char next = readNext();
178         while (next != '"') {
179             if (next == '\\') {
180                 parseEscape();
181             } else {
182                 buf.append(next);
183             }
184             next = readNext();
185         }
186         return buf.toString();
187     }
188 
189     private void parseEscape() throws IOException {
190         char next = readNext();
191         switch (next) {
192             case '"':
193                 buf.append('"');
194                 return;
195             case '/':
196                 buf.append('/');
197                 return;
198             case '\\':
199                 buf.append('\\');
200                 return;
201             case 'b':
202                 buf.append('\b');
203                 return;
204             case 'f':
205                 buf.append('\f');
206                 return;
207             case 'n':
208                 buf.append('\n');
209                 return;
210             case 'r':
211                 buf.append('\r');
212                 return;
213             case 't':
214                 buf.append('\t');
215                 return;
216             case 'u':
217                 int total = 0;
218                 for (int i = 0; i < 4; i++) {
219                     total = total * 16 + acceptHex();
220                 }
221                 buf.append((char) total);
222                 return;
223             default:
224                 throw new IllegalArgumentException("Invalid JSON data: Expected valid escape sequence but found '\\" + next + "'");
225         }
226     }
227 
228     private int acceptHex() throws IOException {
229         char next = readNext();
230         if (next >= '0' && next <= '9') {
231             return next - 48;
232         }
233         if (next >= 'a' && next <= 'f') {
234             return next - 97 + 10;
235         }
236         if (next >= 'A' && next <= 'F') {
237             return next - 65 + 10;
238         }
239         throw new IllegalArgumentException("Invalid JSON data: Expected hex but found '" + next + "'");
240     }

253     private JsonEvent acceptNumber(char first) throws IOException {
254         buf.setLength(0);
255         buf.append(first);
256         char last = first;
257         char next = readNext();
258         while ((next >= '0' && next <= '9') || next == '.' || next == '-' || next == '+' || next == 'e' || next == 'E') {
259             buf.append((char) next);
260             last = next;
261             next = readNext();
262         }
263         pushBack(next);
264         if (last < '0' || last > '9') {
265             throw new IllegalArgumentException("Invalid JSON data: Expected number but found invalid last char '" + last + "'");
266         }
267         String str = buf.toString();
268         if (str.equals("0")) {
269             integral = 0;
270             return JsonEvent.NUMBER_INTEGRAL;
271         } else if (str.startsWith("0") && str.charAt(1) != '.') {
272             throw new IllegalArgumentException("Invalid JSON data: Expected number but found zero at start");
273         } else if (str.contains(".") || str.contains("e") || str.contains("E")) {
274             floating = Double.parseDouble(str);
275             return JsonEvent.NUMBER_FLOATING;
276         } else {
277             integral = Long.parseLong(str);
278             return JsonEvent.NUMBER_INTEGRAL;
279         }
280     }

283     private JsonEvent acceptNull() throws IOException {
284         acceptChar('u');
285         acceptChar('l');
286         acceptChar('l');
287         return JsonEvent.NULL;
288     }

290     private JsonEvent acceptTrue() throws IOException {
291         acceptChar('r');
292         acceptChar('u');
293         acceptChar('e');
294         return JsonEvent.TRUE;
295     }
296 
297     private JsonEvent acceptFalse() throws IOException {
298         acceptChar('a');
299         acceptChar('l');
300         acceptChar('s');
301         acceptChar('e');
302         return JsonEvent.FALSE;
303     }

305     private void acceptChar(char ch) throws IOException {
306         char next = readNext();
307         if (next != ch) {
308             throw new IllegalArgumentException("Invalid JSON data: Expected '" + ch + "' but found '" + next + "'");
309         }
310     }
311 
312     //-----------------------------------------------------------------------
313     private char readNext() throws IOException {
314         if (cachedNext != null) {
315             char next = cachedNext.charValue();
316             cachedNext = null;
317             return next;
318         }
319         int next = input.read();
320         if (next == -1) {
321             throw new IllegalArgumentException("Invalid JSON data: End of file");
322         }
323         return (char) next;
324     }
325 
326     void skipData() throws IOException {
327         skipData(readEvent());
328     }
329 
330     private void skipData(JsonEvent event) throws IOException {
331         switch (event) {
332             case OBJECT:
333                 event = readEvent();
334                 while (event != JsonEvent.OBJECT_END) {
335                     acceptObjectKey(event);
336                     skipData();
337                     event = acceptObjectSeparator();
338                 }
339                 break;
340             case ARRAY:
341                 event = readEvent();
342                 while (event != JsonEvent.ARRAY_END) {
343                     skipData(event);
344                     event = acceptArraySeparator();
345                 }
346                 break;
347             case STRING:
348                 parseString();
349                 break;
350             case NULL:
351             case TRUE:
352             case FALSE:
353             case NUMBER_FLOATING:
354             case NUMBER_INTEGRAL:
355                 break;
356             default:
357                 throw new IllegalArgumentException("Invalid JSON data: Expected data item but found " + event);
358         }
359     }

361     //-----------------------------------------------------------------------
362     // accepts a comma or object end
363     JsonEvent acceptObjectSeparator() throws IOException {
364         JsonEvent event = readEvent();
365         if (event == JsonEvent.COMMA) {
366             return readEvent();  // leniently allow comma before objectEnd
367         } else {
368             return ensureEvent(event, JsonEvent.OBJECT_END);
369         }
370     }
371 
372     // accepts a comma or array end
373     JsonEvent acceptArraySeparator() throws IOException {
374         JsonEvent event = readEvent();
375         if (event == JsonEvent.COMMA) {
376             return readEvent();  // leniently allow comma before arrayEnd
377         } else {
378             return ensureEvent(event, JsonEvent.ARRAY_END);
379         }
380     }
