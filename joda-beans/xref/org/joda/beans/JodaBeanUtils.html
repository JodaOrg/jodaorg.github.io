62      /**
63       * Gets the meta-bean for a class.
64       * <p>
65       * This only works for those beans that have registered their meta-beans.
66       * See {@link #registerMetaBean(MetaBean)}.
67       * <p>
68       * A {@code Class} may use a static initializer block to call {@code registerMetaBean}.
69       * The edge case where the class is loaded but not initialized is handled
70       * by forcing the class to be initialized if necessary.
71       * 
72       * @param cls  the class to get the meta-bean for, not null
73       * @return the meta-bean, not null
74       * @throws IllegalArgumentException if unable to obtain the meta-bean
75       */
76      public static MetaBean metaBean(Class<?> cls) {
77          MetaBean meta = metaBeans.get(cls);
78          if (meta == null) {
79              // handle dynamic beans
80              if (cls == FlexiBean.class) {
81                  return new FlexiBean().metaBean();
82              } else if (cls == MapBean.class) {
83                  return new MapBean().metaBean();
84              } else if (DynamicBean.class.isAssignableFrom(cls)) {
85                  try {
86                      return cls.asSubclass(DynamicBean.class).newInstance().metaBean();
87                  } catch (InstantiationException ex) {
88                      throw new IllegalArgumentException("Unable to find meta-bean for a DynamicBean: " + cls.getName(), ex);
89                  } catch (IllegalAccessException ex) {
90                      throw new IllegalArgumentException("Unable to find meta-bean for a DynamicBean: " + cls.getName(), ex);
91                  }
92              }
93              // a Class can be loaded without being initialized
94              // in this state, the static initializers have not run, and thus the metabean not registered
95              // here initialization is forced to handle that scenario
96              try {
97                  cls = Class.forName(cls.getName(), true, cls.getClassLoader());
98              } catch (ClassNotFoundException ex) {
99                  // should be impossible
100                 throw new IllegalArgumentException("Unable to find meta-bean: " + cls.getName(), ex);
101             } catch (Error ex) {
102                 // should be impossible
103                 throw new IllegalArgumentException("Unable to find meta-bean: " + cls.getName(), ex);
104             }
105             meta = metaBeans.get(cls);
106             if (meta == null) {
107                 throw new IllegalArgumentException("Unable to find meta-bean: " + cls.getName());
108             }
109         }
110         return meta;
111     }

113     /**
114      * Registers a meta-bean.
115      * <p>
116      * This should be done for all beans in a static factory where possible.
117      * If the meta-bean is dynamic, this method should not be called.
118      * 
119      * @param metaBean  the meta-bean, not null
120      * @throws IllegalArgumentException if unable to register
121      */
122     public static void registerMetaBean(MetaBean metaBean) {
123         Class<? extends Bean> type = metaBean.beanType();
124         if (metaBeans.putIfAbsent(type, metaBean) != null) {
125             throw new IllegalArgumentException("Cannot register class twice: " + type.getName());
126         }
127     }

143     /**
144      * Checks if two objects are equal handling null.
145      * 
146      * @param obj1  the first object, may be null
147      * @param obj2  the second object, may be null
148      * @return true if equal
149      */
150     public static boolean equal(Object obj1, Object obj2) {
151         if (obj1 == obj2) {
152             return true;
153         }
154         if (obj1 == null || obj2 == null) {
155             return false;
156         }
157         if (obj1.getClass().isArray() && obj1.getClass() == obj2.getClass()) {
158             if (obj1 instanceof Object[] && obj2 instanceof Object[]) {
159                 return Arrays.deepEquals((Object[]) obj1, (Object[]) obj2);
160             } else if (obj1 instanceof int[] && obj2 instanceof int[]) {
161                 return Arrays.equals((int[]) obj1, (int[]) obj2);
162             } else if (obj1 instanceof long[] && obj2 instanceof long[]) {
163                 return Arrays.equals((long[]) obj1, (long[]) obj2);
164             } else if (obj1 instanceof byte[] && obj2 instanceof byte[]) {
165                 return Arrays.equals((byte[]) obj1, (byte[]) obj2);
166             } else if (obj1 instanceof double[] && obj2 instanceof double[]) {
167                 return Arrays.equals((double[]) obj1, (double[]) obj2);
168             } else if (obj1 instanceof float[] && obj2 instanceof float[]) {
169                 return Arrays.equals((float[]) obj1, (float[]) obj2);
170             } else if (obj1 instanceof char[] && obj2 instanceof char[]) {
171                 return Arrays.equals((char[]) obj1, (char[]) obj2);
172             } else if (obj1 instanceof short[] && obj2 instanceof short[]) {
173                 return Arrays.equals((short[]) obj1, (short[]) obj2);
174             } else if (obj1 instanceof boolean[] && obj2 instanceof boolean[]) {
175                 return Arrays.equals((boolean[]) obj1, (boolean[]) obj2);
176             }
177         }
178         return obj1.equals(obj2);
179     }

233                 return Arrays.hashCode((short[]) obj);
234             } else if (obj instanceof boolean[]) {
235                 return Arrays.hashCode((boolean[]) obj);
236             }

327     /**
328      * Checks if the two beans have the same set of properties.
329      * <p>
330      * This comparison checks that both beans have the same set of property names
331      * and that the value of each property name is also equal.
332      * It does not check the bean type, thus a {@link FlexiBean} may be equal
333      * to a {@link DirectBean}.
334      * <p>
335      * This comparison is usable with the {@link #propertiesHashCode} method.
336      * The result is the same as that if each bean was converted to a {@code Map}
337      * from name to value.
338      * 
339      * @param bean1  the first bean to compare, not null
340      * @param bean2  the second bean to compare, not null
341      * @return true if equal
342      */
343     public static boolean propertiesEqual(Bean bean1, Bean bean2) {
344         Set<String> names = bean1.propertyNames();
345         if (names.equals(bean2.propertyNames()) == false) {
346             return false;
347         }
348         for (String name : names) {
349             Object value1 = bean1.property(name).get();
350             Object value2 = bean2.property(name).get();
351             if (equal(value1, value2) == false) {
352                 return false;
353             }
354         }
355         return true;
356     }

358     /**
359      * Returns a hash code based on the set of properties on a bean.
360      * <p>
361      * This hash code is usable with the {@link #propertiesEqual} method.
362      * The result is the same as that if each bean was converted to a {@code Map}
363      * from name to value.
364      * 
365      * @param bean  the bean to generate a hash code for, not null
366      * @return the hash code
367      */
368     public static int propertiesHashCode(Bean bean) {
369         int hash = 7;
370         Set<String> names = bean.propertyNames();
371         for (String name : names) {
372             Object value = bean.property(name).get();
373             hash += hashCode(value);
374         }
375         return hash;
376     }

378     /**
379      * Returns a string describing the set of properties on a bean.
380      * <p>
381      * The result is the same as that if the bean was converted to a {@code Map}
382      * from name to value.
383      * 
384      * @param bean  the bean to generate a string for, not null
385      * @param prefix  the prefix to use, null ignored
386      * @return the string form of the bean, not null
387      */
388     public static String propertiesToString(Bean bean, String prefix) {
389         Set<String> names = bean.propertyNames();
390         StringBuilder buf = new StringBuilder((names.size()) * 32 + prefix.length());
391         if (prefix != null) {
392             buf.append(prefix);
393         }
394         buf.append('{');
395         if (names.size() > 0) {
396             for (String name : names) {
397                 Object value = bean.property(name).get();
398                 buf.append(name).append('=').append(value).append(',').append(' ');
399             }
400             buf.setLength(buf.length() - 2);
401         }
402         buf.append('}');
403         return buf.toString();
404     }

407     /**
408      * Clones a bean.
409      * <p>
410      * This performs a deep clone. There is no protection against cycles in
411      * the object graph beyond {@code StackOverflowError}.
412      * 
413      * @param <T>  the type of the bean
414      * @param original  the original bean to clone, null returns null
415      * @return the cloned bean, null if null input
416      */
417     @SuppressWarnings("unchecked")
418     public static <T extends Bean> T clone(T original) {
419         if (original == null || original instanceof ImmutableBean) {
420             return original;
421         }
422         BeanBuilder<? extends Bean> builder = original.metaBean().builder();
423         for (MetaProperty<?> mp : original.metaBean().metaPropertyIterable()) {
424             if (mp.style().isBuildable()) {
425                 Object value = mp.get(original);
426                 if (value instanceof Bean) {
427                     value = clone((Bean) value);
428                 }
429                 builder.set(mp.name(), value);
430             }
431         }
432         return (T) builder.build();
433     }

566     private static Class<?> extractTypeClass(Class<?> targetClass, MetaProperty<?> prop, int size, int index) {
567         return eraseToClass(extractType(targetClass, prop, size, index));
568     }
569 
570     private static Type extractType(Class<?> targetClass, MetaProperty<?> prop, int size, int index) {
571         Type genType = prop.propertyGenericType();
572         if (genType instanceof ParameterizedType) {
573             ParameterizedType pt = (ParameterizedType) genType;
574             Type[] types = pt.getActualTypeArguments();
575             if (types.length == size) {
576                 Type type = types[index];
577                 if (type instanceof TypeVariable) {
578                     type = resolveGenerics(targetClass, (TypeVariable<?>) type);
579                 }
580                 return type;
581             }
582         }
583         return null;
584     }
585 
586     private static List<Class<?>> extractTypeClasses(Class<?> targetClass, Type type) {
587         List<Class<?>> result = new ArrayList<Class<?>>();
588         if (type != null) {
589             if (type instanceof ParameterizedType) {
590                 ParameterizedType pt = (ParameterizedType) type;
591                 Type[] actualTypes = pt.getActualTypeArguments();
592                 for (Type actualType : actualTypes) {
593                     if (actualType instanceof TypeVariable) {
594                         actualType = resolveGenerics(targetClass, (TypeVariable<?>) actualType);
595                     }
596                     Class<?> cls = eraseToClass(actualType);
597                     result.add(cls != null ? cls : Object.class);
598                 }
599             }
600         }
601         return result;
602     }
603 
604     private static Type resolveGenerics(Class<?> targetClass, TypeVariable<?> typevar) {
605         // looks up meaning of type variables like T
606         Map<Type, Type> resolved = new HashMap<Type, Type>();
607         Type type = targetClass;
608         while (type != null) {
609             if (type instanceof Class) {
610                 type = ((Class<?>) type).getGenericSuperclass();
611             } else if (type instanceof ParameterizedType) {
612                 // find actual types captured by subclass
613                 ParameterizedType pt = (ParameterizedType) type;
614                 Type[] actualTypeArguments = pt.getActualTypeArguments();
615                 // find type variables declared in source code
616                 Class<?> rawType = eraseToClass(pt.getRawType());
617                 TypeVariable<?>[] typeParameters = rawType.getTypeParameters();
618                 for (int i = 0; i < actualTypeArguments.length; i++) {
619                     resolved.put(typeParameters[i], actualTypeArguments[i]);
620                 }
621                 type = rawType.getGenericSuperclass();
622             }
623         }
624         // resolve type variable to a meaningful type
625         Type result = typevar;
626         while (resolved.containsKey(result)) {
627             result = resolved.get(result);
628         }
629         return result;
630     }

632     private static Class<?> eraseToClass(Type type) {
633         if (type instanceof Class) {
634             return (Class<?>) type;
635         } else if (type instanceof ParameterizedType) {
636             return eraseToClass(((ParameterizedType) type).getRawType());
637         } else if (type instanceof GenericArrayType) {
638             Type componentType = ((GenericArrayType) type).getGenericComponentType();
639             Class<?> componentClass = eraseToClass(componentType);
640             if (componentClass != null) {
641                 return Array.newInstance(componentClass, 0).getClass();
642             }
643         } else if (type instanceof TypeVariable) {
644             Type[] bounds = ((TypeVariable<?>) type).getBounds();
645             if (bounds.length == 0) {
646                 return Object.class;
647             } else {
648                 return eraseToClass(bounds[0]);
649             }
650         }
651         return null;
652     }

655     /**
656      * Checks if two beans are equal ignoring one or more properties.
657      * <p>
658      * This version of {@code equalIgnoring} only checks properties at the top level.
659      * For example, if a {@code Person} bean contains an {@code Address} bean then
660      * only properties on the {@code Person} bean will be checked against the ignore list.
661      * 
662      * @param bean1  the first bean, not null
663      * @param bean2  the second bean, not null
664      * @param properties  the properties to ignore, not null
665      * @return true if equal
666      * @throws IllegalArgumentException if inputs are null
667      */
668     public static boolean equalIgnoring(Bean bean1, Bean bean2, MetaProperty<?>... properties) {
669         JodaBeanUtils.notNull(bean1, "bean1");
670         JodaBeanUtils.notNull(bean2, "bean2");
671         JodaBeanUtils.notNull(properties, "properties");
672         if (bean1 == bean2) {
673             return true;
674         }
