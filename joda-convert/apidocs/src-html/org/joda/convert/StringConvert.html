024/**
025 * Manager for conversion to and from a {@code String}, acting as the main client interface.
026 * <p>
027 * Support is provided for conversions based on the {@link StringConverter} interface
028 * or the {@link ToString} and {@link FromString} annotations.
029 * <p>
030 * StringConvert is thread-safe with concurrent caches.
031 */
032public final class StringConvert {
033
034    /**
035     * An immutable global instance.
036     * <p>
037     * This instance cannot be added to using {@link #register}, however annotated classes
038     * are picked up. To register your own converters, simply create an instance of this class.
039     */
040    public static final StringConvert INSTANCE = new StringConvert();
041    /**
042     * The cached null object.
043     */
044    private static final StringConverter<?> CACHED_NULL = new StringConverter<Object>() {
045        @Override
046        public String convertToString(Object object) {
047            return null;
048        }
049        @Override
050        public Object convertFromString(Class<? extends Object> cls, String str) {
051            return null;
052        }
053    };
054
055    /**
056     * The cache of converters.
057     */
058    private final ConcurrentMap<Class<?>, StringConverter<?>> registered = new ConcurrentHashMap<Class<?>, StringConverter<?>>();

065     * It is strongly recommended to only alter the converters before performing
066     * actual conversions.

145    /**
146     * Tries to register a class using the standard toString/parse pattern.
147     * 
148     * @param className  the class name, not null
149     */
150    private void tryRegister(String className, String fromStringMethodName) {
151        try {
152            Class<?> cls = getClass().getClassLoader().loadClass(className);
153            registerMethods(cls, "toString", fromStringMethodName);
154        } catch (Exception ex) {
155            // ignore
156        }
157    }

160    /**
161     * Converts the specified object to a {@code String}.
162     * <p>
163     * This uses {@link #findConverter} to provide the converter.
164     * 
165     * @param <T>  the type to convert from
166     * @param object  the object to convert, null returns null
167     * @return the converted string, may be null
168     * @throws RuntimeException (or subclass) if unable to convert
169     */
170    @SuppressWarnings("unchecked")
171    public <T> String convertToString(T object) {
172        if (object == null) {
173            return null;
174        }
175        Class<T> cls = (Class<T>) object.getClass();
176        StringConverter<T> conv = findConverter(cls);
177        return conv.convertToString(object);
178    }

181     * Converts the specified object to a {@code String}.
182     * <p>
183     * This uses {@link #findConverter} to provide the converter.
184     * The class can be provided to select a more specific converter.
185     * 
186     * @param <T>  the type to convert from
187     * @param cls  the class to convert from, not null
188     * @param object  the object to convert, null returns null
189     * @return the converted string, may be null
190     * @throws RuntimeException (or subclass) if unable to convert
191     */
192    public <T> String convertToString(Class<T> cls, T object) {
193        if (object == null) {
194            return null;
195        }
196        StringConverter<T> conv = findConverter(cls);
197        return conv.convertToString(object);
198    }

201     * Converts the specified object from a {@code String}.
202     * <p>
203     * This uses {@link #findConverter} to provide the converter.
204     * 
205     * @param <T>  the type to convert to
206     * @param cls  the class to convert to, not null
207     * @param str  the string to convert, null returns null
208     * @return the converted object, may be null
209     * @throws RuntimeException (or subclass) if unable to convert

211    public <T> T convertFromString(Class<T> cls, String str) {
212        if (str == null) {
213            return null;
214        }
215        StringConverter<T> conv = findConverter(cls);
216        return conv.convertFromString(cls, str);

219    /**
220     * Finds a suitable converter for the type.
221     * <p>
222     * This returns an instance of {@code StringConverter} for the specified class.
223     * This could be useful in other frameworks.
224     * <p>
225     * The search algorithm first searches the registered converters in the
226     * class hierarchy and immediate parent interfaces.
227     * It then searches for {@code ToString} and {@code FromString} annotations on the
228     * specified class, class hierarchy or immediate parent interfaces.
229     * 
230     * @param <T>  the type of the converter
231     * @param cls  the class to find a converter for, not null
232     * @return the converter, not null
233     * @throws RuntimeException (or subclass) if no converter found
234     */
235    @SuppressWarnings("unchecked")
236    public <T> StringConverter<T> findConverter(final Class<T> cls) {
237        if (cls == null) {
238            throw new IllegalArgumentException("Class must not be null");
239        }
240        StringConverter<T> conv = (StringConverter<T>) registered.get(cls);
241        if (conv == CACHED_NULL) {
242            throw new IllegalStateException("No registered converter found: " + cls);
243        }
244        if (conv == null) {
245            conv = findAnnotationConverter(cls);
246            if (conv == null) {
247                registered.putIfAbsent(cls, CACHED_NULL);
248                throw new IllegalStateException("No registered converter found: " + cls);
249            }
250            registered.putIfAbsent(cls, conv);
251        }
252        return conv;
253    }
254
255    @SuppressWarnings("unchecked")
256    private <T> StringConverter<T> findAnnotationConverter(final Class<T> cls) {
257        StringConverter<T> conv = null;
258        // check for registered on superclass
259        Class<?> loopCls = cls.getSuperclass();
260        while (loopCls != null && conv == null) {
261            conv = (StringConverter<T>) registered.get(loopCls);
262            if (conv != null && conv != CACHED_NULL) {
263                return conv;
264            }
265            loopCls = loopCls.getSuperclass();
266        }
267        // check for registered on interfaces
268        for (Class<?> loopIfc : cls.getInterfaces()) {
269            conv = (StringConverter<T>) registered.get(loopIfc);
270            if (conv != null && conv != CACHED_NULL) {
271                return conv;
272            }
273        }
274        // check for annotations
275        conv = findAnnotatedConverter(cls);
276        if (conv != null) {
277            return conv;
278        }
279        return null;
280    }

282    /**
283     * Finds the conversion method.
284     * 
285     * @param <T>  the type of the converter
286     * @param cls  the class to find a method for, not null
287     * @return the method to call, null means use {@code toString}
288     */
289    private <T> StringConverter<T> findAnnotatedConverter(final Class<T> cls) {
290        Method toString = findToStringMethod(cls);  // checks superclasses
291        if (toString == null) {
292            return null;
293        }
294        Constructor<T> con = findFromStringConstructor(cls);
295        Method fromString = findFromStringMethod(cls, con == null);  // optionally checks superclasses
296        if (con == null && fromString == null) {
297            throw new IllegalStateException("Class annotated with @ToString but not with @FromString: " + cls.getName());
298        }
299        if (con != null && fromString != null) {
300            throw new IllegalStateException("Both method and constructor are annotated with @FromString: " + cls.getName());
301        }
302        if (con != null) {
303            return new MethodConstructorStringConverter<T>(cls, toString, con);
304        } else {
305            return new MethodsStringConverter<T>(cls, toString, fromString);
306        }
307    }

309    /**
310     * Finds the conversion method.
311     * 
312     * @param cls  the class to find a method for, not null
313     * @return the method to call, null means use {@code toString}
314     */
315    private Method findToStringMethod(Class<?> cls) {
316        Method matched = null;
317        // find in superclass hierarchy
318        Class<?> loopCls = cls;
319        while (loopCls != null && matched == null) {
320            Method[] methods = loopCls.getDeclaredMethods();
321            for (Method method : methods) {
322                ToString toString = method.getAnnotation(ToString.class);
323                if (toString != null) {
324                    if (matched != null) {
325                        throw new IllegalStateException("Two methods are annotated with @ToString: " + cls.getName());
326                    }
327                    matched = method;
328                }
329            }
330            loopCls = loopCls.getSuperclass();
331        }
332        // find in immediate parent interfaces
333        if (matched == null) {
334            for (Class<?> loopIfc : cls.getInterfaces()) {
335                Method[] methods = loopIfc.getDeclaredMethods();
336                for (Method method : methods) {
337                    ToString toString = method.getAnnotation(ToString.class);
338                    if (toString != null) {
339                        if (matched != null) {
340                            throw new IllegalStateException("Two methods are annotated with @ToString on interfaces: " + cls.getName());
341                        }
342                        matched = method;
343                    }
344                }
345            }
346        }
347        return matched;
348    }

350    /**
351     * Finds the conversion method.
352     * 
353     * @param <T>  the type of the converter
354     * @param cls  the class to find a method for, not null
355     * @return the method to call, null means use {@code toString}
356     */
357    private <T> Constructor<T> findFromStringConstructor(Class<T> cls) {
358        Constructor<T> con;
359        try {
360            con = cls.getDeclaredConstructor(String.class);
361        } catch (NoSuchMethodException ex) {
362            try {
363                con = cls.getDeclaredConstructor(CharSequence.class);
364            } catch (NoSuchMethodException ex2) {
365                return null;
366            }
367        }
368        FromString fromString = con.getAnnotation(FromString.class);
369        return fromString != null ? con : null;
370    }

372    /**
373     * Finds the conversion method.
374     * 
375     * @param cls  the class to find a method for, not null
376     * @return the method to call, null means not found
377     */
378    private Method findFromStringMethod(Class<?> cls, boolean searchSuperclasses) {
379        Method matched = null;
380        // find in superclass hierarchy
381        Class<?> loopCls = cls;
382        while (loopCls != null && matched == null) {
383            matched = findFromString(loopCls, matched);
384            if (searchSuperclasses == false) {
385                break;
386            }
387            loopCls = loopCls.getSuperclass();
388        }
389        // find in immediate parent interfaces
390        if (searchSuperclasses && matched == null) {
391            for (Class<?> loopIfc : cls.getInterfaces()) {
392                matched = findFromString(loopIfc, matched);
393            }
394        }
395        return matched;
396    }

398    /**
399     * Finds the conversion method.
400     * 
401     * @param cls  the class to find a method for, not null
402     * @param matched  the matched method, may be null
403     * @return the method to call, null means not found
404     */
405    private Method findFromString(Class<?> cls, Method matched) {
406        // find in declared methods
407        Method[] methods = cls.getDeclaredMethods();
408        for (Method method : methods) {
409            FromString fromString = method.getAnnotation(FromString.class);
410            if (fromString != null) {
411                if (matched != null) {
412                    throw new IllegalStateException("Two methods are annotated with @FromString: " + cls.getName());
413                }
414                matched = method;
415            }
416        }
417        // check for factory
418        FromStringFactory factory = cls.getAnnotation(FromStringFactory.class);
419        if (factory != null) {
420            if (matched != null) {
421                throw new IllegalStateException("Class annotated with @FromString and @FromStringFactory: " + cls.getName());
422            }
423            Method[] factoryMethods = factory.factory().getDeclaredMethods();
424            for (Method method : factoryMethods) {
425                // handle factory containing multiple FromString for different types
426                if (cls.isAssignableFrom(method.getReturnType())) {
427                    FromString fromString = method.getAnnotation(FromString.class);
428                    if (fromString != null) {
429                        if (matched != null) {
430                            throw new IllegalStateException("Two methods are annotated with @FromString on the factory: " + factory.factory().getName());
431                        }
432                        matched = method;
433                    }
434                }
435            }
436        }
437        return matched;
438    }

441    /**
442     * Registers a converter for a specific type.
443     * <p>
444     * The converter will be used for subclasses unless overidden.
445     * <p>
446     * No new converters may be registered for the global singleton.
447     * 
448     * @param <T>  the type of the converter
449     * @param cls  the class to register a converter for, not null
450     * @param converter  the String converter, not null
451     * @throws IllegalArgumentException if the class or converter are null
452     * @throws IllegalStateException if trying to alter the global singleton
453     */
454    public <T> void register(final Class<T> cls, StringConverter<T> converter) {
455        if (cls == null) {
456            throw new IllegalArgumentException("Class must not be null");
457        }
458        if (converter == null) {
459            throw new IllegalArgumentException("StringConverter must not be null");
460        }
461        if (this == INSTANCE) {
462            throw new IllegalStateException("Global singleton cannot be extended");
463        }
464        registered.put(cls, converter);
465    }

467    /**
468     * Registers a converter for a specific type using two separate converters.
469     * <p>
470     * This method registers a converter for the specified class.
471     * It is primarily intended for use with JDK 1.8 method references or lambdas:
472     * <pre>
473     *  sc.register(Distance.class, Distance::toString, Distance::parse);
474     * </pre>
475     * The converter will be used for subclasses unless overidden.
476     * <p>
477     * No new converters may be registered for the global singleton.
478     * 
479     * @param <T>  the type of the converter
480     * @param cls  the class to register a converter for, not null
481     * @param toString  the to String converter, typically a method reference, not null
482     * @param fromString  the from String converter, typically a method reference, not null
483     * @throws IllegalArgumentException if the class or converter are null
484     * @throws IllegalStateException if trying to alter the global singleton
485     * @since 1.3
486     */
487    public <T> void register(final Class<T> cls, final ToStringConverter<T> toString, final FromStringConverter<T> fromString) {
488        if (fromString == null || toString == null) {
489            throw new IllegalArgumentException("Converters must not be null");
490        }
491        register(cls, new StringConverter<T>() {
492            public String convertToString(T object) {
493                return toString.convertToString(object);
494            }
495            public T convertFromString(Class<? extends T> cls, String str) {
496                return fromString.convertFromString(cls, str);
497            }
498        });
499    }

501    /**
502     * Registers a converter for a specific type by method names.
503     * <p>
504     * This method allows the converter to be used when the target class cannot have annotations added.
505     * The two method names must obey the same rules as defined by the annotations
506     * {@link ToString} and {@link FromString}.
507     * The converter will be used for subclasses unless overidden.
508     * <p>
509     * No new converters may be registered for the global singleton.
510     * <p>
511     * For example, {@code convert.registerMethods(Distance.class, "toString", "parse");}
512     * 
513     * @param <T>  the type of the converter
514     * @param cls  the class to register a converter for, not null
515     * @param toStringMethodName  the name of the method converting to a string, not null
516     * @param fromStringMethodName  the name of the method converting from a string, not null
517     * @throws IllegalArgumentException if the class or method name are null or invalid
518     * @throws IllegalStateException if trying to alter the global singleton
519     */
520    public <T> void registerMethods(final Class<T> cls, String toStringMethodName, String fromStringMethodName) {
521        if (cls == null) {
522            throw new IllegalArgumentException("Class must not be null");
523        }
524        if (toStringMethodName == null || fromStringMethodName == null) {
525            throw new IllegalArgumentException("Method names must not be null");
526        }
527        if (this == INSTANCE) {
528            throw new IllegalStateException("Global singleton cannot be extended");
529        }
530        Method toString = findToStringMethod(cls, toStringMethodName);
531        Method fromString = findFromStringMethod(cls, fromStringMethodName);
532        MethodsStringConverter<T> converter = new MethodsStringConverter<T>(cls, toString, fromString);
533        registered.putIfAbsent(cls, converter);
534    }

536    /**
537     * Registers a converter for a specific type by method and constructor.
538     * <p>
539     * This method allows the converter to be used when the target class cannot have annotations added.
540     * The two method name and constructor must obey the same rules as defined by the annotations
541     * {@link ToString} and {@link FromString}.
542     * The converter will be used for subclasses unless overidden.
543     * <p>
544     * No new converters may be registered for the global singleton.
545     * <p>
546     * For example, {@code convert.registerMethodConstructor(Distance.class, "toString");}
547     * 
548     * @param <T>  the type of the converter
549     * @param cls  the class to register a converter for, not null
550     * @param toStringMethodName  the name of the method converting to a string, not null
551     * @throws IllegalArgumentException if the class or method name are null or invalid
552     * @throws IllegalStateException if trying to alter the global singleton
553     */
554    public <T> void registerMethodConstructor(final Class<T> cls, String toStringMethodName) {
555        if (cls == null) {
556            throw new IllegalArgumentException("Class must not be null");
557        }
558        if (toStringMethodName == null) {
559            throw new IllegalArgumentException("Method name must not be null");
560        }
561        if (this == INSTANCE) {
562            throw new IllegalStateException("Global singleton cannot be extended");
563        }
564        Method toString = findToStringMethod(cls, toStringMethodName);
565        Constructor<T> fromString = findFromStringConstructorByType(cls);
566        MethodConstructorStringConverter<T> converter = new MethodConstructorStringConverter<T>(cls, toString, fromString);
567        registered.putIfAbsent(cls, converter);
568    }

570    /**
571     * Finds the conversion method.
572     * 
573     * @param cls  the class to find a method for, not null
574     * @param methodName  the name of the method to find, not null
575     * @return the method to call, null means use {@code toString}
576     */
577    private Method findToStringMethod(Class<?> cls, String methodName) {
578        Method m;
579        try {
580            m = cls.getMethod(methodName);
581        } catch (NoSuchMethodException ex) {
582            throw new IllegalArgumentException(ex);
583        }
584        if (Modifier.isStatic(m.getModifiers())) {
585            throw new IllegalArgumentException("Method must not be static: " + methodName);
586        }
587        return m;
588    }

590    /**
591     * Finds the conversion method.
592     * 
593     * @param cls  the class to find a method for, not null
594     * @param methodName  the name of the method to find, not null
595     * @return the method to call, null means use {@code toString}
596     */
597    private Method findFromStringMethod(Class<?> cls, String methodName) {
598        Method m;
599        try {
600            m = cls.getMethod(methodName, String.class);
601        } catch (NoSuchMethodException ex) {
602            try {
603                m = cls.getMethod(methodName, CharSequence.class);
604            } catch (NoSuchMethodException ex2) {
605                throw new IllegalArgumentException("Method not found", ex2);
606            }
607        }
608        if (Modifier.isStatic(m.getModifiers()) == false) {
609            throw new IllegalArgumentException("Method must be static: " + methodName);
610        }
611        return m;
612    }
