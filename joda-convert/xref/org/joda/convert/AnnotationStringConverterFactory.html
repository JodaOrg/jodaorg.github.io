16  package org.joda.convert;

18  import java.lang.reflect.Constructor;
19  import java.lang.reflect.Method;

21  /**
22   * Factory for {@code StringConverter} looking up annotations.
23   * <p>
24   * This class is immutable and thread-safe.
25   * 
26   * @since 1.5
27   */
28  final class AnnotationStringConverterFactory implements StringConverterFactory {
29  
30      /**
31       * Singleton instance.
32       */
33      static final StringConverterFactory INSTANCE = new AnnotationStringConverterFactory();
34  
35      /**
36       * Restricted constructor.
37       */
38      private AnnotationStringConverterFactory() {
39      }
40  
41      //-----------------------------------------------------------------------
42      /**
43       * Finds a converter by type.
44       * 
45       * @param cls  the type to lookup, not null
46       * @return the converter, null if not found
47       * @throws RuntimeException (or subclass) if source code is invalid
48       */
49      @Override
50      public StringConverter<?> findConverter(Class<?> cls) {
51          return findAnnotatedConverter(cls);  // capture generics
52      }
53  
54      /**
55       * Finds a converter searching annotated.
56       * 
57       * @param <T>  the type of the converter
58       * @param cls  the class to find a method for, not null
59       * @return the converter, not null
60       * @throws RuntimeException if none found
61       */
62      private <T> StringConverter<T> findAnnotatedConverter(final Class<T> cls) {
63          Method toString = findToStringMethod(cls);  // checks superclasses
64          if (toString == null) {
65              return null;
66          }
67          MethodConstructorStringConverter<T> con = findFromStringConstructor(cls, toString);
68          MethodsStringConverter<T> mth = findFromStringMethod(cls, toString, con == null);  // optionally checks superclasses
69          if (con == null && mth == null) {
70              throw new IllegalStateException("Class annotated with @ToString but not with @FromString: " + cls.getName());
71          }
72          if (con != null && mth != null) {
73              throw new IllegalStateException("Both method and constructor are annotated with @FromString: " + cls.getName());
74          }
75          return (con != null ? con : mth);
76      }
77  
78      /**
79       * Finds the conversion method.
80       * 
81       * @param cls  the class to find a method for, not null
82       * @return the method to call, null means use {@code toString}
83       * @throws RuntimeException if invalid
84       */
85      private Method findToStringMethod(Class<?> cls) {
86          Method matched = null;
87          // find in superclass hierarchy
88          Class<?> loopCls = cls;
89          while (loopCls != null && matched == null) {
90              Method[] methods = loopCls.getDeclaredMethods();
91              for (Method method : methods) {
92                  ToString toString = method.getAnnotation(ToString.class);
93                  if (toString != null) {
94                      if (matched != null) {
95                          throw new IllegalStateException("Two methods are annotated with @ToString: " + cls.getName());
96                      }
97                      matched = method;
98                  }
99              }
100             loopCls = loopCls.getSuperclass();
101         }
102         // find in immediate parent interfaces
103         if (matched == null) {
104             for (Class<?> loopIfc : eliminateEnumSubclass(cls).getInterfaces()) {
105                 Method[] methods = loopIfc.getDeclaredMethods();
106                 for (Method method : methods) {
107                     ToString toString = method.getAnnotation(ToString.class);
108                     if (toString != null) {
109                         if (matched != null) {
110                             throw new IllegalStateException("Two methods are annotated with @ToString on interfaces: " + cls.getName());
111                         }
112                         matched = method;
113                     }
114                 }
115             }
116         }
117         return matched;
118     }

120     /**
121      * Finds the conversion method.
122      * 
123      * @param <T>  the type of the converter
124      * @param cls  the class to find a method for, not null
125      * @param toString  the toString method, not null
126      * @return the method to call, null means none found
127      * @throws RuntimeException if invalid
128      */
129     private <T> MethodConstructorStringConverter<T> findFromStringConstructor(Class<T> cls, Method toString) {
130         Constructor<T> con;
131         try {
132             con = cls.getDeclaredConstructor(String.class);
133         } catch (NoSuchMethodException ex) {
134             try {
135                 con = cls.getDeclaredConstructor(CharSequence.class);
136             } catch (NoSuchMethodException ex2) {
137                 return null;
138             }
139         }
140         FromString fromString = con.getAnnotation(FromString.class);
141         if (fromString == null) {
142             return null;
143         }
144         return new MethodConstructorStringConverter<T>(cls, toString, con);
145     }

147     /**
148      * Finds the conversion method.
149      * 
150      * @param cls  the class to find a method for, not null
151      * @param toString  the toString method, not null
152      * @param searchSuperclasses  whether to search superclasses
153      * @return the method to call, null means not found
154      * @throws RuntimeException if invalid
155      */
156     private <T> MethodsStringConverter<T> findFromStringMethod(Class<T> cls, Method toString, boolean searchSuperclasses) {
157         // find in superclass hierarchy
158         Class<?> loopCls = cls;
159         while (loopCls != null) {
160             Method fromString = findFromString(loopCls);
161             if (fromString != null) {
162                 return new MethodsStringConverter<T>(cls, toString, fromString, loopCls);
163             }
164             if (searchSuperclasses == false) {
165                 break;
166             }
167             loopCls = loopCls.getSuperclass();
168         }
169         // find in immediate parent interfaces
170         MethodsStringConverter<T> matched = null;
171         if (searchSuperclasses) {
172             for (Class<?> loopIfc : eliminateEnumSubclass(cls).getInterfaces()) {
173                 Method fromString = findFromString(loopIfc);
174                 if (fromString != null) {
175                     if (matched != null) {
176                         throw new IllegalStateException("Two different interfaces are annotated with " +
177                             "@FromString or @FromStringFactory: " + cls.getName());
178                     }
179                     matched = new MethodsStringConverter<T>(cls, toString, fromString, loopIfc);
180                 }
181             }
182         }
183         return matched;
184     }

186     /**
187      * Finds the conversion method.
188      * 
189      * @param cls  the class to find a method for, not null
190      * @param matched  the matched method, may be null
191      * @return the method to call, null means not found
192      * @throws RuntimeException if invalid
193      */
194     private Method findFromString(Class<?> cls) {
195         // find in declared methods
196         Method[] methods = cls.getDeclaredMethods();
197         Method matched = null;
198         for (Method method : methods) {
199             FromString fromString = method.getAnnotation(FromString.class);
200             if (fromString != null) {
201                 if (matched != null) {
202                     throw new IllegalStateException("Two methods are annotated with @FromString: " + cls.getName());
203                 }
204                 matched = method;
205             }
206         }
207         // check for factory
208         FromStringFactory factory = cls.getAnnotation(FromStringFactory.class);
209         if (factory != null) {
210             if (matched != null) {
211                 throw new IllegalStateException("Class annotated with @FromString and @FromStringFactory: " + cls.getName());
212             }
213             Method[] factoryMethods = factory.factory().getDeclaredMethods();
214             for (Method method : factoryMethods) {
215                 // handle factory containing multiple FromString for different types
216                 if (cls.isAssignableFrom(method.getReturnType())) {
217                     FromString fromString = method.getAnnotation(FromString.class);
218                     if (fromString != null) {
219                         if (matched != null) {
220                             throw new IllegalStateException("Two methods are annotated with @FromString on the factory: " + factory.factory().getName());
221                         }
222                         matched = method;
223                     }
224                 }
225             }
226         }
227         return matched;
228     }

230     // eliminates enum subclass as they are pesky
231     private Class<?> eliminateEnumSubclass(Class<?> cls) {
232         Class<?> sup = cls.getSuperclass();
233         if (sup != null && sup.getSuperclass() == Enum.class) {
234             return sup;
235         }
236         return cls;
237     }

242         return getClass().getSimpleName();
243     }
